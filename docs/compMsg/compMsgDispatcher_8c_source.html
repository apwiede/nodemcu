83

88  return COMP_MSG_ERR_OUT_OF_MEMORY;
89  } else {
90  compMsgDispatcherHandles.handles[compMsgDispatcherHandles.numHandles].handle = handle;
91  compMsgDispatcherHandles.handles[compMsgDispatcherHandles.numHandles].compMsgDispatcher = self;
92  compMsgDispatcherHandles.numHandles++;
93  return COMP_MSG_ERR_OK;
94  }
95  } else {
96  // check for unused slot first
97  idx = 0;
98  while (idx < compMsgDispatcherHandles.numHandles) {
99  if (compMsgDispatcherHandles.handles[idx].handle == NULL) {
100  compMsgDispatcherHandles.handles[idx].handle = handle;
101  compMsgDispatcherHandles.handles[idx].compMsgDispatcher = self;
102  return COMP_MSG_ERR_OK;
103  }
104  idx++;
105  }
106  compMsgDispatcherHandles.handles = os_realloc(compMsgDispatcherHandles.handles, sizeof(handle2Dispatcher_t)*(compMsgDispatcherHandles.numHandles+1));
107  checkAllocOK(compMsgDispatcherHandles.handles);
108  compMsgDispatcherHandles.handles[compMsgDispatcherHandles.numHandles].handle = handle;
109  compMsgDispatcherHandles.handles[compMsgDispatcherHandles.numHandles].compMsgDispatcher = self;
110  compMsgDispatcherHandles.numHandles++;

141  numUsed++;

144  idx++;
145  }
146  if (numUsed == 0) {
147  os_free(compMsgDispatcherHandles.handles);
148  compMsgDispatcherHandles.handles = NULL;
149  }
150  if (found) {
151  return COMP_MSG_ERR_OK;
