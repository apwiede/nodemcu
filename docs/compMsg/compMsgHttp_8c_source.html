178  idx++;

180  }

223 static uint8_t getHttpHeaders(compMsgDispatcher_t *self, char * data, size_t size, socketUserData_t *sud) {
224  uint8_t result;
225  char *cp;
226  bool hadColon;
227  size_t idx;
228  size_t headerLgth;
229  size_t httpHeaderIdx;
230  httpHeaderPart_t *httpHeaderPart;
231  httpMsgInfo_t *httpMsgInfo;
232  uint8_t *headerStr;
233  uint8_t *lowerHeaderStr;
234  char *lowerData;
235  char *dp;
236 
237  idx = 0;
238  httpMsgInfo = &sud->httpMsgInfos[sud->numHttpMsgInfos];
239  httpHeaderIdx = 0;
240  cp = data;
241  hadColon = false;
242  lowerData = os_zalloc(size + 1);
243  dp = lowerData;
244  lowerHeaderStr = dp;
245  headerLgth = (char *)httpMsgInfo->content - data;
246  while (idx < headerLgth) {
247  if ((*cp == '\n')) {
248  if (httpHeaderIdx > 0) {
249  COMP_MSG_DBG(self, "H", 2, "dp: %d cp: %d", dp-(char *)lowerHeaderStr, cp-(char *)headerStr);
250  COMP_MSG_DBG(self, "H", 2, "id: %d name: %s value: %s!", httpHeaderPart->httpHeaderId, httpHeaderPart->httpHeaderName, httpHeaderPart->httpHeaderValue);
251  }
252  cp[-1] = '\0'; // ignore \r
253  cp++;
254  idx++;
255  httpHeaderIdx++;
256  hadColon = false;
257  httpHeaderPart = &httpMsgInfo->receivedHeaders[httpHeaderIdx];
258  dp = lowerData + (cp - data);
259  lowerHeaderStr = dp;
260  headerStr = cp;
261  COMP_MSG_DBG(self, "H", 2, "lh: %s!hs: %s!", lowerHeaderStr, headerStr);
262  }
263  if (httpHeaderIdx > 0) {
264  if (!hadColon && (*cp == ':')) {
265  hadColon = true;
266  *cp = '\0';
267  *dp = '\0';
268  result = sud->compMsgDispatcher->compMsgHttp->getHttpHeaderKeyIdFromLowerKey(sud->compMsgDispatcher, lowerHeaderStr, &httpHeaderPart->httpHeaderId);
269  checkErrOK(result);
270  httpHeaderPart->httpHeaderName = headerStr;
271  while (cp[1] == ' ') {
272  cp++;
273  idx++;
274  }
275  httpHeaderPart->httpHeaderValue = cp + 1;
276  } else {
277  if ((*cp >= 'A') && (*cp <= 'Z')) {
278  *dp = _tolower(*cp);
279  } else {
280  *dp = *cp;
281  }
282  }
283  } else {
284  *dp = *cp;
285  }
286  dp++;
287  cp++;
288  idx++;
289  }
290  return COMP_MSG_ERR_OK;
291 }

310

312  httpMsgInfo->data = os_zalloc(size + 1);
313  checkAllocOK(httpMsgInfo->data);
314  c_memcpy(httpMsgInfo->data, data, size);
315  result = getContentAndNumHeaders(self, httpMsgInfo->data, size, sud);
316  checkErrOK(result);
317 
318  COMP_MSG_DBG(self, "H", 2, "httpParse: numHeaders: %d sud->content: %s!len: %d!", httpMsgInfo->numHeaders, httpMsgInfo->content, httpMsgInfo->currLgth);
319  COMP_MSG_DBG(self, "H", 2, "receivedHeadersSize: %d", httpMsgInfo->numHeaders * sizeof(httpHeaderPart_t));
320  httpMsgInfo->receivedHeaders = os_zalloc(httpMsgInfo->numHeaders * sizeof(httpHeaderPart_t));
321  checkAllocOK(httpMsgInfo->receivedHeaders);
322  COMP_MSG_DBG(self, "H", 2, "httpMsgInfo->receivedHeaders: %p", httpMsgInfo->receivedHeaders);
323 
324  result = getHttpRequestCode(self, httpMsgInfo->data, sud);
325  checkErrOK(result);
326 
327  result = getHttpHeaders(self, httpMsgInfo->data, size, sud);
328  checkErrOK(result);
329 
330  idx = 0;
331  while (idx < httpMsgInfo->numHeaders) {
332  httpHeaderPart = &httpMsgInfo->receivedHeaders[idx];
333  COMP_MSG_DBG(self, "H", 1, "idx: %d id: %d name: %s value: %s!\n", idx, httpHeaderPart->httpHeaderId, httpHeaderPart->httpHeaderName, httpHeaderPart->httpHeaderValue);
334  if (httpHeaderPart->httpHeaderId == COMP_MSG_HTTP_CONTENT_LENGTH) {
335  if (httpMsgInfo->expectedLgth == 0) {
336  uval = c_strtoul(httpHeaderPart->httpHeaderValue, &endPtr, 10);
337  COMP_MSG_DBG(self, "H", 2, "expectedLgth: %d", uval);
338  httpMsgInfo->expectedLgth = (size_t)uval;
339  }
340  }
341  idx++;
342  }
343  COMP_MSG_DBG(self, "Y", 2, "content: %s lgth: %d", httpMsgInfo->content, httpMsgInfo->currLgth);
344  COMP_MSG_DBG(self, "Y", 2, "Code:: %d", httpMsgInfo->httpRequestCode);
345 // FIXME need to call addRequest here !!!
346  switch (httpMsgInfo->httpRequestCode) {
347  case 200:
348  COMP_MSG_DBG(self, "H", 1, "httpRequestCode: 200 OK");
349  //FIXME handle message here need code!!
350  msgLgth = httpMsgInfo->currLgth;
351  switch (sud->connectionType) {
352  case NET_SOCKET_TYPE_SOCKET:
353  result = self->compMsgUtil->fromBase64(self, httpMsgInfo->content, &msgLgth, &msgData);
354  checkErrOK(result);
355  self->compMsgData->currHdr->hdrHandleType = 'W';
356  COMP_MSG_DBG(self, "H", 1, "currHdr: %c", self->compMsgData->currHdr->hdrHandleType);
357  self->compMsgData->sud = sud;
358  self->compMsgData->direction = COMP_MSG_RECEIVED_DATA;
359  self->compMsgData->receivedData = msgData;
360  self->compMsgData->receivedLgth = msgLgth;
361  result = self->compMsgRequest->addRequest(self, COMP_DISP_INPUT_NET_SOCKET, sud, self->compMsgData);
362 // result = self->compMsgSendReceive->sendMsg(self, msgData, msgLgth);
363  COMP_MSG_DBG(self, "H", 1, "sendMsg: result: %d", result);
364  checkErrOK(result);
365  break;
366  }
367  break;
368  case 0:
369  COMP_MSG_DBG(self, "H", 1, "httpRequestCode: 0 OK");
370  msgLgth = httpMsgInfo->currLgth;
371  switch (sud->connectionType) {
372  case NET_SOCKET_TYPE_CLIENT:
373  self->compMsgData->currHdr->hdrHandleType = 'W';
374  msgData = httpMsgInfo->content;
375  COMP_MSG_DBG(self, "H", 1, "currHdr: %c", self->compMsgData->currHdr->hdrHandleType);
376  self->compMsgData->sud = sud;
377  self->compMsgData->direction = COMP_MSG_TRANSFER_DATA;
378  self->compMsgData->receivedData = msgData;
379  self->compMsgData->receivedLgth = msgLgth;

382  checkErrOK(result);
383  break;
384  }
385  break;
386  }
387  httpMsgInfo->expectedLgth = 0;
388  return COMP_MSG_ERR_OK;
389 }
