79 static uint8_t nextFittingEntry(compMsgDispatcher_t *self, msgParts_t *received, uint8_t u8CmdKey, uint16_t u16CmdKey) {
80  msgHeaderInfos_t *hdrInfos;
81  headerPart_t *hdr;
82  int hdrIdx;
83  int found;
84 
85  hdrInfos = &self->dispatcherCommon->msgHeaderInfos;
86  hdrIdx = hdrInfos->currPartIdx;
87  hdr = &hdrInfos->headerParts[hdrIdx];
88  COMP_MSG_DBG(self, "I", 2, "HEAD:!to:0x%04x!from:0x%04x!totalLgth:0x%04x!seqIdx!%d!\n", received->toPart, received->fromPart, received->totalLgth, hdrInfos->seqIdx);
89  // and now search in the headers to find the appropriate message
90  hdrInfos->seqIdx = hdrInfos->seqIdxAfterHeader;
91  found = 0;
92  while (hdrIdx < hdrInfos->numHeaderParts) {
93  hdr = &hdrInfos->headerParts[hdrIdx];
94  COMP_MSG_DBG(self, "I", 2, "hdrIdx: %d to: %d %d\n", hdrIdx, hdr->hdrToPart, received->toPart);
95  if (hdr->hdrToPart == received->toPart) {
96  COMP_MSG_DBG(self, "I", 2, "to: %d from: %d %d\n", hdr->hdrToPart, hdr->hdrFromPart, received->fromPart);
97  if (hdr->hdrFromPart == received->fromPart) {
98  COMP_MSG_DBG(self, "I", 2, "lgth: hdr: %d received: %d\n", hdr->hdrTotalLgth, received->totalLgth);
99  if ((hdr->hdrTotalLgth == received->totalLgth) || (hdr->hdrTotalLgth == 0)) {
100  COMP_MSG_DBG(self, "I", 2, "cmdKey: 0x%04x 0x%04x", u16CmdKey, received->u16CmdKey);
101  if (u16CmdKey != 0) {
102  if (u16CmdKey == received->u16CmdKey) {
103  found = 1;
104  break;
105  }
106  } else {
107  found = 1;
108  break;
109  }
110  }
111  }
112  }
113  hdrIdx++;
114  }
115  if (!found) {
116  COMP_MSG_DBG(self, "Y", 0, "nextFitting HEADER_NOT_FOUND\n");
117  return COMP_MSG_ERR_HEADER_NOT_FOUND;
118  }
119  hdrInfos->currPartIdx = hdrIdx;
120  COMP_MSG_DBG(self, "I", 2, "encryption: %c handleType: %c", hdr->hdrEncryption, hdr->hdrHandleType);
121  received->encryption = hdr->hdrEncryption;
122  COMP_MSG_DBG(self, "I", 2, "nextFitting!found!%d!hdrIdx!%d!hdr->cmdKey:0x%04x!received->cmdKey: 0x%04x", found, hdrIdx, hdr->hdrU16CmdKey, received->u16CmdKey);
123  return COMP_MSG_ERR_OK;
124 }

128 static uint8_t getHeaderIndexFromHeaderFields(compMsgDispatcher_t *self, msgParts_t *received) {
129  int result;
130  dataView_t *dataView;
131  headerPart_t *hdr;
132  int hdrIdx;
133  int found;
134  uint8_t myHeaderLgth;
135  uint16_t seqVal;
136  msgHeaderInfos_t *hdrInfos;
137  uint8_t *cp;
138  uint8_t lgth;
139 
140  hdrInfos = &self->dispatcherCommon->msgHeaderInfos;
141  COMP_MSG_DBG(self, "I", 2, "getHeaderIndexFromHeaderFields newCompMsgDataView");
142  received->compMsgDataView = newCompMsgDataView(received->buf, received->lgth);
143  checkAllocOK(received->compMsgDataView);
144  dataView = received->compMsgDataView->dataView;
145  COMP_MSG_DBG(self, "I", 2, "lgth: %d", received->lgth);
146  received->fieldOffset = 0;
147  myHeaderLgth = 0;
148  hdrInfos->seqIdx = 0;
149  seqVal = hdrInfos->headerSequence[hdrInfos->seqIdx];
150  while (seqVal != 0) {
151  switch(seqVal) {
152  case COMP_DISP_U16_DST:
153  result = dataView->getUint16(dataView, received->fieldOffset, &received->toPart);
154  checkErrOK(result);
155  COMP_MSG_DBG(self, "I", 2, "to: 0x%04x", received->toPart);
156  received->fieldOffset += sizeof(uint16_t);
157  break;
158  case COMP_DISP_U16_SRC:
159  result = dataView->getUint16(dataView, received->fieldOffset, &received->fromPart);
160  checkErrOK(result);
161  COMP_MSG_DBG(self, "I", 2, "from: 0x%04x", received->fromPart);
162  received->fieldOffset += sizeof(uint16_t);
163  break;
164  case COMP_DISP_U16_TOTAL_LGTH:
165  result = dataView->getUint16(dataView, received->fieldOffset, &received->totalLgth);
166  checkErrOK(result);
167  COMP_MSG_DBG(self, "I", 2, "total: 0x%04x", received->totalLgth);
168  received->fieldOffset += sizeof(uint16_t);
169  break;
170  case COMP_DISP_U16_SRC_ID:
171  result = dataView->getUint32(dataView, received->fieldOffset, &received->srcId);
172  checkErrOK(result);
173  COMP_MSG_DBG(self, "I", 2, "srcId: 0x%08x", received->srcId);
174  received->fieldOffset += sizeof(uint32_t);
175  break;
176  case COMP_DISP_HDR_RESERVE:
177  result = dataView->getUint16(dataView, received->fieldOffset, &received->hdrReserve);
178  checkErrOK(result);
179  COMP_MSG_DBG(self, "I", 2, "hdrReserve: 0x%04x", received->hdrReserve);
180  received->fieldOffset += sizeof(uint16_t);
181  break;
182  case COMP_DISP_U8_VECTOR_GUID:
183  cp = received->GUID;
184  result = dataView->getUint8Vector(dataView, received->fieldOffset, &cp, sizeof(received->GUID));
185  checkErrOK(result);
186  COMP_MSG_DBG(self, "I", 2, "GUID: %s", received->GUID);
187  received->fieldOffset += sizeof(received->GUID);
188  break;
189  case COMP_DISP_U8_VECTOR_HDR_FILLER:
190  cp = received->hdrFiller;
191  lgth = hdrInfos->headerLgth - received->fieldOffset;
192  result = dataView->getUint8Vector(dataView, received->fieldOffset, &cp, lgth);
193  checkErrOK(result);
194  received->fieldOffset += lgth;
195  break;
196  }
197  hdrInfos->seqIdx++;
198  seqVal = hdrInfos->headerSequence[hdrInfos->seqIdx];
199  }
200  COMP_MSG_DBG(self, "I", 2, "seqIdx: %d received->filedOffset: %d received->totalLgth: %d", hdrInfos->seqIdx, received->fieldOffset, received->totalLgth);
201  hdrInfos->seqIdxAfterHeader = hdrInfos->seqIdx;
202  hdrInfos->currPartIdx = 0;
203  result = nextFittingEntry(self, received, 0, 0);
204  COMP_MSG_DBG(self, "I", 2, "getHeaderIndexFromHeaderFields!result!%d!currPartIdx!%d!", result, hdrInfos->currPartIdx);
205  if (received->compMsgDataView != NULL) {
206  COMP_MSG_DBG(self, "I", 2, "os getHeaderIndexFromHeaderFields: free dataView: %p compMsgDataView: %p", received->compMsgDataView->dataView, received->compMsgDataView);
207  os_free(received->compMsgDataView->dataView);
208  os_free(received->compMsgDataView);
209  received->compMsgDataView = NULL;
210  }
211  return result;
212 }

216 static uint8_t prepareAnswerMsg(compMsgDispatcher_t *self, uint8_t type, uint8_t **handle) {
217  int result;
218  headerPart_t *hdr;

223  hdrInfos = &self->dispatcherCommon->msgHeaderInfos;
224  hdrIdx = hdrInfos->currPartIdx;
225  switch (type) {

237  checkErrOK(result);

243 static uint8_t handleReceivedHeader(compMsgDispatcher_t *self, msgParts_t *received) {
244  int result;
245  headerPart_t *hdr;
246  int hdrIdx;
247  msgHeaderInfos_t *hdrInfos;
248  compMsgField_t fieldInfo;
249  uint16_t u16;
250  uint8_t u8;
251  uint32_t sequenceEntry;
252  bool u8TotalCrc;
253  bool u16TotalCrc;
254  size_t startOffset;
255  size_t idx;
256 
257  COMP_MSG_DBG(self, "I", 2, "handleReceivedHeader");
258 //ets_printf("data fields3: %p\n", self->compMsgData->fields);
259  hdrInfos = &self->dispatcherCommon->msgHeaderInfos;
260 //ets_printf("handleReceivedHeader compMsgData->received1: %p lgth: %d\n", received->buf, received->lgth);
261  u8TotalCrc = false;
262  u16TotalCrc = false;
263  COMP_MSG_DBG(self, "I", 2, "handleReceivedHeader newCompMsgDataView");
264  received->compMsgDataView = newCompMsgDataView(received->buf, received->totalLgth);
265  checkAllocOK(received->compMsgDataView);
266 //received->compMsgDataView->dataView->dumpBinary(received->buf+30, 10, "Received->buf");
267  hdrIdx = hdrInfos->currPartIdx;
268  COMP_MSG_DBG(self, "I", 2, "handleReceivedHeader: currPartIdx: %d totalLgth: %d", hdrInfos->currPartIdx, received->totalLgth);
269  hdr = &hdrInfos->headerParts[hdrIdx];
270 
271  // set received->lgth to end of the header
272  received->lgth = hdrInfos->headerLgth;
273  // we loop over the fieldSequence entries and handle them as needed
274  // attention not all entries of the message are handled here, only some special entries!
275 
276 //self->compMsgMsgDesc->dumpHeaderPart(self, hdr);
277  // check if we have a U8_TOTAL_CRC or a U16_TOTAL_CRC or no TOTAL_CRC
278 //ets_printf("header1 heap: %d\n", system_get_free_heap_size());
279  idx = 0;
280  while (hdr->fieldSequence[idx] != 0) {
281  if (hdr->fieldSequence[idx] == COMP_DISP_U8_TOTAL_CRC) {
282  u8TotalCrc = true;
283  }
284  if (hdr->fieldSequence[idx] == COMP_DISP_U16_TOTAL_CRC) {
285  u16TotalCrc = true;
286  }
287  idx++;
288  }
289  while (hdr->fieldSequence[hdrInfos->seqIdx] != 0) {
290  sequenceEntry = hdr->fieldSequence[hdrInfos->seqIdx];
291  COMP_MSG_DBG(self, "I", 2, "sequenceEntry: 0x%04x seqIdx:%d", sequenceEntry, hdrInfos->seqIdx);
292  received->fieldOffset = hdrInfos->headerLgth;
293  switch (sequenceEntry) {
294  case COMP_DISP_U16_CMD_KEY:
295  result = received->compMsgDataView->dataView->getUint16(received->compMsgDataView->dataView, received->fieldOffset, &u16);
296  received->u16CmdKey = u16;
297  COMP_MSG_DBG(self, "I", 2, "1 u16CmdKey: 0x%04x!hdr: 0x%04x!offset: %d", received->u16CmdKey, hdr->hdrU16CmdKey, received->fieldOffset);
298  received->fieldOffset += 2;
299  received->partsFlags |= COMP_DISP_U16_CMD_KEY;
300  while (received->u16CmdKey != hdr->hdrU16CmdKey) {
301  hdrInfos->currPartIdx++;
302  result = self->compMsgIdentify->nextFittingEntry(self, received, 0, received->u16CmdKey);
303  checkErrOK(result);
304  hdr = &hdrInfos->headerParts[hdrInfos->currPartIdx];
305  COMP_MSG_DBG(self, "I", 2, "2 u16CmdKey: 0x%04x!hdr: 0x%04x", received->u16CmdKey, hdr->hdrU16CmdKey);
306  }
307  COMP_MSG_DBG(self, "I", 2, "found: currPartIdx: %d", hdrInfos->currPartIdx);
308  break;
309  case COMP_DISP_U0_CMD_LGTH:
310  received->fieldOffset += 2;
311  COMP_MSG_DBG(self, "I", 2, "u0CmdLgth!0!");
312  break;
313  case COMP_DISP_U16_CMD_LGTH:
314  result = received->compMsgDataView->dataView->getUint16(received->compMsgDataView->dataView, received->fieldOffset, &u16);
315  received->u16CmdLgth = u16;
316  received->fieldOffset += 2;
317  COMP_MSG_DBG(self, "I", 2, "u16CmdLgth!0x%04x!", received->u16CmdLgth);
318  break;
319  case COMP_DISP_U0_CRC:
320  COMP_MSG_DBG(self, "I", 2, "u0Crc!0!");
321  result = COMP_MSG_ERR_OK;
322  break;
323  case COMP_DISP_U8_CRC:
324  fieldInfo.fieldLgth = 1;
325  fieldInfo.fieldOffset = received->totalLgth - 1;
326  if (hdr->hdrFlags & COMP_DISP_TOTAL_CRC) {
327  if (u8TotalCrc) {
328  fieldInfo.fieldOffset -= 1;
329  } else {
330  fieldInfo.fieldOffset -= 2;
331  }
332  }
333  startOffset = hdrInfos->headerLgth;
334 //received->compMsgDataView->dataView->dumpBinaryWide(received->compMsgDataView->dataView->data, fieldInfo.fieldOffset, "beforegetcrc");
335  result = received->compMsgDataView->getCrc(self, received->compMsgDataView->dataView, &fieldInfo, startOffset, fieldInfo.fieldOffset);
336  COMP_MSG_DBG(self, "I", 2, "u8Crc!res!%d!", result);
337  break;
338  case COMP_DISP_U16_CRC:
339  fieldInfo.fieldLgth = 2;
340  fieldInfo.fieldOffset = received->totalLgth - 2;
341  if (hdr->hdrFlags & COMP_DISP_TOTAL_CRC) {
342  if (u8TotalCrc) {
343  fieldInfo.fieldOffset -= 1;
344  } else {
345  fieldInfo.fieldOffset -= 2;
346  }
347  }
348  startOffset = hdrInfos->headerLgth;
349  result = received->compMsgDataView->getCrc(self, received->compMsgDataView->dataView, &fieldInfo, startOffset, fieldInfo.fieldOffset);
350  COMP_MSG_DBG(self, "I", 2, "u16Crc!res!%d!", result);
351  break;
352  case COMP_DISP_U0_TOTAL_CRC:
353  COMP_MSG_DBG(self, "I", 2, "u0TotalCrc!0!");
354  result = COMP_MSG_ERR_OK;
355  break;
356  case COMP_DISP_U8_TOTAL_CRC:
357  // has to be done before decryption!!
358  result = COMP_MSG_ERR_OK;
359  break;
360  case COMP_DISP_U16_TOTAL_CRC:
361  // has to be done before decryption!!
362  result = COMP_MSG_ERR_OK;
363  break;
364  }
365  checkErrOK(result);
366  hdrInfos->seqIdx++;
367  }
368 //ets_printf("data fields4: %p\n", self->compMsgData->fields);
369 //ets_printf("header2 heap: %d\n", system_get_free_heap_size());
370  // free all space of received message
371  COMP_MSG_DBG(self, "I", 2, "call deleteMsg");
372 //ets_printf("data fields5: %p\n", self->compMsgData->fields);
373  self->compMsgData->deleteMsg(self);
374  COMP_MSG_DBG(self, "I", 2, "received msg deleted");
375 //ets_printf("handleReceivedHeader after deleteMsg heap: %d\n", system_get_free_heap_size());
376 //ets_printf("handleReceivedHeader compMsgData->received2: %p lgth: %d\n", received->buf, received->lgth);
377  return COMP_MSG_ERR_OK;
378 }
