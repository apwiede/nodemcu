121 static uint8_t readLine(compMsgMsgDesc_t *self, uint8_t **buffer, uint8_t *lgth) {
122  size_t n = BUFSIZ;
123  char buf[BUFSIZ];
124  int i;
125  uint8_t *cp;
126  uint8_t end_char = '\n';
127 
128  if (fileFd == (FS_OPEN_OK - 1)) {
129  return COMP_MSG_ERR_FILE_NOT_OPENED;
130  }
131  n = fs_read(fileFd, buf, n);
132  cp = *buffer;
133  *lgth = 0;
134  for (i = 0; i < n; ++i) {
135  cp[i] = buf[i];
136  if (buf[i] == end_char) {
137  ++i;
138  break;
139  }
140  }
141  cp[i] = 0;
142  *lgth = i;
143  fs_seek (fileFd, -(n - i), SEEK_CUR);
144  return COMP_MSG_ERR_OK;
145 }

244 static uint8_t handleMsgFileInternal(compMsgDispatcher_t *self, uint8_t *fileName, handleMsgLine_t handleMsgLine) {
245  int result;
246  uint8_t numEntries;
247  uint8_t lgth;
248  uint8_t buf[100];
249  uint8_t *buffer;
250  uint8_t *myStr;
251  int numLines;
252  int idx;
253  compMsgMsgDesc_t *compMsgMsgDesc;
254  msgDescIncludeInfo_t *msgDescIncludeInfo;
255 
256 ets_printf(">>>handleMsgFileInternal\n");
257  compMsgMsgDesc = self->compMsgMsgDesc;
258  buffer = buf;
259  result = compMsgMsgDesc->openFile(compMsgMsgDesc, fileName, "r");
260 //ets_printf("fileName: %s result: %d\n", fileName, result);
261  checkErrOK(result);
262 #undef checkErrOK
263 #define checkErrOK(result) if(result != COMP_MSG_ERR_OK) { compMsgMsgDesc->closeFile(compMsgMsgDesc); return result; }
264  numLines = 0;
265  while (1) {
266  result = compMsgMsgDesc->readLine(compMsgMsgDesc, &buffer, &lgth);
267  checkErrOK(result);
268  if (lgth == 0) {
269  if (numLines - 1 != self->compMsgMsgDesc->expectedLines) {
270 ets_printf("Error numLines: %d expectedLines: %d\n", numLines, self->compMsgMsgDesc->expectedLines);
271  return COMP_MSG_ERR_BAD_NUM_DESC_FILE_LINES;
272  }
273  break;
274  }
275  buffer[lgth] = 0;
276  result = compMsgMsgDesc->getLineFields(self, buffer, lgth);
277  checkErrOK(result);
278  // check if it is eventually an empty line and ignore that
279  if (compMsgMsgDesc->numLineFields < 2) {
280  if (c_strlen(compMsgMsgDesc->lineFields[0]) == 0) {
281  continue;
282  }
283  }
284  if ((buffer[0] == '#') && (buffer[1] == ' ')) {
285 ets_printf("comment\n");
286  // a comment line skip
287  continue;
288  }
289  // numLines is always without any comment lines!!
290  numLines++;
291  if (numLines == 1) {
292  // check if it is the number of lines line
293  if (c_strcmp(compMsgMsgDesc->lineFields[0], "#") == 0) {
294  result = compMsgMsgDesc->getIntFieldValue(self, 1, &self->compMsgMsgDesc->expectedLines);
295  checkErrOK(result);
296 ets_printf("expectedLines: %d\n", self->compMsgMsgDesc->expectedLines);
297  } else {
298  COMP_MSG_DBG(self, "E", 0, "wrong desc file number lines line");
299  return COMP_MSG_ERR_WRONG_DESC_FILE_LINE;
300  }
301  } else {
302  // check if it is a #use line
303  if (c_strcmp(compMsgMsgDesc->lineFields[0], "#use") == 0) {
304  if (compMsgMsgDesc->numLineFields != 2) {
305  COMP_MSG_DBG(self, "E", 0, "bad desc file #use line");
306  return COMP_MSG_ERR_BAD_DESC_FILE_USE_LINE;
307  } else {
308  result = compMsgMsgDesc->addUseFileName(self, compMsgMsgDesc->lineFields[1]);
309  checkErrOK(result);
310  }
311  } else {
312 //ets_printf("numLines: %d expectedLines: %d\n", numLines - 1, compMsgMsgDesc->expectedLines);
313  result = handleMsgLine(self);
314  checkErrOK(result);
315  }
316  }
317  }
318 #undef checkErrOK
319 #define checkErrOK(result) if(result != COMP_MSG_ERR_OK) { return result; }
320  result = compMsgMsgDesc->closeFile(compMsgMsgDesc);
321  checkErrOK(result);
322 ets_printf("<<<handleMsgFileInternal\n");
323  return result;
324 }
