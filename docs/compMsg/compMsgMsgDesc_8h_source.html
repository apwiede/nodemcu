43 /* ==========================================================================
44  * Layout of message description files:
45  * #< ><text> comment line (possible every where)
46  * #,<number lines>[,<action callback>] number of lines in this file
47  * eventually a callback for an action to call
48  * before building this message
49  * #use,<file name> use this file name (relative path)
50  * #include,<file name> include this file name (relative path)
51  * @<special field name>,<type>,<lgth> special fields
52  * @#<key/value field name>,<type>,<lgth>,<numValues>[,@<field size callback>]
53  * (see key/value fields description)
54  * <field name>,<type>,<lgth> normal field
55  *
56  * Example:
57  *
58  * Example MsgDescUse.txt
59  * # this is an example message use description file
60  * #,3
61  * #use,MsgDescHeader.txt
62  * #use,MsgDescMidPart.txt
63  * #use,MsgDescTrailer.txt
64  *
65  * Example MsgDescA1.txt
66  * # this is an example message description file
67  * #,7,@getAPList
68  * #include,MsgDescHeader.txt
69  * #include,MsgDescMidPart.txt
70  * @numKeyValues,unit8_t,1
71  * @#sequenceNum,uint32_t,4,0,@getSequenceNumSize
72  * @#ssid,uint8_t*,0,0,@getWifiAPSssidSize
73  * @#channel,uint8_t*,0,0,@getChannelSize
74  * #include,MsgDescTrailer.txt
75  *
76  * Example MsgDescHeader.txt
77  * # this is an example message header file
78  * @src,uint8_t,1
79  * @grp,uint8_t,1
80  * @dst,uint8_t,1
81  * @ipAddr,uint32_t,4
82  * @totalLgth,uint16_t,2
83  * @hdrFiller,uint8_t*,7
84  *
85  * Example MsgDescMidPart.txt
86  * # this is an example message mid part file
87  * @cmdKey,uint16_t,2
88  * @cmdLgth,uint16_t,2
89  *
90  * Example MsgDescTrailer.txt
91  * # this is an example message trailer file
92  * @filler,uint8_t*,0
93  * @crc,uint16_t,2
94  * @totalCrc,uint16_t,2
95  *
96  * ==========================================================================
97  */
98 
99 /* ==========================================================================
100  * Layout of message value files:
101  * #< ><text> comment line (possible every where)
102  * #,<number lines>[,<action callback>] number of lines in this file
103  * eventually a callback for an action to call
104  * before building this message
105  * #use,<file name> use this file name (relative path)
106  * #include,<file name> include this file name (relative path)
107  * @<special field name>,<valueType>,<value/callback>
108  * special fields
109  * @#<key/value field name>,<valueType>,<value/callback>
110  * (see key/value fields description)
111  * <field name>,<valueType>,<value/callback>
112  * normal field
113  *
114  * Example MsgValA1.txt
115  * # this is an example message value file
116  * #,5
117  * #include,MsgValHeader.txt
118  * @numKeyValues,I,3
119  * @#sequenceNum,I,@getClientSequenceNum
120  * @#ssid,S,@getWifiAPSsids
121  * @#channel,I,@getWifiAPChannels
122  *
123  * Example MsgValHeader.txt
124  * # this is an example message value header file
125  * #,5
126  * @src,I,12
127  * @grp,I,2
128  * @dst,I,15
129  * @srcId,I,@getSrcId
130  * @hdrReserve,S,@getHdrReserve
131  *
132  * ==========================================================================
133  */
