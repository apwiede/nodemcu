62 static uint8_t startNextRequest(compMsgDispatcher_t *self) {
63  uint8_t result;
64 
65  if (self->compMsgRequest->msgRequestInfos.currRequestIdx < 0) {
66  if (self->compMsgRequest->msgRequestInfos.currRequestIdx < self->compMsgRequest->msgRequestInfos.lastRequestIdx) {
67  self->compMsgRequest->msgRequestInfos.currRequestIdx = 0;
68  result = startRequest(self);
69  checkErrOK(result);
70  return COMP_MSG_ERR_OK;
71  }
72  }
73  if (self->compMsgRequest->msgRequestInfos.currRequestIdx < self->compMsgRequest->msgRequestInfos.lastRequestIdx) {
74  self->compMsgRequest->msgRequestInfos.currRequestIdx++;
75  result = startRequest(self);
76  checkErrOK(result);
77  }
78  return COMP_MSG_ERR_OK;

102 static uint8_t addRequest(compMsgDispatcher_t *self, uint8_t requestType, void *requestHandle, compMsgData_t *requestData) {
103  uint8_t result;
104  compMsgData_t *compMsgData;
105 
106  if (self->compMsgRequest->msgRequestInfos.lastRequestIdx >= COMP_DISP_MAX_REQUESTS) {
107  COMP_MSG_DBG(self, "Y", 0, "COMP_MSG_ERR_TOO_MANY_REQUESTS");
108  return COMP_MSG_ERR_TOO_MANY_REQUESTS;
109  }
110  self->compMsgRequest->msgRequestInfos.lastRequestIdx++;
111  self->compMsgRequest->msgRequestInfos.requestTypes[self->compMsgRequest->msgRequestInfos.lastRequestIdx] = requestType;
112  self->compMsgRequest->msgRequestInfos.requestHandles[self->compMsgRequest->msgRequestInfos.lastRequestIdx] = requestHandle;
113  self->compMsgRequest->msgRequestInfos.requestData[self->compMsgRequest->msgRequestInfos.lastRequestIdx] = requestData;
114  COMP_MSG_DBG(self, "R", 2, "addRequest: lastRequestIdx: %d requestType: %d compMsgData: %p\n", self->compMsgRequest->msgRequestInfos.lastRequestIdx, requestType, requestData);
115 //FIXME TEMPORARY last if clause!!
116  COMP_MSG_DBG(self, "R", 2, "addRequest 2 %d %p\n", self->compMsgRequest->msgRequestInfos.currRequestIdx, requestData);
117  if ((self->compMsgRequest->msgRequestInfos.currRequestIdx < 1) || (requestData->direction == COMP_MSG_TO_SEND_DATA)) {
118  self->compMsgRequest->msgRequestInfos.currRequestIdx++;
119  checkErrOK(result);
120  compMsgData = self->compMsgRequest->msgRequestInfos.requestData[self->compMsgRequest->msgRequestInfos.currRequestIdx];
121  switch (compMsgData->direction) {
122  case COMP_MSG_TO_SEND_DATA:
123  COMP_MSG_DBG(self, "R", 2, "addRequest: toSendData: %p %d\n", compMsgData->toSendData, compMsgData->toSendLgth);
124  result = self->compMsgIdentify->handleToSendPart(self, compMsgData->toSendData, compMsgData->toSendLgth);
125  break;
126  case COMP_MSG_RECEIVED_DATA:
127  COMP_MSG_DBG(self, "R", 2, "addRequest: receivedData: %p %d\n", compMsgData->receivedData, compMsgData->receivedLgth);
128  result = self->compMsgIdentify->handleReceivedPart(self, compMsgData->receivedData, compMsgData->receivedLgth);
129  break;
130  default:
131  COMP_MSG_DBG(self, "R", 2, "bad direction: 0x%02x 0x%02x\n", compMsgData->direction, requestData->direction);
132  return COMP_MSG_ERR_BAD_VALUE;
133  }
134  } else {
135  COMP_MSG_DBG(self, "R", 2, "direction: %d %d\n", requestData->direction, COMP_MSG_RECEIVED_DATA);
136  self->compMsgRequest->msgRequestInfos.currRequestIdx = self->compMsgRequest->msgRequestInfos.lastRequestIdx;
137  compMsgData = self->compMsgRequest->msgRequestInfos.requestData[self->compMsgRequest->msgRequestInfos.currRequestIdx];
138  requestData = compMsgData;
139  if (requestData->direction == COMP_MSG_RECEIVED_DATA) {
140 // FIXME TEMPORARY need flag to see if no uart activity!!
141  switch (compMsgData->direction) {
142  case COMP_MSG_TO_SEND_DATA:
143  break;
144  case COMP_MSG_RECEIVED_DATA:
145  COMP_MSG_DBG(self, "R", 2, "COMP_MSG_RECEIVED_DATA: compMsgData: %p\n", compMsgData);
146  COMP_MSG_DBG(self, "R", 2, "received: %p lgth: %d\n", compMsgData->receivedData, compMsgData->receivedLgth);
147  result = self->compMsgIdentify->handleReceivedPart(self, compMsgData->receivedData, compMsgData->receivedLgth);
148  break;
149  default:
150  COMP_MSG_DBG(self, "Y", 0, "bad direction: 0x%02x 0x%02x\n", compMsgData->direction, requestData->direction);
151  return COMP_MSG_ERR_BAD_VALUE;
152  }
153  }
154  }
155  return COMP_MSG_ERR_OK;
156 }

160 static uint8_t deleteRequest(compMsgDispatcher_t *self, uint8_t requestType, void *requestHandle) {
161  uint8_t result;
162  int idx;
163  bool found;
164  int idxToStart;
165  int idxDeleted;
166 
167  idx = 0;
168  idxToStart = -1;
169  idxDeleted = -1;
170  found = false;
171  while (idx < self->compMsgRequest->msgRequestInfos.lastRequestIdx) {
172  if (idx >= COMP_DISP_MAX_REQUESTS) {
173  return COMP_MSG_ERR_REQUEST_NOT_FOUND;
174  }
175  if (!found) {
176  if ((self->compMsgRequest->msgRequestInfos.requestTypes[idx] == requestType) && (self->compMsgRequest->msgRequestInfos.requestHandles[idx] == requestHandle)) {
177  found = true;
178  idxDeleted = idx;
179  if (idx == self->compMsgRequest->msgRequestInfos.currRequestIdx) {
180  if (idx < self->compMsgRequest->msgRequestInfos.lastRequestIdx) {
181  idxToStart = idx;
182  }
183  self->compMsgRequest->msgRequestInfos.currRequestIdx = -1;
184  }
185  }
186  } else {
187  // move the following entries one idx down
188  if (idx < self->compMsgRequest->msgRequestInfos.lastRequestIdx) {
189  self->compMsgRequest->msgRequestInfos.requestTypes[idx] = self->compMsgRequest->msgRequestInfos.requestTypes[idx + 1];
190  self->compMsgRequest->msgRequestInfos.requestHandles[idx] = self->compMsgRequest->msgRequestInfos.requestHandles[idx + 1];
191  if (idx + 1 == self->compMsgRequest->msgRequestInfos.currRequestIdx) {
192  self->compMsgRequest->msgRequestInfos.currRequestIdx--;
193  }
194  }
195  }
196  idx++;
197  }
198  if (self->compMsgRequest->msgRequestInfos.lastRequestIdx >= 0) {
199  self->compMsgRequest->msgRequestInfos.lastRequestIdx--;
200  }
201  if (self->compMsgRequest->msgRequestInfos.currRequestIdx < 0) {
202  self->compMsgRequest->msgRequestInfos.currRequestIdx++;
203  // start handling the request
204  self->compMsgRequest->startNextRequest(self);
205  } else {
206  // nothing to do the current request is different from the deleted one
207  // so just let the current one continue
208  }
209  return COMP_MSG_ERR_OK;
210 }
