121 /**
122  * \brief send message data to a web scoket
123  * \param sud The webSocket user data
124  * \param payload The message data
125  * \param size The number of characters of the message
126  * \param opcode The type for sending the data (text/binary/...)
127  * \return Error code or ErrorOK
128  *
129  */
130 static uint8_t webSocketSendData(socketUserData_t *sud, const char *payload, int size, int opcode)
131 {
132  uint8_t hdrBytes[4]; // we have either 2 or 4 bytes depending on length of message
133  int hdrLgth;
134  int fsize;
135  uint8_t*buff;
136  uint8_t result;
137  int i;
138  compMsgDispatcher_t *self;
139 
140  self = sud->compMsgDispatcher;
141  COMP_MSG_DBG(self, "W", 2, "webSocketSendData: size: %d\n", size);
142  hdrLgth = 2;
143  hdrBytes[0] = 0x80; //set first bit
144  hdrBytes[0] |= opcode; //frame->opcode; //set op code
145  if (size < 126) {
146  hdrBytes[1] = size;
147  } else {
148  if (size < SSL_BUFFER_SIZE - sizeof(hdrBytes)) {
149  hdrLgth += 2;
150  hdrBytes[1] = 126;
151  hdrBytes[2] = (size >> 8) & 0xFF;
152  hdrBytes[3] = size & 0xFF;
153  } else {
154  return WEBSOCKET_ERR_TOO_MUCH_DATA;
155  }
156  }
157  fsize = size + hdrLgth;
158  buff = os_malloc(fsize);
159  if (buff == NULL) {
160  return WEBSOCKET_ERR_OUT_OF_MEMORY;
161  }
162  for (i = 0; i < hdrLgth; i++) {
163  buff[i] = hdrBytes[i];
164  }
165 
166  os_memcpy(&buff[hdrLgth], (uint8_t *)payload, size);
167  result = espconn_sent(sud->pesp_conn, (unsigned char *)buff, fsize);
168  COMP_MSG_DBG(self, "W", 2, "espconn_sent: result: %d\n", result);
169  os_free(buff);
170  COMP_MSG_DBG(self, "W", 2, "espconn_sent: done\n");
171  return WEBSOCKET_ERR_OK;
172 }

176 static int ICACHE_FLASH_ATTR webSocketParse(char * data, size_t dataLenb, char **resData, int *len, socketUserData_t *sud) {
177  uint8_t byte = data[0];
178  uint8_t FIN = byte & 0x80;
179  uint8_t opcode = byte & 0x0F;
180  compMsgDispatcher_t *self;
181 
182  self = sud->compMsgDispatcher;
183  if ((opcode > 0x03 && opcode < 0x08) || opcode > 0x0B) {
184  ets_sprintf(err_opcode, "%d", opcode);
185  checkErrOK(WEBSOCKET_ERR_INVALID_FRAME_OPCODE);
186  }
187 
188  // opcodes: {1 text 2 binary 8 close 9 ping 10 pong}
189  switch (opcode) {
190  case OPCODE_TEXT:
191  COMP_MSG_DBG(self, "W", 2, "parse text\n");
192  break;
193  case OPCODE_BINARY:
194  COMP_MSG_DBG(self, "W", 2, "parse binary\n");
195  break;
196  case OPCODE_CLOSE:
197  break;
198  case OPCODE_PING:
199  webSocketSendData(sud, " ", 1, OPCODE_PONG);
200  return WEBSOCKET_ERR_OK;
201  break;
202  case OPCODE_PONG:
203  return WEBSOCKET_ERR_OK;
204  break;
205  }
206  byte = data[1];
207 
208  char * recv_data;
209  int masked = byte & 0x80;
210  int size = byte & 0x7F;
211  int offset = 2;
212 
213  if (size == 126) {
214  size = 0;
215  size = data[3]; //LSB
216  size |= (uint64_t) data[2] << 8; //MSB
217  offset = 4;
218  } else if (size == 127) {
219  size = 0;
220  size |= (uint64_t) data[2] << 56;
221  size |= (uint64_t) data[3] << 48;
222  size |= (uint64_t) data[4] << 40;
223  size |= (uint64_t) data[5] << 32;
224  size |= (uint64_t) data[6] << 24;
225  size |= (uint64_t) data[7] << 16;
226  size |= (uint64_t) data[8] << 8;
227  size |= (uint64_t) data[9];
228  offset = 10;
229  }
230 
231  if (masked) {
232  //read mask key
233  char mask[4];
234  uint64_t i;
235 
236  mask[0] = data[offset];
237  mask[1] = data[offset + 1];
238  mask[2] = data[offset + 2];
239  mask[3] = data[offset + 3];
240  offset += 4;
241  for (i = 0; i < size; i++) {
242  data[i + offset] ^= mask[i % 4];
243  }
244  }
245  recv_data = &data[offset];
246 //for (int i = 0; i < size; i++) {
247 // COMP_MSG_DBG(self, "W", 2, "i: %d 0x%02x\n", i, recv_data[i]&0xFF);
248 //}
249  switch (opcode) {
250  case OPCODE_TEXT:
251  COMP_MSG_DBG(self, "W", 2, "cb text\n");
252  sud->webSocketTextReceived(sud->compMsgDispatcher, sud, recv_data, size);
253  break;
254  case OPCODE_BINARY:
255  COMP_MSG_DBG(self, "W", 2, "cb binary\n");
256  COMP_MSG_DBG(self, "W", 2, "parse binary: size: %d\n", size);
257  sud->webSocketBinaryReceived(sud->compMsgDispatcher, sud, recv_data, size);
258  break;
259  }
260  *resData = recv_data;
261  *len = size;
262  recv_data[size] = 0;
263 
264  if (size + offset < dataLenb) {
265  webSocketParse(&data[size + offset], dataLenb - (size + offset), resData, len, sud);
266  }
267  return WEBSOCKET_ERR_OK;
268 }

283  if ((sud->curr_url != NULL) && (strstr(string, sud->curr_url) != NULL)) {

285  char *begin = strstr(string, header_key) + os_strlen(header_key);

288  checkAllocOK(key);

318  os_free(base64Digest);
319  struct espconn *pesp_conn = NULL;
320  pesp_conn = sud->pesp_conn;
321 
322  // FIXME!! reboot called if setting here for socket!!

324  sud->isWebsocket = 1;
325  }
326 
327  COMP_MSG_DBG(self, "W", 1, "payload: %d!%s!\n", payloadLen, payload);

329  os_free(key);
