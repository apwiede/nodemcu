252 static void bssScanDoneCb(void *arg, STATUS status) {
253  struct bss_info *bss_link;
254  uint8_t numEntries;
255  bssScanInfo_t *scanInfo;
256 
257 //COMP_MSG_DBG(self, "w", 1, "bssScanDoneCb bssScanRunning: arg: %p %d status: %d!\n", arg, bssScanRunning, status);
258  if (arg == NULL) {
259  return;
260  }
261  if (status != COMP_MSG_ERR_OK) {
262  return;
263  }
264  if (bssScanRunning == false) {
265  return;
266  }
267  if (bssScanRunning == true) {
268  bssScanRunning = false;
269  }
270 //COMP_MSG_DBG(self, "w", 1, "bssScanDoneCb bssScanRunning2: %d status: %d!\n", bssScanRunning, status);
271  numEntries = 0;
272  bss_link = (struct bss_info *)arg;
273  while (bss_link != NULL) {
274  numEntries++;
275  bss_link = bss_link->next.stqe_next;
276  }
277  if (bssScanInfos.infos != NULL) {
278  os_free(bssScanInfos.infos);
279  bssScanInfos.infos = NULL;
280  bssScanInfos.numScanInfos = 0;
281  bssScanInfos.maxScanInfos = 0;
282  }
283  bssScanInfos.maxScanInfos = numEntries;
284  bssScanInfos.infos = os_zalloc(bssScanInfos.maxScanInfos * sizeof(bssScanInfo_t));
285  if (bssScanInfos.infos == NULL) {
286  return;
287  }
288  bss_link = (struct bss_info *)arg;
289  compMsgWifiData.bssScanSizes.bssidSize = 0;
290  compMsgWifiData.bssScanSizes.bssidStrSize = 0;
291  compMsgWifiData.bssScanSizes.ssidSize = 0;
292  compMsgWifiData.bssScanSizes.channelSize = 0;
293  compMsgWifiData.bssScanSizes.rssiSize = 0;
294  compMsgWifiData.bssScanSizes.authmodeSize = 0;
295  compMsgWifiData.bssScanSizes.is_hiddenSize = 0;
296  compMsgWifiData.bssScanSizes.freq_offsetSize = 0;
297  compMsgWifiData.bssScanSizes.freqcal_valSize = 0;
298  while (bss_link != NULL) {
299  scanInfo = &bssScanInfos.infos[bssScanInfos.numScanInfos];
300  c_memset(scanInfo->ssid, '\t', sizeof(scanInfo->ssid));
301  if (bss_link->ssid_len <= sizeof(scanInfo->ssid)) {
302  c_memcpy(scanInfo->ssid, bss_link->ssid, bss_link->ssid_len);
303  compMsgWifiData.bssScanSizes.ssidSize += bss_link->ssid_len + 2;
304  } else {
305  c_memcpy(scanInfo->ssid, bss_link->ssid, sizeof(scanInfo->ssid));
306  compMsgWifiData.bssScanSizes.ssidSize += sizeof(scanInfo->ssid) + 2;
307  }
308  scanInfo->ssid_len = bss_link->ssid_len;
309  c_memset(scanInfo->bssidStr, '\t', sizeof(scanInfo->bssidStr));
310  c_memcpy(scanInfo->bssid, bss_link->bssid, sizeof(scanInfo->bssid));
311  compMsgWifiData.bssScanSizes.bssidSize += sizeof(scanInfo->bssid) + 2;
312  c_sprintf(scanInfo->bssidStr,MACSTR, MAC2STR(bss_link->bssid));
313  scanInfo->channel = bss_link->channel;
314  compMsgWifiData.bssScanSizes.channelSize += 1;
315  scanInfo->rssi = bss_link->rssi;
316  compMsgWifiData.bssScanSizes.rssiSize += 1;
317  scanInfo->authmode = bss_link->authmode;
318  compMsgWifiData.bssScanSizes.authmodeSize += 1;
319  scanInfo->is_hidden = bss_link->is_hidden;
320  compMsgWifiData.bssScanSizes.is_hiddenSize += 1;
321  scanInfo->freq_offset = bss_link->freq_offset;
322  compMsgWifiData.bssScanSizes.freq_offsetSize += 2;
323  scanInfo->freqcal_val = bss_link->freqcal_val;
324  compMsgWifiData.bssScanSizes.freqcal_valSize += 2;
325  bss_link = bss_link->next.stqe_next;
326  bssScanInfos.numScanInfos++;
327  }
328  bssScanInfos.scanInfoComplete = true;
329  bssScanInfos.compMsgDispatcher->compMsgBuildMsg->buildMsg(bssScanInfos.compMsgDispatcher);
330 }
