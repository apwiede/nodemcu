87 static uint8_t getAddressSizeAndMax(dwarfDbgPtr_t self, Dwarf_Half *size, Dwarf_Addr *max, Dwarf_Error *err) {
88  int dres = 0;
89  Dwarf_Half lsize = 4;
90  /* Get address size and largest representable address */
91  dres = dwarf_get_address_size(self->elfInfo.dbg, &lsize, err);
92  if (dres != DW_DLV_OK) {
93  printf("get_address_size() dres: %d err: %p", dres, *err);
94  return DWARF_DBG_ERR_CANNOT_GET_ADDR_SIZE;
95  }
96  if (max) {
97  *max = (lsize == 8 ) ? 0xffffffffffffffffULL : 0xffffffff;
98  }
99  if (size) {
100  *size = lsize;
101  }
102  return DWARF_DBG_ERR_OK;
103 }

128  if (self->dwarfDbgCompileUnitInfo->compileUnits == NULL) {
129  return DWARF_DBG_ERR_OUT_OF_MEMORY;

131  }
132  }

134  compileUnit = &self->dwarfDbgCompileUnitInfo->compileUnits[self->dwarfDbgCompileUnitInfo->numCompileUnit];

171  if (res != DW_DLV_OK) {

176  return DWARF_DBG_ERR_CANNOT_GET_NAME_FORMSTRING;
177  }
178 DWARF_DBG_PRINT(self, "C", 1, " >>name: %d %s\n", nameAttr, longFileName);
179  fileName = strrchr(longFileName,'/');
180  if (!fileName) {
181  fileName = strrchr(longFileName,'\\');
182  }
183  if (fileName) {
184  ++fileName;
185  } else {
186  fileName = longFileName;
187  }
188  shortFileName = fileName;
189  compileUnit->shortFileName = ckalloc(strlen(shortFileName) + 1);
190  memset(compileUnit->shortFileName, 0, strlen(shortFileName) + 1);
191  memcpy(compileUnit->shortFileName, shortFileName, strlen(shortFileName));
192  compileUnit->longFileName = ckalloc(strlen(longFileName) + 1);
193  memset(compileUnit->longFileName, 0, strlen(longFileName) + 1);
194  memcpy(compileUnit->longFileName, longFileName, strlen(longFileName));
195 DWARF_DBG_PRINT(self, "C", 1, " >>name2: %s %s!\n", compileUnit->shortFileName, compileUnit->longFileName);
196  return result;
197 }
