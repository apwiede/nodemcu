69 static uint32_t getDebugFlags(dwarfDbgPtr_t self, uint8_t *dbgChars) {
70  uint8_t *cp;
71  debugCh2Id_t *dp;
72  uint32_t flags;
73 
74  cp = dbgChars;
75  flags = 0;
76  while (*cp != '\0') {
77  dp = &debugCh2Id[0];
78  while (dp->ch != '\0') {
79  if (dp->ch == *cp) {
80  flags |= dp->id;
81  break;
82  }
83  dp++;
84  }
85  cp++;
86  }
87  flags &= self->dwarfDbgDebugInfo->currDebugFlags;
88  if (strstr(dbgChars, "Y") != NULL) {
89  flags |= DEBUG_DWARF_DBG_ALWAYS;
90  }
91  return flags;
92 }
