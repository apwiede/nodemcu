48 static uint8_t showDieEntries(dwarfDbgPtr_t self, int dieAndChildrenIdx, Dwarf_Bool isSibling, const char *indent) {
49  int result;
50  int entryIdx = 0;
51  int maxEntries = 0;
52  int attrIdx = 0;
53  int sres = 0;
54  const char *tagStringValue;
55  const char *attrStringValue;
56  const char *formStringValue;
57  dieAndChildrenInfo_t *dieAndChildrenInfo;
58  compileUnit_t *compileUnit;
59  dieInfo_t *dieInfo;
60  dieAttr_t *attrInfo;
61  Dwarf_Error err;
62  char *temps;
63  char buf[255];
64 
65  result = DWARF_DBG_ERR_OK;
66  compileUnit = self->dwarfDbgCompileUnitInfo->currCompileUnit;
67  dieAndChildrenInfo = &compileUnit->dieAndChildrenInfos[dieAndChildrenIdx];
68  if (isSibling) {
69  maxEntries = dieAndChildrenInfo->numSiblings;
70  } else {
71  maxEntries = dieAndChildrenInfo->numChildren;
72  }
73 fprintf(showFd, "++ numEntries: %d\n", maxEntries);
74  for(entryIdx = 0; entryIdx < maxEntries; entryIdx++) {
75  if (isSibling) {
76  dieInfo = &dieAndChildrenInfo->dieSiblings[entryIdx];
77  } else {
78  dieInfo = &dieAndChildrenInfo->dieChildren[entryIdx];
79  }
80 tagStringValue = NULL;
81  result = self->dwarfDbgStringInfo->getDW_TAG_string(self, dieInfo->tag, &tagStringValue);
82  checkErrOK(result);
83  fprintf(showFd, "%s%s: %04d offset: 0x%08x tag: 0x%04x numAttr: %d\n", indent, tagStringValue, entryIdx, dieInfo->offset, dieInfo->tag, dieInfo->numAttr);
84  for(attrIdx = 0; attrIdx < dieInfo->numAttr; attrIdx++) {
85  attrInfo = &dieInfo->dieAttrs[attrIdx];
86  attrStringValue = NULL;
87  result = self->dwarfDbgStringInfo->getDW_AT_string(self, attrInfo->attr, &attrStringValue);
88  checkErrOK(result);
89  result = self->dwarfDbgStringInfo->getDW_FORM_string(self, attrInfo->theform, &formStringValue);
90  checkErrOK(result);
91 sres = dwarf_formstring(attrInfo->attrIn, &temps, &err);
92 if (sres == DW_DLV_OK) {
93 sprintf(buf, "%s", temps);
94 }
95 switch (attrInfo->attr) {
96 case DW_AT_call_file:
97 case DW_AT_decl_file:
98 {
99  int fileIdx;
100  pathNameInfo_t *pathNameInfo;
101  dirNamesInfo_t *dirNamesInfo;
102  int compileUnitIdx = self->dwarfDbgCompileUnitInfo->currCompileUnitIdx;
103  int j;
104 
105  sres = DW_DLV_OK;
106  fileIdx = compileUnit->sourceFiles[attrInfo->uval];
107  if (fileIdx < self->dwarfDbgFileInfo->pathNamesInfo.numPathName) {
108  pathNameInfo = &self->dwarfDbgFileInfo->pathNamesInfo.pathNames[fileIdx];
109  dirNamesInfo = &self->dwarfDbgFileInfo->dirNamesInfo;
110  sprintf(buf, "%s/%s", dirNamesInfo->dirNames[pathNameInfo->dirNameIdx], pathNameInfo->fileName);
111  } else {
112  sres = DW_DLV_ERROR;
113  }
114 }
115  break;
116 case DW_AT_encoding:
117  formStringValue = NULL;
118  result = self->dwarfDbgStringInfo->getDW_ATE_string(self, attrInfo->uval, &formStringValue);
119  checkErrOK(result);
120  sres = DW_DLV_OK;
121  sprintf(buf, "%s", formStringValue);
122  break;
123 }
124 if (sres == DW_DLV_OK) {
125  fprintf(showFd, "%s %s: attr_in: 0x%08x theform: 0x%04x %s uval: 0x%08x refOffset: 0x%08x %s\n", indent, attrStringValue, attrInfo->attrIn, attrInfo->theform, formStringValue, attrInfo->uval, attrInfo->refOffset, buf);
126 } else {
127  fprintf(showFd, "%s %s: attr_in: 0x%08x theform: 0x%04x %s uval: 0x%08x refOffset: 0x%08x\n", indent, attrStringValue, attrInfo->attrIn, attrInfo->theform, formStringValue, attrInfo->uval, attrInfo->refOffset);
128 }
129  }
130  }
131  return result;
132 }

136 static uint8_t addAttrStr(dwarfDbgPtr_t self, const char *str, int *attrStrIdx) {
137  uint8_t result;
138  char **attrStr;
139  int strIdx;
140 
141  result = DWARF_DBG_ERR_OK;
142  for(strIdx = 0; strIdx < self->dwarfDbgCompileUnitInfo->numAttrStr; strIdx++) {
143  if (strcmp(self->dwarfDbgCompileUnitInfo->attrStrs[strIdx], str) == 0) {
144  *attrStrIdx = strIdx;
145  return result;
146  }
147  }
148  if (self->dwarfDbgCompileUnitInfo->maxAttrStr <= self->dwarfDbgCompileUnitInfo->numAttrStr) {
149  self->dwarfDbgCompileUnitInfo->maxAttrStr += 10;
150  if (self->dwarfDbgCompileUnitInfo->attrStrs == NULL) {
151  self->dwarfDbgCompileUnitInfo->attrStrs = (char **)ckalloc(sizeof(char *) * self->dwarfDbgCompileUnitInfo->maxAttrStr);
152  if (self->dwarfDbgCompileUnitInfo->attrStrs == NULL) {
153  return DWARF_DBG_ERR_OUT_OF_MEMORY;
154  }
155  } else {
156  self->dwarfDbgCompileUnitInfo->attrStrs = (char **)ckrealloc((char *)self->dwarfDbgCompileUnitInfo->attrStrs, sizeof(char *) * self->dwarfDbgCompileUnitInfo->maxAttrStr);
157  if (self->dwarfDbgCompileUnitInfo->attrStrs == NULL) {
158  return DWARF_DBG_ERR_OUT_OF_MEMORY;
159  }
160  }
161  }
162 //printf("== numAttrStrs: %d %s\n", self->dwarfDbgCompileUnitInfo->numAttrStr, str);
163  attrStr = &self->dwarfDbgCompileUnitInfo->attrStrs[self->dwarfDbgCompileUnitInfo->numAttrStr];
164  *attrStr = ckalloc(strlen(str) + 1);
165  memset(*attrStr, 0, strlen(str) + 1);
166  memcpy(*attrStr, str, strlen(str));
167  *attrStrIdx = self->dwarfDbgCompileUnitInfo->numAttrStr;
168  self->dwarfDbgCompileUnitInfo->numAttrStr++;
169  return result;
170 }

175  uint8_t result;

225  case DW_AT_GNU_call_site_value:
226  dieAttr->locationInfo = (locationInfo_t *)ckalloc(sizeof(locationInfo_t));
227  if (dieAttr->locationInfo == NULL) {
228  return DWARF_DBG_ERR_OUT_OF_MEMORY;
229  }
230  memset(dieAttr->locationInfo, 0, sizeof(locationInfo_t));
231  break;
232  default:
233  dieAttr->locationInfo = NULL;
234  break;
235  }
236  *attrIdx = dieInfo->numAttr;
237  dieInfo->numAttr++;
238  return result;
239 }
