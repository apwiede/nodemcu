28 /* dwarfDbgEsb.c
29  extensible string buffer.
30 
31  A simple means (vaguely like a C++ class) that
32  enables safely saving strings of arbitrary length built up
33  in small pieces.
34 
35  We really do allow only C strings here. NUL bytes
36  in a string result in adding only up to the NUL (and
37  in the case of certain interfaces here a warning
38  to stderr).
39 
40  The functions assume that
41  pointer arguments of all kinds are not NULL.
42 */

59 static uint8_t initEsbString(dwarfDbgPtr_t self, dwarfDbgEsb_t *data, size_t min_len) {
60  char* d;
61 
62  if (data->esbAllocatedSize > 0) {
63  return;
64  }
65  /* Only esbConstructor applied. Allow for string space. */
66  if (min_len <= allocSize) {
67  min_len = allocSize + 1;/* Allow for NUL at end */
68  } else {
69  min_len++ ; /* Allow for NUL at end */
70  }
71  d = malloc(min_len);
72  if (!d) {
73  fprintf(stderr, "dwarfdump is out of memory allocating %lu bytes\n", (unsigned long) min_len);
74  return TCL_ERROR;
75  }
76  data->esbString = d;
77  data->esbAllocatedSize = min_len;
78  data->esbString[0] = 0;
79  data->esbUsedBytes = 0;
80  return TCL_OK;
81 }

85 /* Make more room. Leaving contents unchanged, effectively.
86  The NUL byte at end has room and this preserves that room.
87 */
88 static uint8_t esbAllocateMore(dwarfDbgPtr_t self, dwarfDbgEsb_t *data, size_t len) {
89  size_t new_size = data->esbAllocatedSize + len;
90  char* newd = 0;
91 
92  if (new_size < allocSize) {
93  new_size = allocSize;
94  }
95  newd = realloc(data->esbString, new_size);
96  if (!newd) {
97  fprintf(stderr, "dwarfdump is out of memory re-allocating %lu bytes\n", (unsigned long) new_size);
98  return TCL_ERROR;
99  }
100  /* If the area was reallocated by realloc() the earlier
101  space was free()d by realloc(). */
102  data->esbString = newd;
103  data->esbAllocatedSize = new_size;
104  return TCL_OK;
105 }

147 // =============================== esbAppendnInternal ===========================
148 
149 /* The 'len' is believed. Do not pass in strings < len bytes long. */
150 static uint8_t esbAppendnInternal(dwarfDbgPtr_t self, dwarfDbgEsb_t *data, const char * in_string, size_t len) {
151  size_t remaining = 0;
152  size_t needed = len;
153 
154  if (data->esbAllocatedSize == 0) {
155  size_t maxlen = (len >= allocSize) ? (len) : allocSize;
156 
157  self->dwarfDbgEsb->initEsbString(self, data, maxlen);
158  }
159  /* ASSERT: data->esbAllocatedSize > data->esbUsedBytes */
160  remaining = data->esbAllocatedSize - data->esbUsedBytes;
161  if (remaining <= needed) {
162  self->dwarfDbgEsb->esbAllocateMore(self, data,len);
163  }
164  strncpy(&data->esbString[data->esbUsedBytes], in_string, len);
165  data->esbUsedBytes += len;
166  /* Insist on explicit NUL terminator */
167  data->esbString[data->esbUsedBytes] = 0;
168  return TCL_OK;
169 }

241 /* Make more room. Leaving contents unchanged, effectively.
242  The NUL byte at end has room and this preserves that room.
243 */
244 static uint8_t esbAllocateMoreIfNeeded(dwarfDbgPtr_t self, dwarfDbgEsb_t *data, const char *in_string,va_list ap) {
245 #ifndef _WIN32
246  static char a_buffer[512];
247 #endif /* _WIN32*/
248 
249  int netlen = 0;
250  va_list ap_copy;
251 
252  /* Preserve the original argument list, to be used a second time */
253  va_copy(ap_copy,ap);
254 
255  netlen = vsnprintf(a_buffer,sizeof(a_buffer),in_string,ap_copy);
256 
257  /* "The object ap may be passed as an argument to another
258  function; if that function invokes the va_arg()
259  macro with parameter ap, the value of ap in the calling
260  function is unspecified and shall be passed to the va_end()
261  macro prior to any further reference to ap."
262  Single Unix Specification. */
263  va_end(ap_copy);
264 
265  /* Allocate enough space to hold the full text */
266  self->dwarfDbgEsb->esbForceAllocation(self, data,netlen + 1);
267  return TCL_OK;
268 }

272 /* Append a formatted string */
273 static uint8_t esbAppendPrintfAp(dwarfDbgPtr_t self, dwarfDbgEsb_t *data,const char *in_string,va_list ap) {
274  int netlen = 0;
275  int expandedlen = 0;
276 
277  /* Allocate enough space for the input string */
278  self->dwarfDbgEsb->esbAllocateMoreIfNeeded(self, data,in_string,ap);
279 
280  netlen = data->esbAllocatedSize - data->esbUsedBytes;
281  expandedlen = vsnprintf(&data->esbString[data->esbUsedBytes], netlen,in_string,ap);
282  if (expandedlen < 0) {
283  /* There was an error.
284  Do nothing. */
285  return;
286  }
287  if (netlen < expandedlen) {
288  /* If data was too small, the max written was one less than
289  netlen. */
290  data->esbUsedBytes += netlen - 1;
291  } else {
292  data->esbUsedBytes += expandedlen;
293  }
294  return TCL_OK;
295 }
