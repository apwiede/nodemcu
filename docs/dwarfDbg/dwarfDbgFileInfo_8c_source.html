117 static uint8_t addSourceFile(dwarfDbgPtr_t self, char *pathName, size_t compileUnitIdx, size_t *fileNameIdx, size_t *fileInfoIdx) {
118  uint8_t result;
119  char *cp;
120  int i;
121  int dirIdx;
122  int fileIdx;
123  fileNameInfo_t *fileNameInfo;
124 
125  result = DWARF_DBG_ERR_OK;
126  cp = strrchr(pathName, '/');
127  *cp++ = '\0';
128 printf("path: %s name: %s\n", pathName, cp);
129  dirIdx = -1;
130  for (i = 0; i < self->dwarfDbgFileInfo->dirNamesInfo.numDirName; i++) {
131  if (strcmp(pathName, self->dwarfDbgFileInfo->dirNamesInfo.dirNames[i]) == 0) {
132 printf("found: dirName %d num: %d\n", i, self->dwarfDbgFileInfo->dirNamesInfo.numDirName);
133  dirIdx = i;
134  break;
135  }
136  }
137  if (dirIdx < 0) {
138  dirIdx = self->dwarfDbgFileInfo->dirNamesInfo.numDirName;
139  result = self->dwarfDbgFileInfo->addDirName(self, pathName);
140  checkErrOK(result);
141  }
142  fileIdx = -1;
143  for (i = 0; i < self->dwarfDbgFileInfo->fileNamesInfo.numFileName; i++) {
144  fileNameInfo = &self->dwarfDbgFileInfo->fileNamesInfo.fileNames[i];
145  if ((strcmp(cp, fileNameInfo->fileName) == 0) && (fileNameInfo->dirNameIdx == dirIdx)) {
146 printf("found: fileName %d num: %d\n", i, self->dwarfDbgFileInfo->fileNamesInfo.numFileName);
147  fileIdx = i;
148  break;
149  }
150  }
151  if (fileIdx < 0) {
152  fileIdx = self->dwarfDbgFileInfo->fileNamesInfo.numFileName;
153  result = self->dwarfDbgFileInfo->addFileName(self, cp, dirIdx);
154  checkErrOK(result);
155  }
156  *fileNameIdx = fileIdx;
157  result = self->dwarfDbgFileInfo->addFileInfo(self, compileUnitIdx, fileIdx, fileInfoIdx);
158  checkErrOK(result);
159  return result;
160 }
