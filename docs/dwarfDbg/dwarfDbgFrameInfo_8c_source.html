60 static int addrMapCompareFunc(const void *l, const void *r) {
61  const struct Addr_Map_Entry *ml = l;
62  const struct Addr_Map_Entry *mr = r;
63  if (ml->mp_key < mr->mp_key) {
64  return -1;
65  }
66  if (ml->mp_key > mr->mp_key) {
67  return 1;
68  }
69  return 0;
70 }

103 struct Addr_Map_Entry *addrMapFind(Dwarf_Unsigned addr, void **tree1) {
104  void *retval = 0;
105  struct Addr_Map_Entry *re = 0;
106  struct Addr_Map_Entry *e = 0;
107 
108  e = addrMapCreateEntry(addr,NULL);
109  retval = dwarf_tfind(e,tree1, addrMapCompareFunc);
110  if (retval) {
111  re = *(struct Addr_Map_Entry **)retval;
112  }
113  /* The one we created here must be deleted, it is dead.
114  We look at the returned one instead. */
115  addrMapFreeFunc(e);
116  return re;
117 }

231 static uint8_t getFrameList(dwarfDbgPtr_t self) {
232  uint8_t result;
233  int fres = 0;
234  Dwarf_Error err = NULL;
235  Dwarf_Cie *cie_data = NULL;
236  Dwarf_Signed cie_element_count = 0;
237  Dwarf_Fde *fde_data = NULL;
238  Dwarf_Signed fde_element_count = 0;
239  Dwarf_Half address_size = 0;
240  Dwarf_Half offset_size = 0;
241  Dwarf_Fde fde;
242  int i = 0;
243 
244  result = DWARF_DBG_ERR_OK;
245  fres = dwarf_get_fde_list(self->elfInfo.dbg, &cie_data, &cie_element_count, &fde_data, &fde_element_count, &err);
246  if (fres != DW_DLV_OK) {
247  return DWARF_DBG_ERR_CANNOT_GET_FDE_LIST;
248  }
249 DWARF_DBG_PRINT(self, "f", 1, "getFrameList: cie_element_count: %d fde_element_count: %d\n", cie_element_count, fde_element_count);
250  fres = dwarf_get_address_size(self->elfInfo.dbg, &address_size, &err);
251  for (i = 0; i < fde_element_count; i++) {
252  int j = 0;
253  Dwarf_Ptr fde_bytes = NULL;
254  Dwarf_Unsigned fde_bytes_length = 0;
255  Dwarf_Off cieOffset = 0;
256  Dwarf_Off fde_offset = 0;
257  Dwarf_Addr lowPc = 0;
258  Dwarf_Unsigned funcLgth = 0;
259  Dwarf_Signed cieIdx = 0;
260  struct Addr_Map_Entry *mp = 0;
261  void *lowpcSet = NULL;
262  size_t cieFdeIdx;
263 
264  fde = fde_data[i];
265  fres = dwarf_get_fde_range(fde, &lowPc, &funcLgth, &fde_bytes, &fde_bytes_length, &cieOffset, &cieIdx, &fde_offset, &err);
266  if (fres != DW_DLV_OK) {
267  return DWARF_DBG_ERR_CANNOT_GET_FDE_RANGE;
268  }
269 // mp = addrMapFind(low_pc, &lowpcSet);
270 //printf("mp: %p low_pc: 0x%08x lowpcSet: %p\n", mp, low_pc, lowpcSet);
271 //printf("i: %d low_pc: 0x%08x func_length: %d pc+fl: 0x%08x cie_offset: 0x%08x cie_index: %d fde_offset: 0x%08x\n", i, lowPc, funcLgth, lowPc+funcLgth, cieOffset, cieIdx, fde_offset);
272  self->dwarfDbgFrameInfo->addCieFde(self, cieIdx, lowPc, funcLgth, &cieFdeIdx);
273  for (j = lowPc; j < lowPc + funcLgth; j++) {
274  Dwarf_Half k = 0;
275  Dwarf_Addr jsave = 0;
276  Dwarf_Signed reg = 0;
277  Dwarf_Signed offsetRelevant = 0;
278  Dwarf_Small valueType = 0;
279  Dwarf_Signed offsetOrBlockLen = 0;
280  Dwarf_Signed offset = 0;
281  Dwarf_Ptr blockPtr = 0;
282  Dwarf_Addr rowPc = 0;
283  Dwarf_Bool hasMoreRows = 0;
284  Dwarf_Addr subsequentPc = 0;
285  int fires = 0;
286  size_t fdeIdx;
287 
288  unsigned cf_table_entry_count = 100;
289  unsigned cf_cfa_reg = 1436; // DW_FRAME_CFA_COL3
290  unsigned cf_initial_rule_value = 1035; // DW_FRAME_SAME_VAL
291  unsigned cf_same_val = 1035;
292  unsigned cf_undefined_val = 1034;
293 
294  jsave = j;
295  fires = dwarf_get_fde_info_for_cfa_reg3_b(fde, j, &valueType, &offsetRelevant, &reg,
296  &offsetOrBlockLen, &blockPtr, &rowPc, &hasMoreRows, &subsequentPc, &err);
297  offset = offsetOrBlockLen;
298  if (fires == DW_DLV_ERROR) {
299  return DWARF_DBG_ERR_GET_FDE_INFO_FOR_CFA_REG3_B;
300  }
301  if (fires == DW_DLV_NO_ENTRY) {
302  continue;
303  }
304 //printf("value_type1: %d offset_relevant: %d lowPc: 0x%08x funcLgth: %d reg: 0x%04x offset: %d\n", valueType, offsetRelevant, lowPc, funcLgth, reg, offset);
305  if (offsetRelevant) {
306  result = addFde(self, cieIdx, cieFdeIdx, jsave, funcLgth, reg, offset, &fdeIdx);
307  }
308  if (!hasMoreRows) {
309  j = lowPc + funcLgth - 1;
310  } else {
311  if (subsequentPc > j) {
312  /* Loop head will increment j to make up
313  for -1 here. */
314  j = subsequentPc - 1;
315  }
316  }
317  if (offsetRelevant) {
318  size_t frcIdx = 0;
319 //printf("j2: 0x%08x value_type3: %d offset_relevant: %d reg: %d offset: %d\n", j, valueType, offsetRelevant, reg, offset);
320  result = addFrameRegCol(self, cieIdx, cieFdeIdx, fdeIdx, jsave, offset, reg, &frcIdx);
321  checkErrOK(result);
322  }
323  for (k = 0; k < cf_table_entry_count; k++) {
324  Dwarf_Signed reg = 0;
325  Dwarf_Signed offset_relevant = 0;
326  int fires = 0;
327  Dwarf_Small value_type = 0;
328  Dwarf_Ptr block_ptr = 0;
329  Dwarf_Signed offset_or_block_len = 0;
330  Dwarf_Signed offset = 0;
331  Dwarf_Addr row_pc = 0;
332  size_t frcIdx = 0;
333 
334  fires = dwarf_get_fde_info_for_reg3(fde, k, jsave, &value_type, &offset_relevant, &reg,
335  &offset_or_block_len, &block_ptr, &row_pc, &err);
336  offset = offset_or_block_len;
337  if (fires == DW_DLV_ERROR) {
338  return DWARF_DBG_ERR_GET_FDE_INFO_FOR_REG3;
339  }
340  if (fires == DW_DLV_NO_ENTRY) {
341  continue;
342  }
343  if (offset_relevant) {
344 //printf("k: %d value_type2: %d offset_relevant: %d reg: %d offset: %d\n", k, value_type, offset_relevant, reg, offset);
345 //printf("k: %d value_type2: %d offset_relevant: %d reg: %d offset: %d\n", k, value_type, offset_relevant, reg, offset);
346  result = addFrameRegCol(self, cieIdx, cieFdeIdx, fdeIdx, jsave, offset, reg, &frcIdx);
347  checkErrOK(result);
348  }
349  }
350  }
351  }
352  return result;
353 }
