52 static uint8_t handleOneDie(dwarfDbgPtr_t self, Dwarf_Die die, char **srcfiles, Dwarf_Signed cnt, Dwarf_Bool isSibling, int dieAndChildrenIdx, int *dieInfoIdx) {
53  uint8_t result;
54  int tres = 0;
55  int ores = 0;
56  int atres = 0;
57  int res = 0;
58  int sres = 0;
59  int cdres = DW_DLV_OK;
60  const char * tagName = 0;
61  int i = 0;
62  int typeIdx = 0;
63  Dwarf_Error err;
64  Dwarf_Half tag = 0;
65  Dwarf_Off offset = 0;
66  Dwarf_Signed attrCnt = 0;
67  Dwarf_Attribute *atList = 0;
68  Dwarf_Die child = NULL;
69  Dwarf_Die sibling = NULL;
70  compileUnit_t *compileUnit;
71  dieAndChildrenInfo_t *dieAndChildrenInfo;
72  dieInfo_t *dieInfo = NULL;
73  attrValues_t *attrValues;
74 
75  result = DWARF_DBG_ERR_OK;
76 //printf(">>handleOneDie die: %p numDies: %d\n", die, ++numDies);
77  compileUnit = self->dwarfDbgCompileUnitInfo->currCompileUnit;
78 //printf("handleOneDie: level: %d\n", compileUnit->level);
79  tres = dwarf_tag(die, &tag, &err);
80  if (tres != DW_DLV_OK) {
81  printf("accessing tag of die! tres: %d, err: %p", tres, err);
82  }
83  result = self->dwarfDbgStringInfo->getDW_TAG_string(self, tag, &tagName);
84  checkErrOK(result);
85 
86  ores = dwarf_die_CU_offset(die, &offset, &err);
87  if (ores != DW_DLV_OK) {
88  printf("dwarf_die_CU_offset ores: %d err: %p", ores, err);
89  return DWARF_DBG_ERR_CANNOT_GET_CU_OFFSET;
90  }
91  DWARF_DBG_PRINT(self, "G", 1, "<%2d><0x%08x> %*s%s\n", compileUnit->level, offset, compileUnit->level * 2, " ", tagName);
92  dieAndChildrenInfo = &compileUnit->dieAndChildrenInfos[dieAndChildrenIdx];
93  if (isSibling) {
94 //printf(" >>numAddSibling: %d\n", ++numAddSibling);
95  result = self->dwarfDbgDieInfo->addDieSibling(self, dieAndChildrenIdx, offset, tag, dieInfoIdx);
96  checkErrOK(result);
97  dieInfo = &dieAndChildrenInfo->dieSiblings[*dieInfoIdx];
98  } else {
99 //printf(" >>numAddChild: %d\n", ++numAddChild);
100  result = self->dwarfDbgDieInfo->addDieChild(self, dieAndChildrenIdx, offset, tag, dieInfoIdx);
101  checkErrOK(result);
102  dieInfo = &dieAndChildrenInfo->dieChildren[*dieInfoIdx];
103  }
104  atres = dwarf_attrlist(die, &atList, &attrCnt, &err);
105 //printf(" >>attrCnt: %d\n", attrCnt);
106  attrValues = &compileUnit->attrValues;
107  memset(attrValues, 0, sizeof(attrValues_t));
108  for (i = 0; i < attrCnt; i++) {
109  Dwarf_Half attr;
110  Dwarf_Attribute attrIn;
111  int ares;
112  int dieAttrIdx;
113 
114  attrIn = atList[i];
115  ares = dwarf_whatattr(attrIn, &attr, &err);
116  if (ares == DW_DLV_OK) {
117  result = self->dwarfDbgAttributeInfo->handleAttribute(self, die, attr, attrIn, srcfiles, cnt, dieAndChildrenIdx, *dieInfoIdx, isSibling, &dieAttrIdx);
118 //printf("result: %d\n", result);
119  }
120  }
121  switch (tag) {
122  case DW_TAG_base_type:
123  result = self->dwarfDbgTypeInfo->addBaseType(self, attrValues->name, attrValues->byteSize, attrValues->encoding, &typeIdx);
124  checkErrOK(result);
125  dieInfo->tagRefIdx = typeIdx;
126  dieInfo->flags = TAG_REF_BASE_TYPE;
127 //printf("baseType tagRef: %d offset: 0x%08x\n", dieInfo->tagRefIdx, dieInfo->offset);
128  break;
129  case DW_TAG_subroutine_type:
130 #ifdef NODTEF
131  result = self->dwarfDbgTypeInfo->addSubroutineType(self, attrValues->name, attrValues->byteSize, attrValues->encoding, &typeIdx);
132  checkErrOK(result);
133  dieInfo->tagRefIdx = typeIdx;
134  dieInfo->flags = TAG_REF_SUBROUTINE_TYPE;
135 printf("subroutineType tagRef: %d offset: 0x%08x\n", dieInfo->tagRefIdx, dieInfo->offset);
136 #endif
137  break;
138  case DW_TAG_typedef:
139 #ifdef NODTEF
140  result = self->dwarfDbgTypeInfo->addTypeDef(self, attrValues->name, attrValues->pathNameIdx, attrValues->lineNo, attrValues->dwTypeIdx, &typeIdx);
141  checkErrOK(result);
142  dieInfo->tagRefIdx = typeIdx;
143  dieInfo->flags = TAG_REF_TYPEDEF;
144 printf("typeDef tagRef: %d offset: 0x%08x\n", dieInfo->tagRefIdx, dieInfo->offset);
145 #endif
146  break;
147  case DW_TAG_pointer_type:
148 #ifdef NODTEF
149  result = self->dwarfDbgTypeInfo->addPointerTypf(self, attrValues->name, attrValues->pathNameIdx, attrValues->lineNo, attrValues->dwTypeIdx, &typeIdx);
150  checkErrOK(result);
151  dieInfo->tagRefIdx = typeIdx;
152  dieInfo->flags = TAG_REF_POINTER_TYPE;
153 printf("pointerType tagRef: %d offset: 0x%08x\n", dieInfo->tagRefIdx, dieInfo->offset);
154 #endif
155  break;
156  case DW_TAG_const_type:
157  case DW_TAG_member:
158  case DW_TAG_structure_type:
159  case DW_TAG_array_type:
160  case DW_TAG_enumeration_type:
161  case DW_TAG_enumerator:
162  case DW_TAG_union_type:
163  case DW_TAG_volatile_type:
164 //printf("should store type\n");
165  break;
166  }
167  return result;
168 }

172 static uint8_t handleDieAndChildren(dwarfDbgPtr_t self, Dwarf_Die in_die_in, char **srcfiles, Dwarf_Signed cnt) {
173  uint8_t result;
174  int dieInfoIdx = 0;
175  int dieAndChildrenIdx = 0;
176  Dwarf_Error err;
177  Dwarf_Die child = 0;
178  Dwarf_Die sibling = 0;
179  Dwarf_Die in_die = in_die_in;
180  int cdres = DW_DLV_OK;
181  int numChildren = 0;
182  Dwarf_Bool isSibling = 0;
183  compileUnit_t *compileUnit = NULL;
184 
185 
186  result = DWARF_DBG_ERR_OK;
187  compileUnit = self->dwarfDbgCompileUnitInfo->currCompileUnit;
188 printf("@@handleDieAndChildren die: %p level: %d isCompileUnitDie: %d\n", in_die_in, ++childrenLevel, compileUnit->isCompileUnitDie);
189  result = self->dwarfDbgDieInfo->addDieAndChildren(self, in_die_in, &dieAndChildrenIdx);
190  checkErrOK(result);
191  for(;;) {
192  ++numChildren;
193 //printf("pre-descent numChildren: %d\n", numChildren);
194  /* Here do pre-descent processing of the die. */
195  {
196 //printf("before handleOneDie in_die: %p\n", in_die);
197  result = handleOneDie(self, in_die, srcfiles, cnt, isSibling, dieAndChildrenIdx, &dieInfoIdx);
198  compileUnit->isCompileUnitDie = 0;
199 //printf("after handleOneDie in_die: %p\n", in_die);
200 //printf("call dwarf_child in_die: %p\n", in_die);
201  child = NULL;
202  cdres = dwarf_child(in_die, &child, &err);
203 //printf("after call dwarf_child in_die: %p child: %p cdres: %d\n", in_die, child, cdres);
204  /* child first: we are doing depth-first walk */
205  if (cdres == DW_DLV_OK) {
206 //printf("child first\n");
207 //printf("call recursive handleDieAndChildren: child: %p numChildren: %d\n", child, numChildren);
208  compileUnit->level++;
209  handleDieAndChildren(self, child, srcfiles, cnt);
210  compileUnit->level--;
211  dwarf_dealloc(self->elfInfo.dbg, child, DW_DLA_DIE);
212  child = 0;
213  }
214  cdres = dwarf_siblingof_b(self->elfInfo.dbg, in_die, /* is_info */1, &sibling, &err);
215 //printf("dwarf_siblingof_b: numSiblings: %d in_die: %p sibling: %p\n", ++numSiblings, in_die, sibling);
216  if (cdres == DW_DLV_OK) {
217  /* handleDieAndChildren(dbg, sibling, srcfiles, cnt); We
218  loop around to actually print this, rather than
219  recursing. Recursing is horribly wasteful of stack
220  space. */
221  } else if (cdres == DW_DLV_ERROR) {
222  printf("error in dwarf_siblingofi_b cdres: %d err: %p", cdres, err);
223  return DWARF_DBG_ERR_CANNOT_GET_SIBLING_OF_COMPILE_UNIT;
224  }
225 //printf("post-descent\n");
226  if (in_die != in_die_in) {
227  /* Dealloc our in_die, but not the argument die, it belongs
228  to our caller. Whether the siblingof call worked or not. */
229  dwarf_dealloc(self->elfInfo.dbg, in_die, DW_DLA_DIE);
230  in_die = 0;
231  }
232  if (cdres == DW_DLV_OK) {
233  /* Set to process the sibling, loop again. */
234  in_die = sibling;
235  isSibling = 1;
236  } else {
237  /* We are done, no more siblings at this level. */
238  break;
239  }
240  }
241  } /* end for loop on siblings */
242 //printf("handleDieAndChildren done die: %p level: %d\n", in_die_in, childrenLevel--);
243  return result;
244 }

248 static uint8_t handleCompileUnits(dwarfDbgPtr_t self) {
249  uint8_t result;
250  const char * sectionName = 0;
251  int res = 0;
252  Dwarf_Bool isInfo = TRUE;
253  Dwarf_Error err;
254  unsigned loopCount = 0;
255  int nres = DW_DLV_OK;
256  int compileUnitCount = 0;
257  char * compileUnitShortName = NULL;
258  char * compileUnitLongName = NULL;
259  int i = 0;
260  int fileLineIdx;
261  Dwarf_Signed srcCnt = 0;
262  char **srcfiles = 0;
263 
264  isInfo = TRUE;
265  result = DWARF_DBG_ERR_OK;
266  res = dwarf_get_die_section_name(self->elfInfo.dbg, isInfo, &sectionName, &err);
267  if (res != DW_DLV_OK || !sectionName || !strlen(sectionName)) {
268  sectionName = ".debug_info";
269  }
270  result = self->dwarfDbgCompileUnitInfo->getAddressSizeAndMax(self, &self->dwarfDbgCompileUnitInfo->addrSize, &self->dwarfDbgCompileUnitInfo->maxAddr, &err);
271  DWARF_DBG_PRINT(self, "G", 1, "addrSize: 0x%08x maxAddr: 0x%08x\n", self->dwarfDbgCompileUnitInfo->addrSize, self->dwarfDbgCompileUnitInfo->maxAddr);
272  /* Loop over compile units until it fails. */
273  for (;;++loopCount) {
274  int sres = DW_DLV_OK;
275  compileUnit_t *compileUnit = 0;
276 
277  result = self->dwarfDbgCompileUnitInfo->addCompileUnit(self);
278  if (result == DWARF_DBG_ERR_NO_ENTRY) {
279  // we have processed all entries
280  result = DWARF_DBG_ERR_OK;
281  break;
282  }
283  checkErrOK(result);
284  compileUnit = self->dwarfDbgCompileUnitInfo->currCompileUnit;
285  DWARF_DBG_PRINT(self, "G", 1, "handle srcfiles\n");
286  {
287  Dwarf_Error srcerr = 0;
288  int srcf = dwarf_srcfiles(compileUnit->compileUnitDie, &srcfiles, &srcCnt, &err);
289 
290  if (srcf == DW_DLV_ERROR) {
291  return DWRAF_DBG_ERR_GET_SRC_FILES;
292  } /*DW_DLV_NO_ENTRY generally means there
293  there is no DW_AT_stmt_list attribute.
294  and we do not want to print anything
295  about statements in that case */
296 
297  DWARF_DBG_PRINT(self, "G", 1, "srcCnt: %d currCompileUnitIdx: %d\n", srcCnt, self->dwarfDbgCompileUnitInfo->currCompileUnitIdx);
298  for (i = 0; i < srcCnt; i++) {
299  int pathNameIdx;
300  int fileInfoIdx;
301 
302  DWARF_DBG_PRINT(self, "G", 1, " src: %s\n", srcfiles[i]);
303  result = self->dwarfDbgFileInfo->addSourceFile(self, srcfiles[i], &pathNameIdx, &fileInfoIdx);
304 //printf(" src: %s %d pathNameIdx: %d fileInfoIdx: %d\n", srcfiles[i], i, pathNameIdx, fileInfoIdx);
305  checkErrOK(result);
306  }
307 
308  compileUnit->isCompileUnitDie = 1;
309  compileUnit->level = 0;
310  result = handleDieAndChildren(self, compileUnit->compileUnitDie, srcfiles, srcCnt);
311  }
312 
313  // add the typedefs here
314  {
315  int dieAndChildrenIdx;
316 printf("++ numDieAndChildren: %d cu: %s\n", compileUnit->numDieAndChildren, compileUnit->shortFileName);

324

327 printf("++ siblings: constTypes\n");

329 
330 printf("++ children: subroutineTypes\n");
331  self->dwarfDbgTypeInfo->addChildrenTypes(self, dieAndChildrenIdx, TAG_REF_SUBROUTINE_TYPE);
332 printf("++ siblings: subroutineTypes\n");
333  self->dwarfDbgTypeInfo->addSiblingsTypes(self, dieAndChildrenIdx, TAG_REF_SUBROUTINE_TYPE);
