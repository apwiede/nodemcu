48 static uint8_t addCompileUnit(dwarfDbgPtr_t self, size_t *compileUnitIdx) {
49  uint8_t result;
50  compileUnit_t *compileUnit;
51 
52  result = DWARF_DBG_ERR_OK;
53  if (self->dwarfDbgGetInfo->maxCompileUnit <= self->dwarfDbgGetInfo->numCompileUnit) {
54  self->dwarfDbgGetInfo->maxCompileUnit += 50;
55  if (self->dwarfDbgGetInfo->compileUnits == NULL) {
56  self->dwarfDbgGetInfo->compileUnits = (compileUnit_t *)ckalloc(sizeof(compileUnit_t) * self->dwarfDbgGetInfo->maxCompileUnit);
57  if (self->dwarfDbgGetInfo->compileUnits == NULL) {
58  return DWARF_DBG_ERR_OUT_OF_MEMORY;
59  }
60  } else {
61  self->dwarfDbgGetInfo->compileUnits = (compileUnit_t *)ckrealloc((char *)self->dwarfDbgGetInfo->compileUnits, sizeof(compileUnit_t) * self->dwarfDbgGetInfo->maxCompileUnit);
62  if (self->dwarfDbgGetInfo->compileUnits == NULL) {
63  return DWARF_DBG_ERR_OUT_OF_MEMORY;
64  }
65  }
66  }
67 printf("compileUnitIdx: %d\n", self->dwarfDbgGetInfo->numCompileUnit);
68  compileUnit = &self->dwarfDbgGetInfo->compileUnits[self->dwarfDbgGetInfo->numCompileUnit];
69  memset(compileUnit, 0, sizeof(compileUnit_t));
70  *compileUnitIdx = self->dwarfDbgGetInfo->numCompileUnit;
71  self->dwarfDbgGetInfo->numCompileUnit++;
72  return result;
73 }

82 /* Returns the producer of the CU
83  Caller must ensure producernameout is
84  a valid, constructed, empty dwarfDbgEsb_t instance before calling.
85  Never returns DW_DLV_ERROR. */
86 static int getProducerName(dwarfDbgPtr_t self, char **producerName) {
87  Dwarf_Attribute producerAttr = 0;
88  Dwarf_Error pnerr = 0;
89 
90  int ares = dwarf_attr(self->dwarfDbgGetInfo->currCompileUnit->compileUnitDie, DW_AT_producer, &producerAttr, &pnerr);
91  if (ares == DW_DLV_ERROR) {
92  printf("hassattr on DW_AT_producer ares: %d pnerr: %d", ares, pnerr);
93  }
94  if (ares == DW_DLV_NO_ENTRY) {
95  /* We add extra quotes so it looks more like
96  the names for real producers that getAttrValue
97  produces. */
98  *producerName = "\"<CU-missing-DW_AT_producer>\"";
99  } else {
100  /* DW_DLV_OK */
101  /* The string return is valid until the next call to this
102  function; so if the caller needs to keep the returned
103  string, the string must be copied (makename()). */
104  getAttrValue(self, 1, producerAttr, NULL, 0, producerName);
105  }
106  return ares;
107 }

111 /* Returns the name of the compile unit. In case of error, give up, do not return. */
112 static int getCompileUnitName(dwarfDbgPtr_t self, char **shortName, char **longName) {
113  Dwarf_Attribute nameAttr = 0;
114  Dwarf_Error lerr = 0;
115  int ares;
116 
117  ares = dwarf_attr(self->dwarfDbgGetInfo->currCompileUnit->compileUnitDie, DW_AT_name, &nameAttr, &lerr);
118  if (ares == DW_DLV_ERROR) {
119  printf("hassattr on DW_AT_name ares: %d lerr: %d", ares, lerr);
120  } else {
121  if (ares == DW_DLV_NO_ENTRY) {
122  *shortName = "<unknown name>";
123  *longName = "<unknown name>";
124  } else {
125  /* DW_DLV_OK */
126  /* The string return is valid until the next call to this
127  function; so if the caller needs to keep the returned
128  string, the string must be copied (makename()). */
129  char *fileName = 0;
130 
131 // self->dwarfDbgEsb->esbEmptyString(self, &esbLongCuName);
132  getAttrValue(self, 1, nameAttr, NULL, 0, longName);
133  /* Generate the short name (fileName) */
134  fileName = strrchr(*longName,'/');
135  if (!fileName) {
136  fileName = strrchr(*longName,'\\');
137  }
138  if (fileName) {
139  ++fileName;
140  } else {
141  fileName = *longName;
142  }
143  *shortName = fileName;
144  }
145  }
146  dwarf_dealloc(self->elfInfo.dbg, nameAttr, DW_DLA_ATTR);
147  return ares;
148 }

348 static uint8_t handleOneDieSection(dwarfDbgPtr_t self) {
349  uint8_t result;

353  Dwarf_Error err;

356  int compileUnitCount = 0;
357  char * compileUnitShortName = NULL;
358  char * compileUnitLongName = NULL;
359  size_t compileUnitIdx = 0;
360 
361 
362 printf("handleOneDieSection dbg: %p\n", self->elfInfo.dbg);
363  isInfo = TRUE;
364  result = DWARF_DBG_ERR_OK;
365  res = dwarf_get_die_section_name(self->elfInfo.dbg, isInfo, &sectionName, &err);
366  if (res != DW_DLV_OK || !sectionName || !strlen(sectionName)) {
367  sectionName = ".debug_info";
368  }
369 printf("sectionName: %s\n", sectionName);
370  /* Loop over compile units until it fails. */
371  for (;;++loopCount) {
372  int sres = DW_DLV_OK;
373  compileUnit_t *compileUnit;
374 // struct Dwarf_Debug_Fission_Per_CU_s fission_data;
375 // int fission_data_result = 0;
376 
377  result = self->dwarfDbgGetInfo->addCompileUnit(self, &compileUnitIdx);
378 //printf("addCompileUnit: result: %d compileUnitIdx: %d\n", result, compileUnitIdx);
379  checkErrOK(result);
380 // memset(&fission_data,0,sizeof(fission_data));
381  compileUnit = &self->dwarfDbgGetInfo->compileUnits[compileUnitIdx];
382 //printf("call dwarf_next_cu_header_d\n");

384  &compileUnit->compileUnitHeaderLength,

395 //printf("after dwarf_next_cu_header_d nres: %d loop_count: %d\n", nres, loopCount);
396  if (nres == DW_DLV_NO_ENTRY) {
397  // we have processed all entries
398  break;
399  }
400  if (nres != DW_DLV_OK) {
401  return DWARF_DBG_ERR_CANNOT_GET_NEXT_COMPILE_UNIT;
402  }
403  self->dwarfDbgGetInfo->currCompileUnit = compileUnit;
404  /* get basic information about the current compile unit: producer, name */

513 printf(" NAME: %s result: %d\n", buf, result);
514  checkErrOK(result);
515  }
516  break;
517  default:
518  break;
519  }
520  // here we need to handle children etc.
521  }
522  }

525 printf("fileLineIdx: %d\n", fileLineIdx);
526  checkErrOK(result);

528  size_t fileNameIdx;
529  size_t fileInfoIdx;
530 
531 printf(" src: %s\n", srcFiles[i]);
532  result = self->dwarfDbgFileInfo->addSourceFile(self, srcFiles[i], compileUnitIdx, &fileNameIdx, &fileInfoIdx);
533  checkErrOK(result);
534  }
535  }
536  }
537  /* Release the 'compileUnitDie' created by the call
538  to 'dwarf_siblingof' at the top of the main loop. */
539  dwarf_dealloc(self->elfInfo.dbg, compileUnit->compileUnitDie, DW_DLA_DIE);

541  } // end loop
