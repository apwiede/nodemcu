62 static uint8_t addLineInfo(dwarfDbgPtr_t self, Dwarf_Addr pc, int lineNo, int flags, uint16_t isa, uint16_t discriminator, int fileInfoIdx, int *fileLineIdx) {
63  uint8_t result;
64  compileUnit_t *compileUnit;
65  fileInfo_t *fileInfo;
66  lineInfo_t *lineInfo;
67 
68 //printf("addFileLine: pc: 0x%08x lineNo: %d fileInfoIdx: %d\n", pc, lineNo, fileInfoIdx);
69  result = DWARF_DBG_ERR_OK;
70  compileUnit = self->dwarfDbgCompileUnitInfo->currCompileUnit;
71  if (compileUnit->fileInfos == NULL) {
72  // seems to be no file infos!!
73  return result;
74  }
75  fileInfo = &compileUnit->fileInfos[fileInfoIdx];
76  if (fileInfo->maxFileLine <= fileInfo->numFileLine) {
77  fileInfo->maxFileLine += 5;
78  if (fileInfo->fileLines == NULL) {
79  fileInfo->fileLines = (lineInfo_t *)ckalloc(sizeof(lineInfo_t) * fileInfo->maxFileLine);
80  if (fileInfo->fileLines == NULL) {
81  return DWARF_DBG_ERR_OUT_OF_MEMORY;
82  }
83  } else {
84  fileInfo->fileLines = (lineInfo_t *)ckrealloc((char *)fileInfo->fileLines, sizeof(lineInfo_t) * fileInfo->maxFileLine);
85  if (fileInfo->fileLines == NULL) {
86  return DWARF_DBG_ERR_OUT_OF_MEMORY;
87  }
88  }
89  }
90  lineInfo = &fileInfo->fileLines[fileInfo->numFileLine];
91  lineInfo->lineNo = lineNo;
92  lineInfo->pc = pc;
93  lineInfo->flags = flags;
94  lineInfo->isa = isa;
95  lineInfo->discriminator = discriminator;
96  *fileLineIdx = fileInfo->numFileLine;
97  fileInfo->numFileLine++;
98  return result;
99 }

103 static uint8_t handleLineInfos(dwarfDbgPtr_t self, int *fileLineIdx) {
104  uint8_t result;
105  Dwarf_Unsigned lineVersion = 0;
106  Dwarf_Line_Context lineContext = 0;
107  Dwarf_Small tableCount = 0;
108  Dwarf_Error err = 0;
109  int lres = 0;
110  Dwarf_Signed lineCount = 0;
111  Dwarf_Line *lineBuf = NULL;
112  Dwarf_Signed lineCountActuals = 0;
113  Dwarf_Line *lineBufActuals = NULL;
114  Dwarf_Addr pc = 0;
115  Dwarf_Unsigned lineNo = 0;
116  Dwarf_Bool newstatement = 0;
117  Dwarf_Bool lineendsequence = 0;
118  Dwarf_Bool new_basic_block = 0;
119  int i = 0;
120  int fileInfoIdx;
121  compileUnit_t *compileUnit;
122 
123  result = DWARF_DBG_ERR_OK;
124 //printf("getCompileUnitLineInfos\n");
125  compileUnit = self->dwarfDbgCompileUnitInfo->currCompileUnit;
126  lres = dwarf_srclines_b(compileUnit->compileUnitDie, &lineVersion, &tableCount, &lineContext, &err);
127 if (tableCount > 0) {
128 printf(">>table_count: %d lineVersion: %d\n", tableCount, lineVersion);
129 }
130  if (lres == DW_DLV_OK) {
131 //printf("dwarf_srclines_two_level_from_linecontext\n");
132  lres = dwarf_srclines_two_level_from_linecontext(lineContext, &lineBuf, &lineCount, &lineBufActuals, &lineCountActuals, &err);
133 //printf("lineCount: %d lineCountActuals: %d\n", lineCount, lineCountActuals);
134  if (lres != DW_DLV_OK) {
135  return DWARF_DBG_ERR_GET_SRC_LINES;
136  }
137  if (lineCount > 0) {
138 //printf(">>>lineCount: %d\n", lineCount);
139  for (i = 0; i < lineCount; i++) {
140  Dwarf_Line line = lineBuf[i];
141  char* fileName = 0;
142  int ares = 0;
143  int lires = 0;
144  int nsres = 0;
145  int disres = 0;
146  Dwarf_Bool prologue_end = 0;
147  Dwarf_Bool epilogue_begin = 0;
148  Dwarf_Unsigned isa = 0;
149  Dwarf_Unsigned discriminator = 0;
150  int flags;
151  char *ns;
152  char *bb;
153  char *et;
154  char *pe;
155  char *eb;
156 
157  pc = 0;
158  ares = dwarf_lineaddr(line, &pc, &err);
159  if (ares != DW_DLV_OK) {
160  return DWARF_DBG_ERR_GET_LINE_ADDR;
161  }
162  lires = dwarf_lineno(line, &lineNo, &err);
163  if (lires != DW_DLV_OK) {
164  return DWARF_DBG_ERR_GET_LINE_NO;
165  }
166  flags = 0;
167  ns = "";
168  bb = "";
169  et = "";
170  pe = "";
171  eb = "";
172  nsres = dwarf_linebeginstatement(line, &newstatement, &err);
173  if (nsres == DW_DLV_OK) {
174 //printf("NS\n");
175  if (newstatement) {
176  ns = " NS";
177  flags |= LINE_NEW_STATEMENT;
178  }
179  }
180  nsres = dwarf_lineblock(line, &new_basic_block, &err);
181  if (nsres == DW_DLV_OK) {
182  if (new_basic_block) {
183 //printf("BB\n");
184  bb = " BB";
185  flags |= LINE_NEW_BASIC_BLOCK;
186  }
187  }
188  nsres = dwarf_lineendsequence(line, &lineendsequence, &err);
189  if (nsres == DW_DLV_OK) {
190  if (lineendsequence) {
191 //printf("ET\n");
192  et = " ET";
193  flags |= LINE_END_SEQUENCE;
194  }
195  }
196  disres = dwarf_prologue_end_etc(line, &prologue_end, &epilogue_begin, &isa, &discriminator, &err);
197  if (disres == DW_DLV_OK) {
198 //printf("prologue_end: %d epilogue_begin: %d isa: %d discriminator: %d\n", prologue_end, epilogue_begin, isa, discriminator);
199  if (prologue_end) {
200  pe = " PE";
201  flags |= LINE_PROLOGUE_END;
202  }
203  if (epilogue_begin) {
204  eb = " EB";
205  flags |= LINE_PROLOGUE_BEGIN;
206  }
207  }
208  DWARF_DBG_PRINT(self, "L", 1, "dwarf_lineaddr: line: 0x%08x pc: 0x%08x lineNo: %d%s%s%s%s%s isa: %d dis: %d\n", line, pc, lineNo, ns, bb, et, eb, pe, isa, discriminator);
209  result = self->dwarfDbgLineInfo->addLineInfo(self, pc, lineNo, flags, (uint16_t)isa, (uint16_t)discriminator, compileUnit->fileInfoIdx, fileLineIdx);
210 
211  }
212  }
213  } else {
214  return DWARF_DBG_ERR_GET_SRC_LINES;
215  }
216  return result;
217 }
