151 static int opHasNoOperands(int op) {
152  unsigned i = 0;
153 
154  if (op >= DW_OP_lit0 && op <= DW_OP_reg31) {
155  return TRUE;
156  }
157  for (; ; ++i) {
158  operationDescr_t *odp = &operationDescInfos[i];
159  if (odp->op_code == 0) {
160  break;
161  }
162  if (odp->op_code != op) {
163  continue;
164  }
165  if (odp->op_count == 0) {
166  return TRUE;
167  }
168  return FALSE;
169  }
170  return FALSE;
171 }

175 static void showContents(char **stringOut, unsigned int length, const unsigned char *bp) {
176  unsigned int i = 0;
177  char buf[20];
178 
179  if(!length) {
180  return;
181  }
182  strcat(*stringOut," contents 0x");
183  for (; i < length; ++i,++bp) {
184  /* Do not use DW_PR_DUx here, the value *bp is a const unsigned char. */
185  snprintf(buf, sizeof(buf), "%02x", *bp);
186  strcat(*stringOut, buf);
187  }
188 }

192 static uint8_t handleOneExprOp(dwarfDbgPtr_t self, locationOp_t *locationOp, int index, char **stringOut) {
193  /* local_space_needed is intended to be 'more than big enough' for a short group of loclist entries. */
194  int result;
195  char buf[100];
196  Dwarf_Small op = 0;
197  Dwarf_Unsigned opd1 = 0;
198  Dwarf_Unsigned opd2 = 0;
199  Dwarf_Unsigned opd3 = 0;
200  Dwarf_Unsigned offsetForBranch = 0;
201  const char *opName = NULL;
202  Dwarf_Error err = NULL;
203 
204  if (index > 0) {
205  strcat(*stringOut, " ");
206  }
207  /* DWARF 2,3,4 and DWARF5 style */
208  op = locationOp->op;
209  opd1 = locationOp->opd1;
210  opd2 = locationOp->opd2;
211  opd3 = locationOp->opd3;
212  offsetForBranch = locationOp->offsetForBranch;
213  result = self->dwarfDbgStringInfo->getDW_OP_string(self, op, &opName);
214  strcat(*stringOut, opName);
215  if (opHasNoOperands(op)) {
216  /* Nothing to add. */
217  } else {
218  if (op >= DW_OP_breg0 && op <= DW_OP_breg31) {
219  snprintf(buf, sizeof(buf), "%+d", (Dwarf_Signed) opd1);
220  strcat(*stringOut, buf);
221  } else {
222 //printf("opName: %s op: 0x%02x opd1: %d opd2: 0x%08x\n", opName, op, opd1, opd2);
223  switch (op) {
224  case DW_OP_addr:
225  snprintf(buf, sizeof(buf), " <0x%08x>", (Dwarf_Unsigned) opd1);
226  strcat(*stringOut, buf);
227  break;
228  case DW_OP_const1s:
229  case DW_OP_const2s:
230  case DW_OP_const4s:
231  case DW_OP_const8s:
232  case DW_OP_consts:
233  case DW_OP_skip:
234  case DW_OP_bra:
235  case DW_OP_fbreg:
236  strcat(*stringOut," ");
237  snprintf(buf, sizeof(buf), " %+d", (Dwarf_Signed) opd1);
238  strcat(*stringOut, buf);
239  break;
240  case DW_OP_GNU_addr_index: /* unsigned val */
241  case DW_OP_addrx: /* DWARF5: unsigned val */
242  case DW_OP_GNU_const_index:
243  case DW_OP_constx: /* DWARF5: unsigned val */
244  case DW_OP_const1u:
245  case DW_OP_const2u:
246  case DW_OP_const4u:
247  case DW_OP_const8u:
248  case DW_OP_constu:
249  case DW_OP_pick:
250  case DW_OP_plus_uconst:
251  case DW_OP_regx:
252  case DW_OP_piece:
253  case DW_OP_deref_size:
254  case DW_OP_xderef_size:
255  snprintf(buf, sizeof(buf), " %u", opd1);
256  strcat(*stringOut, buf);
257  break;
258  case DW_OP_bregx:
259  snprintf(buf, sizeof(buf), " <0x%08x> %+d", (Dwarf_Unsigned) opd1, (Dwarf_Signed)opd2);
260  strcat(*stringOut, buf);
261  break;
262  case DW_OP_call2:
263  case DW_OP_call4:
264  case DW_OP_call_ref:
265  snprintf(buf, sizeof(buf), " <0x%08x>", (Dwarf_Unsigned) opd1);
266  strcat(*stringOut, buf);
267  case DW_OP_bit_piece:
268  snprintf(buf, sizeof(buf), " <0x%08x> offset <0x%08x>", (Dwarf_Unsigned) opd1, (Dwarf_Unsigned)opd2);
269  strcat(*stringOut, buf);
270  break;
271  case DW_OP_implicit_value:
272  {
273 #define IMPLICIT_VALUE_PRINT_MAX 12
274  unsigned int len = 0;
275  snprintf(buf, sizeof(buf), " <0x%08x>", (Dwarf_Unsigned) opd1);
276  strcat(*stringOut, buf);
277  /* The other operand is a block of opd1 bytes. */
278  /* FIXME */
279  len = opd1;
280  if (len > IMPLICIT_VALUE_PRINT_MAX) {
281  len = IMPLICIT_VALUE_PRINT_MAX;
282  }
283 #undef IMPLICIT_VALUE_PRINT_MAX
284  {
285  const unsigned char *bp = 0;
286  /* This is a really ugly cast, a way
287  to implement DW_OP_implicit value in
288  this libdwarf context. */
289  bp = (const unsigned char *) opd2;
290  showContents(stringOut, len, bp);
291  }
292  }
293  break;
294 
295  /* We do not know what the operands, if any, are. */
296  case DW_OP_HP_unknown:
297  case DW_OP_HP_is_value:
298  case DW_OP_HP_fltconst4:
299  case DW_OP_HP_fltconst8:
300  case DW_OP_HP_mod_range:
301  case DW_OP_HP_unmod_range:
302  case DW_OP_HP_tls:
303  case DW_OP_INTEL_bit_piece:
304  case DW_OP_stack_value: /* DWARF4 */
305  break;
306  case DW_OP_GNU_uninit: /* DW_OP_APPLE_uninit */
307  /* No operands. */
308  break;
309  case DW_OP_GNU_encoded_addr:
310  snprintf(buf, sizeof(buf), " <0x%08x>", (Dwarf_Unsigned) opd1);
311  strcat(*stringOut, buf);
312  break;
313  case DW_OP_implicit_pointer: /* DWARF5 */
314  case DW_OP_GNU_implicit_pointer:
315  snprintf(buf, sizeof(buf), " <0x%08x> %+d", (Dwarf_Unsigned) opd1, (Dwarf_Signed)opd2);
316  strcat(*stringOut, buf);
317  break;
318  case DW_OP_entry_value: /* DWARF5 */
319  case DW_OP_GNU_entry_value:
320  {
321  const unsigned char *bp = 0;
322  unsigned int length = 0;
323 
324  length = opd1;
325  snprintf(buf, sizeof(buf), " <0x%08x>", (Dwarf_Unsigned) opd1);
326  strcat(*stringOut, buf);
327  bp = (Dwarf_Small *) opd2;
328  if (!bp) {
329  fprintf(stderr, "ERROR: Null databyte pointer DW_OP_entry_value ");
330  } else {
331  showContents(stringOut, length, bp);
332  }
333  }
334  break;
335  case DW_OP_const_type: /* DWARF5 */
336  case DW_OP_GNU_const_type:
337  {
338  const unsigned char *bp = 0;
339  unsigned int length = 0;
340 
341  length = opd2;
342  snprintf(buf, sizeof(buf), " <0x%08x> const length: %u", (Dwarf_Unsigned) opd1, length);
343  strcat(*stringOut, buf);
344  strcat(*stringOut," const length: ");
345 
346  /* Now point to the data bytes of the const. */
347  bp = (Dwarf_Small *) opd3;
348  if (!bp) {
349  fprintf(stderr, "ERROR: Null databyte pointer DW_OP_const_type ");
350  } else {
351  showContents(stringOut, length, bp);
352  }
353  }
354  break;
355  case DW_OP_regval_type: /* DWARF5 */
356  case DW_OP_GNU_regval_type:
357  {
358  snprintf(buf, sizeof(buf), " 0x%08x <0x%08x>", opd1, opd2);
359  strcat(*stringOut, buf);
360  }
361  break;
362  case DW_OP_deref_type: /* DWARF5 */
363  case DW_OP_GNU_deref_type:
364  {
365  snprintf(buf, sizeof(buf), " 0x%02x <0x%08x>", opd1, opd2);
366  strcat(*stringOut, buf);
367  }
368  break;
369  case DW_OP_convert: /* DWARF5 */
370  case DW_OP_GNU_convert:
371  case DW_OP_reinterpret: /* DWARF5 */
372  case DW_OP_GNU_reinterpret:
373  case DW_OP_GNU_parameter_ref:
374  snprintf(buf, sizeof(buf), " 0x%02x", opd1);
375  strcat(*stringOut, buf);
376  break;
377  default:
378  {
379  snprintf(buf, sizeof(buf), " dwarf_op unknown 0x%x", (unsigned)op);
380  strcat(*stringOut, buf);
381  }
382  break;
383  }
384  }
385  }
386  return DWARF_DBG_ERR_OK;
387 }

391 static uint8_t handleLocationExprloc(dwarfDbgPtr_t self, attrInInfo_t *attrInInfo) {
392  uint8_t result;
393  int res = 0;
394  Dwarf_Error err = NULL;
395  Dwarf_Unsigned tempud = 0;
396  Dwarf_Ptr x = 0;
397  char extraBuf[1024];
398  char *extraBufPtr = extraBuf;
399  char buf[50];
400  int u = 0;
401 
402  result = DWARF_DBG_ERR_OK;
403  if (attrInInfo->dieAttr->theform != DW_FORM_exprloc) {
404  return DWARF_DBG_ERR_BAD_FRAME_BASE_FORM;
405  }
406  res = dwarf_formexprloc(attrInInfo->dieAttr->attrIn, &tempud, &x, &err);
407  if (res != DW_DLV_OK) {
408  return DWARF_DBG_ERR_CANNOT_GET_FORM_EXPRLOC;
409  }
410  attrInInfo->dieAttr->uval = tempud;
411  extraBuf[0] = '\0';
412  buf[0] = '\0';
413  snprintf(buf, sizeof(buf), " len 0x%04x: ", tempud);
414  strcat(extraBuf, buf);
415  for (u = 0; u < tempud; u++) {
416  snprintf(buf, sizeof(buf), "%02x", *(u + (unsigned char *) x));
417  strcat(extraBuf, buf);
418  }
419  strcat(extraBuf, ": ");
420 DWARF_DBG_PRINT(self, "L", 1, "%s", extraBuf);
421  extraBuf[0] = '\0';
422  {
423  Dwarf_Half addressSize = 0;
424  Dwarf_Half offsetSize = 0;
425  Dwarf_Half version = 0;
426  Dwarf_Loc_Head_c head = 0;
427  Dwarf_Locdesc_c locentry = 0;
428  Dwarf_Unsigned lopc = 0;
429  Dwarf_Unsigned hipc = 0;
430  Dwarf_Unsigned ulocentryCount = 0;
431  Dwarf_Unsigned sectionOffset = 0;
432  Dwarf_Unsigned locdescOffset = 0;
433  Dwarf_Small lleValue = 0;
434  Dwarf_Small loclistSource = 0;
435  Dwarf_Unsigned ulistlen = 0;
436  int i = 0;
437 
438  res = dwarf_get_version_of_die(attrInInfo->compileUnit->compileUnitDie, &version, &offsetSize);
439  if (res != DW_DLV_OK) {
440 //char * errmsg = dwarf_errmsg(err);
441  return DWARF_DBG_ERR_GET_VERSION_OF_DIE;
442  }
443  res = dwarf_get_die_address_size(attrInInfo->compileUnit->compileUnitDie, &addressSize, &err);
444  if (res != DW_DLV_OK) {
445  return DWARF_DBG_ERR_GET_DIE_ADDRESS_SIZE;
446  }
447  res = dwarf_loclist_from_expr_c(self->elfInfo.dbg, x, tempud, addressSize, offsetSize,
448  version, &head, &ulistlen, &err);
449  if (res != DW_DLV_OK) {
450  return DWARF_DBG_ERR_GET_LOCLIST_FROM_EXPR_C;
451  }
452  res = dwarf_get_locdesc_entry_c(head, 0, /* Data from 0th LocDesc */ &lleValue, &lopc, &hipc,
453  &ulocentryCount, &locentry, &loclistSource, &sectionOffset, &locdescOffset, &err);
454  if (res != DW_DLV_OK) {
455  return DWARF_DBG_ERR_GET_LOCDESC_ENTRY_C;
456  }
457  // allocate all needed memory here as we know how many entries
458  attrInInfo->dieAttr->locationInfo->maxLocEntry = ulocentryCount;
459  attrInInfo->dieAttr->locationInfo->numLocEntry = ulocentryCount;
460  attrInInfo->dieAttr->locationInfo->locationOps = (locationOp_t *)ckalloc(sizeof(locationOp_t) * ulocentryCount);
461  if (attrInInfo->dieAttr->locationInfo->locationOps == NULL) {
462  return DWARF_DBG_ERR_OUT_OF_MEMORY;
463  }
464  for (i = 0; i < ulocentryCount; i++) {
465  Dwarf_Small op = 0;
466  Dwarf_Unsigned opd1 = 0;
467  Dwarf_Unsigned opd2 = 0;
468  Dwarf_Unsigned opd3 = 0;
469  Dwarf_Unsigned offsetForBranch = 0;
470  locationOp_t *locationOp = NULL;
471  const char *opName = 0;
472 
473  res = dwarf_get_location_op_value_c(locentry, i, &op, &opd1, &opd2, &opd3, &offsetForBranch, &err);
474 //printf("op: 0x%04x opd1: %d opd2: %d opd3: %d offsetForBranch: %d\n", op, opd1, opd2, opd3, offsetForBranch);
475  if (res != DW_DLV_OK) {
476  return DWARF_DBG_ERR_CANNOT_GET_LOCATION_OP_VALUE_C;
477  }
478  locationOp = &attrInInfo->dieAttr->locationInfo->locationOps[i];
479  locationOp->op = op;
480  locationOp->opd1 = opd1;
481  locationOp->opd2 = opd2;
482  locationOp->opd3 = opd3;
483  locationOp->offsetForBranch = offsetForBranch;
484  result = handleOneExprOp(self, locationOp, i, &extraBufPtr);
485  checkErrOK(result);
486 DWARF_DBG_PRINT(self, "A", 1, " %s", extraBuf);
487  }
488  }
489  return result;
490 }

534  char small_buf[150];
