56 static uint8_t addRangeInfo(dwarfDbgPtr_t self, Dwarf_Addr dwr_addr1, Dwarf_Addr dwr_addr2, enum Dwarf_Ranges_Entry_Type dwrType, int *rangeInfoIdx) {
57  uint8_t result;
58  compileUnit_t *compileUnit;
59  rangeInfo_t *rangeInfo;
60  const char *dwrTypeStr;
61 
62  result = self->dwarfDbgStringInfo->getDW_RANGES_TYPE_string(self, dwrType, &dwrTypeStr);
63  checkErrOK(result);
64  DWARF_DBG_PRINT(self, "R", 1, " dwr1: 0x%08x dwr2: 0x%08x dwrType: %s\n", dwr_addr1, dwr_addr2, dwrTypeStr);
65  result = DWARF_DBG_ERR_OK;
66  compileUnit = self->dwarfDbgCompileUnitInfo->currCompileUnit;
67  if (compileUnit->fileInfos == NULL) {
68  // seems to be no file infos!!
69  return result;
70  }
71  if (compileUnit->maxRangeInfo <= compileUnit->numRangeInfo) {
72  compileUnit->maxRangeInfo += 5;
73  if (compileUnit->rangeInfos == NULL) {
74  compileUnit->rangeInfos = (rangeInfo_t *)ckalloc(sizeof(rangeInfo_t) * compileUnit->maxRangeInfo);
75  if (compileUnit->rangeInfos == NULL) {
76  return DWARF_DBG_ERR_OUT_OF_MEMORY;
77  }
78  } else {
79  compileUnit->rangeInfos = (rangeInfo_t *)ckrealloc((char *)compileUnit->rangeInfos, sizeof(rangeInfo_t) * compileUnit->maxRangeInfo);
80  if (compileUnit->rangeInfos == NULL) {
81  return DWARF_DBG_ERR_OUT_OF_MEMORY;
82  }
83  }
84  }
85  rangeInfo = &compileUnit->rangeInfos[compileUnit->numRangeInfo];
86  rangeInfo->dwr_addr1 = dwr_addr1;
87  rangeInfo->dwr_addr2 = dwr_addr2;
88  rangeInfo->dwr_type = dwrType;
89  *rangeInfoIdx = compileUnit->numRangeInfo;
90  compileUnit->numRangeInfo++;
91  return result;
92 }

96 static uint8_t handleRangeInfos(dwarfDbgPtr_t self, Dwarf_Attribute attrIn) {
97  uint8_t result;
98  int rres = 0;
99  int fres = 0;
100  Dwarf_Ranges *rangeset = 0;
101  Dwarf_Signed rangecount = 0;
102  Dwarf_Unsigned bytecount = 0;
103  Dwarf_Unsigned original_off = 0;
104  Dwarf_Error err;
105  compileUnit_t *compileUnit;
106  int i;
107  int rangeIdx = 0;
108 
109  result = DWARF_DBG_ERR_OK;
110  compileUnit = self->dwarfDbgCompileUnitInfo->currCompileUnit;
111  fres = dwarf_global_formref(attrIn, &original_off, &err);
112  if (fres != DW_DLV_OK) {
113  return DWARF_DBG_ERR_CANNOT_GET_GLOBAL_FORMREF;
114  }
115  if (fres == DW_DLV_OK) {
116  rres = dwarf_get_ranges_a(self->elfInfo.dbg, original_off, compileUnit->compileUnitDie, &rangeset, &rangecount, &bytecount, &err);
117  DWARF_DBG_PRINT(self, "R", 1, " count: %d bytecount: %d\n", rangecount, bytecount);
118  for (i = 0; i < rangecount; i++) {
119  Dwarf_Ranges *range = &rangeset[i];
120  result = self->dwarfDbgRangeInfo->addRangeInfo(self, range->dwr_addr1, range->dwr_addr2, range->dwr_type, &rangeIdx);
121  // FIXME need to store rangeIdx somwhere!!
122  }
123  }
124  return result;
125 }
