52 static uint8_t addTypeStr(dwarfDbgPtr_t self, const char *str, int *typeStrIdx) {
53  uint8_t result;
54  char **typeStr;
55  int strIdx;
56 
57  result = DWARF_DBG_ERR_OK;
58  for(strIdx = 0; strIdx < self->dwarfDbgTypeInfo->numTypeStr; strIdx++) {
59  if (strcmp(self->dwarfDbgTypeInfo->typeStrs[strIdx], str) == 0) {
60  *typeStrIdx = strIdx;
61  return result;
62  }
63  }
64  if (self->dwarfDbgTypeInfo->maxTypeStr <= self->dwarfDbgTypeInfo->numTypeStr) {
65  self->dwarfDbgTypeInfo->maxTypeStr += 10;
66  if (self->dwarfDbgTypeInfo->typeStrs == NULL) {
67  self->dwarfDbgTypeInfo->typeStrs = (char **)ckalloc(sizeof(char *) * self->dwarfDbgTypeInfo->maxTypeStr);
68  if (self->dwarfDbgTypeInfo->typeStrs == NULL) {
69  return DWARF_DBG_ERR_OUT_OF_MEMORY;
70  }
71  } else {
72  self->dwarfDbgTypeInfo->typeStrs = (char **)ckrealloc((char *)self->dwarfDbgTypeInfo->typeStrs, sizeof(char *) * self->dwarfDbgTypeInfo->maxTypeStr);
73  if (self->dwarfDbgTypeInfo->typeStrs == NULL) {
74  return DWARF_DBG_ERR_OUT_OF_MEMORY;
75  }
76  }
77  }
78 //printf("== numTypeStrs: %d %s\n", self->dwarfDbgTypeInfo->numTypeStr, str);
79  typeStr = &self->dwarfDbgTypeInfo->typeStrs[self->dwarfDbgTypeInfo->numTypeStr];
80  *typeStr = ckalloc(strlen(str) + 1);
81  memset(*typeStr, 0, strlen(str) + 1);
82  memcpy(*typeStr, str, strlen(str));
83  *typeStrIdx = self->dwarfDbgTypeInfo->numTypeStr;
84  self->dwarfDbgTypeInfo->numTypeStr++;
85  return result;
86 }
