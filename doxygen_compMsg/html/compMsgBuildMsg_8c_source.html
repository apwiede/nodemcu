109 static uint8_t fixOffsetsForKeyValues(compMsgDispatcher_t *self) {
110  uint8_t result;

121  bool found;
122 
123  compMsgData = self->compMsgData;
124  fieldIdx = 0;
125  msgDescPartIdx = 0;
126 //ets_printf("§fixOffsetsForKeyValues: numFields: %d!§", compMsgData->numFields);
127  while (fieldIdx < compMsgData->numFields) {
128  fieldInfo = &compMsgData->fields[fieldIdx];
129  msgDescPart = &self->compMsgData->msgDescParts[msgDescPartIdx];
130  self->compMsgData->msgDescPart = msgDescPart;
131  msgKeyValueDescPart = NULL;
132  if (msgDescPart->fieldNameStr[0] == '#') {
133  // get the corresponding msgKeyValueDescPart

136  while (keyValueIdx < self->numMsgKeyValueDescParts) {
137  msgKeyValueDescPart = &self->msgKeyValueDescParts[keyValueIdx];
138  if (c_strcmp(msgKeyValueDescPart->keyNameStr, msgDescPart->fieldNameStr) == 0) {
139  found = true;
140  break;
141  }

143  }
144  if (!found) {
145  msgKeyValueDescPart = NULL;
146  }
147  }

149  if (msgDescPart->fieldSizeCallback != NULL) {
150  // the key name must have the prefix: "#key_"!
151  if (msgDescPart->fieldNameStr[0] != '#') {
152  return COMP_MSG_ERR_FIELD_NOT_FOUND;
153  }
154  result = msgDescPart->fieldSizeCallback(self, &numericValue, &stringValue);
155 //ets_printf("§fieldSizeCallback for: %s numericValue: %d stringValue: %p\n§", msgDescPart->fieldNameStr, numericValue, stringValue);
156  checkErrOK(result);
157  if (msgKeyValueDescPart != NULL) {
158  fieldInfo->fieldKey = msgKeyValueDescPart->keyId;
159 //ets_printf("fieldKey: %d\n", msgKeyValueDescPart->keyId);
160  } else {
161  fieldInfo->fieldKey = msgDescPart->fieldKey;
162  }
163  fieldInfo->fieldLgth = msgDescPart->fieldSize + 2 * sizeof(uint16_t) + sizeof(uint8_t); // for key, type and lgth in front of value!!
164 //ets_printf("§fixOffsetsForKeyValues: %s size: %d lgth: %d\n§", msgDescPart->fieldNameStr, msgDescPart->fieldSize, fieldInfo->fieldLgth);
165  } else {
166  if (msgKeyValueDescPart != NULL) {
167  fieldInfo->fieldKey = msgKeyValueDescPart->keyId;
168  msgDescPart->fieldSize = msgKeyValueDescPart->keyLgth;
169  fieldInfo->fieldLgth = msgDescPart->fieldSize + 2 * sizeof(uint16_t) + sizeof(uint8_t); // for key, type and lgth in front of value!!
170  }
171  }
172  msgDescPartIdx++;
173  fieldIdx++;
174  }
175 //ets_printf("§fixOffsetsForKeyValues: done!\n§");
176  return COMP_MSG_ERR_OK;
177 }

347  uint8_t *stringValue;
348  compMsgField_t *fieldInfo;
349  int src;
350  int dst;
351 
352  // at this point an eventual callback for getting the values
353  // has been already done using runAction in createMsgFromHeaderPart
354  // so now we can fix the offsets if needed for key value list entries
355  // we can do that in looking for a special key entry @numKeyValues in msgDescParts
356  // could be also done in looking in compMsgData->fields
357  // to get the desired keys we look in msgValParts for fieldNames starting with '#'
358  // the fieldValueStr ther is a callBackFunction for building the key value entries
359  // the entry @numKeyValues in msgValParts tells us how many different keys follow
360  // a key value entry is built like so:
361  // uint16_t key
362  // uint16_t length o value
363  // uint8_t* the bytes of the value
364  // this could if needed also be an array of uint16_t etc. depending on the key
365  // the receiver must know how the value is built depending on the key!!
366 
367 //ets_printf("§buildMsg\n§");
368  result = self->compMsgBuildMsg->fixOffsetsForKeyValues(self);
369  checkErrOK(result);
370  self->compMsgData->direction = COMP_MSG_TO_SEND_DATA;
371  result = self->compMsgData->initMsg(self);
372 //ets_printf("§heap2: %d§", system_get_free_heap_size());
373  result = setMsgValues(self);
374  checkErrOK(result);
375 
376  result = self->compMsgData->getMsgData(self, &msgData, &msgLgth);
377 //ets_printf("§getMsgData res: %d!msgLgth: %d!\n§", result, msgLgth);
378  checkErrOK(result);
379 //ets_printf("§");
380 //self->compMsgData->dumpMsg(self);
381 //ets_printf("§");
382 //ets_printf("§");
383 //self->compMsgData->compMsgDataView->dataView->dumpBinary(msgData, msgLgth, "dumpMsg");
384 //ets_printf("§");
385 //ets_printf("§encryption: %c\n§", self->compMsgData->currHdr->hdrEncryption);
386  if (self->compMsgData->currHdr->hdrEncryption == 'E') {
387  uint8_t *toCryptPtr;
388  uint8_t *encryptedMsgData;
389  size_t encryptedMsgDataLgth;
390  uint16_t mlen;
391  uint8_t headerLgth;
392  size_t totalCrcOffset;
393 
394  cryptKey = "a1b2c3d4e5f6g7h8";
395  ivlen = 16;
396  klen = 16;
397 
398 //self->compMsgData->compMsgDataView->dataView->dumpBinary(self->compMsgData->compMsgDataView->dataView->data, self->compMsgData->compMsgDataView->dataView->lgth, "MSG_AA");
399 //ets_printf("need to encrypt message!\n");

432 //ets_printf("§buildMsg sendMsg has been called result: %d§", result);

434 // result = self->resetMsgInfo(self, self->buildMsgInfos.parts);
