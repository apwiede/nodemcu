54 /**
55  * \brief fix the field offsets of all if we have key/value fields after calling the callbacks for these fields.
56  * \param self The dispatcher struct
57  * \return Error code or ErrorOK
58  *
59  */
60 static uint8_t fixOffsetsForKeyValues(compMsgDispatcher_t *self) {
61  uint8_t result;
62  uint8_t msgDescPartIdx;
63  uint8_t fieldIdx;
64  uint8_t type;
65  compMsgField_t *fieldInfo;
66  msgDescPart_t *msgDescPart;
67  compMsgData_t *compMsgData;
68 
69  compMsgData = self->compMsgData;
70  fieldIdx = 0;
71  msgDescPartIdx = 0;
72 //ets_printf("§fix: numFields: %d!§", compMsgData->numFields);
73  while (fieldIdx < compMsgData->numFields) {
74  fieldInfo = &compMsgData->fields[fieldIdx];
75  msgDescPart = &self->compMsgData->msgDescParts[msgDescPartIdx];
76  self->compMsgData->msgDescPart = msgDescPart;
77 //ets_printf("§fix: idx: %d!%p!%s!§", fieldIdx, msgDescPart->fieldSizeCallback, msgDescPart->fieldNameStr);
78  if (msgDescPart->fieldSizeCallback != NULL) {
79  // the key name must have the prefix: "#key_"!
80  if (msgDescPart->fieldNameStr[0] != '#') {
81  return COMP_DISP_ERR_FIELD_NOT_FOUND;
82  }
83  result = msgDescPart->fieldSizeCallback(self);
84  checkErrOK(result);
85  fieldInfo->fieldKey = msgDescPart->fieldKey;
86  msgDescPart->fieldSize += 2 * sizeof(uint16_t) + sizeof(uint8_t); // for key, type and lgth in front of value!!
87  fieldInfo->fieldLgth = msgDescPart->fieldSize;
88  }
89  msgDescPartIdx++;
90  fieldIdx++;
91  }
92 //ets_printf("§fix: done!§");
93  return COMP_DISP_ERR_OK;
94 }

105 static uint8_t setMsgFieldValue(compMsgDispatcher_t *self, uint8_t type) {
106  uint8_t result;
107  uint8_t *fieldNameStr;
108  uint8_t *stringValue;
109  int numericValue;
110  compMsgData_t *compMsgData;
111 
112 //ets_printf("§setMsgFieldValue: %s %s§", self->compMsgData->msgValPart->fieldNameStr, self->compMsgData->msgValPart->fieldValueStr);
113  compMsgData = self->compMsgData;
114  if (self->compMsgData->msgValPart->fieldValueStr[0] == '@') {
115  // call the callback function for the field!!
116 //ets_printf("§setMsgFieldValue:cb %p§", self->compMsgData->msgValPart->fieldValueCallback);
117  if (self->compMsgData->msgValPart->fieldValueCallback != NULL) {
118  result = self->compMsgData->msgValPart->fieldValueCallback(self);
119  checkErrOK(result);
120  }
121  fieldNameStr = self->compMsgData->msgValPart->fieldNameStr;
122  if (self->compMsgData->msgValPart->fieldFlags & COMP_DISP_DESC_VALUE_IS_NUMBER) {
123  stringValue = NULL;
124  numericValue = self->compMsgData->msgValPart->fieldValue;
125  } else {
126  stringValue = self->compMsgData->msgValPart->fieldKeyValueStr;
127  numericValue = 0;
128  }
129 //ets_printf("§cb field: %s!value: 0x%04x %s!§", fieldNameStr, numericValue, stringValue == NULL ? "nil" : (char *)stringValue);
130  result = self->compMsgData->setFieldValue(self, fieldNameStr, numericValue, stringValue);
131  } else {
132  fieldNameStr = self->compMsgData->msgValPart->fieldNameStr;
133 //ets_printf("§fieldName: %s!id: %d!§", fieldNameStr, self->compMsgData->msgValPart->fieldNameId);
134  if (self->compMsgData->msgValPart->fieldFlags & COMP_DISP_DESC_VALUE_IS_NUMBER) {
135  stringValue = NULL;
136  numericValue = self->compMsgData->msgValPart->fieldValue;
137  } else {
138  stringValue = self->compMsgData->msgValPart->fieldValueStr;
139  numericValue = 0;
140  }
141  switch (self->compMsgData->msgValPart->fieldNameId) {
142  case COMP_MSG_SPEC_FIELD_DST:
143  result = compMsgData->setFieldValue(self, fieldNameStr, numericValue, stringValue);
144  break;
145  case COMP_MSG_SPEC_FIELD_SRC:
146  result = compMsgData->setFieldValue(self, fieldNameStr, numericValue, stringValue);
147  break;
148  case COMP_MSG_SPEC_FIELD_CMD_KEY:
149  numericValue = self->compMsgData->currHdr->hdrU16CmdKey;
150  stringValue = NULL;
151 //ets_printf("§cmdKey value: 0x%04x§", numericValue);
152  result = compMsgData->setFieldValue(self, fieldNameStr, numericValue, stringValue);
153  break;
154  default:
155 //ets_printf("§fieldName: %s!value: 0x%04x %s§", fieldNameStr, numericValue, stringValue == NULL ? "nil" : (char *)stringValue);
156  result = self->compMsgData->setFieldValue(self, fieldNameStr, numericValue, stringValue);
157  break;
158  }
159  checkErrOK(result);
160  }
161 //ets_printf("§setMsgFieldValue: done§");
162  return COMP_DISP_ERR_OK;
163 }

243  * \brief Build a message for sending to Uart or socket
244  *
245  * \param self The dispatcher struct
246  * \return Error code or ErrorOK
247  *
248  */
249 static uint8_t buildMsg(compMsgDispatcher_t *self) {
250  uint8_t result;
251  size_t msgLgth;
252  uint8_t *msgData;
253  size_t defLgth;
254  uint8_t *defData;
255  uint8_t *cryptKey;
256  uint8_t klen;
257  uint8_t ivlen;
258  uint8_t *stringValue;
259  int src;
260  int dst;
261 
262 //ets_printf("§buildMsg§");
263  // at this point an eventual callback for getting the values
264  // has been already done using runAction in createMsgFromHeaderPart
265  // so now we can fix the offsets if needed for key value list entries
266  // we can do that in looking for a special key entry @numKeyValues in msgDescParts
267  // could be also done in looking in compMsgData->fields
268  // to get the desired keys we look in msgValParts for fieldNames starting with '#'
269  // the fieldValueStr ther is a callBackFunction for building the key value entries
270  // the entry @numKeyValues in msgValParts tells us how many different keys follow
271  // a key value entry is built like so:
272  // uint16_t key
273  // uint16_t length o value
274  // uint8_t* the bytes of the value
275  // this could if needed also be an array of uint16_t etc. depending on the key
276  // the receiver must know how the value is built depending on the key!!
277 
278 //ets_printf("§buildMsg§");
279  result = self->fixOffsetsForKeyValues(self);
280  checkErrOK(result);
281  self->compMsgData->direction = COMP_MSG_TO_SEND_DATA;
282  result = self->compMsgData->initMsg(self);
283 //ets_printf("§heap2: %d§", system_get_free_heap_size());
284  result = setMsgValues(self);
285  checkErrOK(result);
286 
287  result = self->compMsgData->getMsgData(self, &msgData, &msgLgth);
288 //ets_printf("§getMsgData res: %d!msgLgth: %d!§", result, msgLgth);
289  checkErrOK(result);
290 //ets_printf("§");
291 //self->compMsgData->compMsgDataView->dataView->dumpBinary(msgData, msgLgth, "dumpMsg");
292 //ets_printf("§");
293  if (self->compMsgData->currHdr->hdrEncryption == 'E') {
294  uint8_t *toCryptPtr;
295  uint8_t *encryptedMsgData;
296  size_t encryptedMsgDataLgth;
297  uint16_t mlen;
298  uint8_t headerLgth;
299 
300  cryptKey = "a1b2c3d4e5f6g7h8";
301  ivlen = 16;
302  klen = 16;
303 
304 //self->compMsgData->compMsgDataView->dataView->dumpBinary(self->compMsgData->compMsgDataView->dataView->data, self->compMsgData->compMsgDataView->dataView->lgth, "MSG_AA");
305 //ets_printf("need to encrypt message!\n");
306  headerLgth = self->compMsgData->headerLgth;
307  mlen = self->compMsgData->totalLgth - headerLgth - 1; // -1 for totalCrc !!
308 //ets_printf("msglen!%d!mlen: %d, headerLgth!%d\n", self->compMsgData->totalLgth, mlen, self->compMsgData->headerLgth);
309  toCryptPtr = msgData + self->compMsgData->headerLgth;
310  result = self->encryptMsg(toCryptPtr, mlen, cryptKey, klen, cryptKey, ivlen, &encryptedMsgData, &encryptedMsgDataLgth);
311  checkErrOK(result);
312  if (encryptedMsgDataLgth != mlen) {
313 ets_printf("WARNING! mlen: %d encryptedMsgDataLgth: %d overwrites eventually totalCrc!\n", mlen, encryptedMsgDataLgth);
314  }
315  c_memcpy(toCryptPtr, encryptedMsgData, encryptedMsgDataLgth);
316 ets_printf("crypted: len: %d!mlen: %d!\n", encryptedMsgDataLgth, mlen);
317  }
318 
319  // here we need to decide where and how to send the message!!
320  // from currHdr we can see the handle type and - if needed - the @dst
321 //ets_printf("§transferType: %c dst: 0x%04x§", self->compMsgData->currHdr->hdrHandleType, self->compMsgData->currHdr->hdrToPart);
322  result = self->sendMsg(self, msgData, msgLgth);
323 //ets_printf("§buildMsg sendMsg has been called result: %d§", result);
324  checkErrOK(result);
325 // result = self->resetMsgInfo(self, self->buildMsgInfos.parts);
326  return result;
327 }
