94 static uint8_t setFiller(compMsgDataView_t *self, compMsgField_t *fieldInfo) {
95  uint32_t val;
96  int idx;
97  int lgth;
98  int result;
99  size_t offset;
100 
101  lgth = fieldInfo->fieldLgth;

105  val = (uint32_t)(rand() & RAND_MAX);
106  result = self->dataView->setUint32(self->dataView, offset, val);
107  checkErrOK(result);
108  offset += 4;
109  lgth -= 4;

131 static uint8_t getCrc(compMsgDataView_t *self, compMsgField_t *fieldInfo, size_t startOffset, size_t lgth) {
132  uint16_t crcVal;
133  uint16_t crc;
134  uint8_t uint8_crc;
135  int crcLgth;
136  int idx;
137  int result;
138 
139 //ets_printf("getCrc: startOffset: %d lgth: %d\n", startOffset, lgth);
140  crcLgth = fieldInfo->fieldLgth;
141  crcVal = 0;
142  idx = startOffset;
143  while (idx < lgth) {
144 //ets_printf("crc idx: %d ch: 0x%02x crc: 0x%04x\n", idx-startOffset, self->dataView->data[idx], crcVal);
145  crcVal += self->dataView->data[idx++];
146  }
147 //ets_printf("crc idx: %d ch: 0x%02x crc: 0x%04x\n", idx-startOffset, self->dataView->data[idx], crcVal);
148 //ets_printf("§crcVal00: 0x%04x§\n", crcVal);
149  crcVal = ~(crcVal);
150  if (crcLgth == 1) {
151 //ets_printf("§crcVal10: 0x%04x§\n", crcVal);
152  crcVal = crcVal & 0xFF;
153  result = self->dataView->getUint8(self->dataView, fieldInfo->fieldOffset, &uint8_crc);
154  checkErrOK(result);
155 //ets_printf("§crcVal1: 0x%02x crc: 0x%02x§\n", crcVal, uint8_crc);
156  if (crcVal != uint8_crc) {
157 ets_printf("§bad crcVal1: 0x%02x crc: 0x%02x§\n", crcVal, uint8_crc);
158  return COMP_MSG_ERR_BAD_CRC_VALUE;
159  }
160  } else {
161  result = self->dataView->getUint16(self->dataView, fieldInfo->fieldOffset, &crc);
162  checkErrOK(result);
163 //ets_printf("§crcVal2: 0x%04x crc: 0x%04x§\n", crcVal, crc);
164  if (crcVal != crc) {
165 ets_printf("§bad crcVal2: 0x%04x crc: 0x%04x§\n", crcVal, crc);
166  return COMP_MSG_ERR_BAD_CRC_VALUE;
167  }
168  }
169  return DATA_VIEW_ERR_OK;
170 }

264 static uint8_t getFieldValue(compMsgDataView_t *self, compMsgField_t *fieldInfo, int *numericValue, uint8_t **stringValue, int fieldIdx) {
265  int idx;
266  int result;
267  int numEntries;
268  int8_t i8;
269  uint8_t ui8;
270  int16_t i16;
271  uint16_t ui16;
272  int32_t i32;
273  uint32_t ui32;
274 
275  *stringValue = NULL;
276  *numericValue = 0;
277  switch (fieldInfo->fieldTypeId) {
278  case DATA_VIEW_FIELD_INT8_T:
279  result = self->dataView->getInt8(self->dataView, fieldInfo->fieldOffset, &i8);
280  checkErrOK(result);
281  *numericValue = (int)i8;
282  break;
283  case DATA_VIEW_FIELD_UINT8_T:
284  result = self->dataView->getUint8(self->dataView, fieldInfo->fieldOffset, &ui8);
285  checkErrOK(result);
286  *numericValue = (int)ui8;
287  break;
288  case DATA_VIEW_FIELD_INT16_T:
289  result = self->dataView->getInt16(self->dataView, fieldInfo->fieldOffset, &i16);
290  checkErrOK(result);
291  *numericValue = (int)i16;
292  break;
293  case DATA_VIEW_FIELD_UINT16_T:
294  result = self->dataView->getUint16(self->dataView, fieldInfo->fieldOffset, &ui16);
295  checkErrOK(result);
296  *numericValue = (int)ui16;
297  break;
298  case DATA_VIEW_FIELD_INT32_T:
299  result = self->dataView->getInt32(self->dataView, fieldInfo->fieldOffset, &i32);
300  checkErrOK(result);
301  *numericValue = (int)i32;
302  break;
303  case DATA_VIEW_FIELD_UINT32_T:
304  result = self->dataView->getUint32(self->dataView, fieldInfo->fieldOffset, &ui32);
305  checkErrOK(result);
306  *numericValue = (int)ui32;
307  break;
308  case DATA_VIEW_FIELD_INT8_VECTOR:
309  *numericValue = fieldInfo->fieldLgth;
310  *stringValue = os_zalloc(fieldInfo->fieldLgth+1);
311  checkAllocOK(stringValue);
312  (*stringValue)[fieldInfo->fieldLgth] = 0;
313  os_memcpy(*stringValue, self->dataView->data + fieldInfo->fieldOffset, fieldInfo->fieldLgth);
314  break;
315  case DATA_VIEW_FIELD_UINT8_VECTOR:
316  *stringValue = os_zalloc(fieldInfo->fieldLgth+1);
317  checkAllocOK(stringValue);
318  (*stringValue)[fieldInfo->fieldLgth] = 0;
319  os_memcpy(*stringValue, self->dataView->data + fieldInfo->fieldOffset, fieldInfo->fieldLgth);
320  break;
321  case DATA_VIEW_FIELD_INT16_VECTOR:
322  if (*stringValue == NULL) {
323  // check for length needed!!
324  result = self->dataView->getInt16(self->dataView, fieldInfo->fieldOffset + fieldIdx * sizeof(int16_t), &i16);
325  checkErrOK(result);
326  *numericValue = (int)i16;
327  } else {
328  return COMP_MSG_ERR_BAD_VALUE;
329  }
330  break;
331  case DATA_VIEW_FIELD_UINT16_VECTOR:
332  if (*stringValue == NULL) {
333  // check for length needed!!
334  result = self->dataView->getUint16(self->dataView, fieldInfo->fieldOffset + fieldIdx * sizeof(uint16_t), &ui16);
335  checkErrOK(result);
336  *numericValue = (int)ui16;
337  } else {
338  return COMP_MSG_ERR_BAD_VALUE;
339  }
340  break;
341 #ifdef NOTDEF
342  case DATA_VIEW_FIELD_INT32_VECTOR:
343  if (stringValue != NULL) {
344  // check for length needed!!
345  os_memcpy((int8_t *)fieldInfo->value.int32Vector, stringValue, fieldInfo->fieldLgth);
346  } else {
347  return COMP_MSG_ERR_BAD_VALUE;
348  }
349  break;
350  case DATA_VIEW_FIELD_UINT32_VECTOR:
351  if (stringValue != NULL) {
352  // check for length needed!!
353  os_memcpy((uint8_t *)fieldInfo->value.uint32Vector, stringValue, fieldInfo->fieldLgth);
354  } else {
355  return COMP_MSG_ERR_BAD_VALUE;
356  }
357  break;
358 #endif
359  default:
360  return COMP_MSG_ERR_BAD_FIELD_TYPE;
361  break;
362  }
363  return DATA_VIEW_ERR_OK;
364 }

368 static uint8_t setFieldValue(compMsgDataView_t *self, compMsgField_t *fieldInfo, int numericValue, const uint8_t *stringValue, int fieldIdx) {
369  int idx;
370  int result;
371  int numEntries;
372 
373  switch (fieldInfo->fieldTypeId) {
374  case DATA_VIEW_FIELD_INT8_T:
375  if (stringValue == NULL) {
376  if ((numericValue > -128) && (numericValue < 128)) {
377  result= self->dataView->setInt8(self->dataView, fieldInfo->fieldOffset, (int8_t)numericValue);
378  checkErrOK(result);
379  } else {
380  return COMP_MSG_ERR_VALUE_TOO_BIG;
381  }
382  } else {
383  return COMP_MSG_ERR_BAD_VALUE;
384  }
385  break;
386  case DATA_VIEW_FIELD_UINT8_T:
387  if (stringValue == NULL) {
388  if ((numericValue >= 0) && (numericValue <= 256)) {
389  result= self->dataView->setUint8(self->dataView, fieldInfo->fieldOffset, (uint8_t)numericValue);
390  checkErrOK(result);
