647  checkErrOK(result);
648  fieldInfo->fieldFlags |= COMP_MSG_FIELD_IS_SET;
649  break;
650  case COMP_MSG_SPEC_FIELD_TOTAL_CRC:
651  break;
652  }
653  idx++;
654  }
655  compMsgData->flags |= COMP_MSG_IS_PREPARED;
656  return COMP_MSG_ERR_OK;
