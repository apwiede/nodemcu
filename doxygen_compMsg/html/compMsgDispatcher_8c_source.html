85 /**
86  * \brief encode message with base64
87  * \param msg The message
88  * \param len The length of the message
89  * \param encoded The out param encoded message
90  * \return Error code or ErrorOK
91  *
92  */
93 static uint8_t toBase64(const uint8_t *msg, size_t *len, uint8_t **encoded) {
94  size_t i;
95  size_t n;
96  uint8_t *q;
97  uint8_t *out;
98  uint8_t bytes64[sizeof(b64)];
99 
100  n = *len;
101  if (!n) { // handle empty string case
102  return COMP_DISP_ERR_OUT_OF_MEMORY;
103  }
104  out = (uint8_t *)os_zalloc((n + 2) / 3 * 4);
105  checkAllocOK(out);
106  c_memcpy(bytes64, b64, sizeof(b64)); //Avoid lots of flash unaligned fetches
107 
108  for (i = 0, q = out; i < n; i += 3) {
109  int a = msg[i];
110  int b = (i + 1 < n) ? msg[i + 1] : 0;
111  int c = (i + 2 < n) ? msg[i + 2] : 0;
112  *q++ = bytes64[a >> 2];
113  *q++ = bytes64[((a & 3) << 4) | (b >> 4)];
114  *q++ = (i + 1 < n) ? bytes64[((b & 15) << 2) | (c >> 6)] : BASE64_PADDING;
115  *q++ = (i + 2 < n) ? bytes64[(c & 63)] : BASE64_PADDING;
116  }
117  *q = '\0';
118  *len = q - out;
119  *encoded = out;
120  return COMP_DISP_ERR_OK;
121 }

125 static uint8_t fromBase64(const uint8_t *encodedMsg, size_t *len, uint8_t **decodedMsg) {
126  int i;
127  int n;
128  int blocks;
129  int pad;
130  const uint8 *p;
131  uint8_t unbytes64[UCHAR_MAX+1];
132  uint8_t *msg;
133  uint8_t *q;
134 
135  n = *len;
136  blocks = (n>>2);
137  pad = 0;
138  if (!n) { // handle empty string case
139  return COMP_DISP_ERR_OUT_OF_MEMORY;
140  }
141  if (n & 3) {
142  return COMP_DISP_ERR_INVALID_BASE64_STRING;
143  }
144  c_memset(unbytes64, BASE64_INVALID, sizeof(unbytes64));
145  for (i = 0; i < sizeof(b64)-1; i++) {
146  unbytes64[b64[i]] = i; // sequential so no exceptions
147  }
148  if (encodedMsg[n-1] == BASE64_PADDING) {
149  pad = (encodedMsg[n-2] != BASE64_PADDING) ? 1 : 2;
150  blocks--; //exclude padding block
151  }
152 
153  for (i = 0; i < n - pad; i++) {
154  if (!ISBASE64(encodedMsg[i])) {
155  return COMP_DISP_ERR_INVALID_BASE64_STRING;
156  }
157  }
158  unbytes64[BASE64_PADDING] = 0;
159  q = (uint8_t *) os_zalloc(1+ (3 * n / 4));
160  checkAllocOK(q);
161  msg = q;
162  for (i = 0, p = encodedMsg; i<blocks; i++) {
163  uint8 a = unbytes64[*p++];
164  uint8 b = unbytes64[*p++];
165  uint8 c = unbytes64[*p++];
166  uint8 d = unbytes64[*p++];
167  *q++ = (a << 2) | (b >> 4);
168  *q++ = (b << 4) | (c >> 2);
169  *q++ = (c << 6) | d;
170  }
171  if (pad) { //now process padding block bytes
172  uint8 a = unbytes64[*p++];
173  uint8 b = unbytes64[*p++];
174  *q++ = (a << 2) | (b >> 4);
175  if (pad == 1) *q++ = (b << 4) | (unbytes64[*p] >> 2);
176  }
177  *len = q - msg;
178  *decodedMsg = msg;
179  return COMP_DISP_ERR_OK;;
180 }

715 
716 static uint8_t deleteRequest(compMsgDispatcher_t *self, uint8_t requestType, void *requestHandle) {
717  uint8_t result;
718  int idx;
719  bool found;
720  int idxToStart;
721  int idxDeleted;
722 
723  idx = 0;
724  idxToStart = -1;
725  idxDeleted = -1;
726  found = false;
727  while (idx < self->msgRequestInfos.lastRequestIdx) {
728  if (idx >= COMP_DISP_MAX_REQUESTS) {
729  return COMP_DISP_ERR_REQUEST_NOT_FOUND;
730  }
731  if (!found) {
732  if ((self->msgRequestInfos.requestTypes[idx] == requestType) && (self->msgRequestInfos.requestHandles[idx] == requestHandle)) {
733  found = true;
734  idxDeleted = idx;
735  if (idx == self->msgRequestInfos.currRequestIdx) {
736  if (idx < self->msgRequestInfos.lastRequestIdx) {
737  idxToStart = idx;
738  }
739  self->msgRequestInfos.currRequestIdx = -1;
740  }
741  }
742  } else {
743  // move the following entries one idx down
744  if (idx < self->msgRequestInfos.lastRequestIdx) {
745  self->msgRequestInfos.requestTypes[idx] = self->msgRequestInfos.requestTypes[idx + 1];
746  self->msgRequestInfos.requestHandles[idx] = self->msgRequestInfos.requestHandles[idx + 1];
747  if (idx + 1 == self->msgRequestInfos.currRequestIdx) {
748  self->msgRequestInfos.currRequestIdx--;
749  }
750  }
751  }
752  idx++;
753  }
754  if (self->msgRequestInfos.lastRequestIdx >= 0) {
755  self->msgRequestInfos.lastRequestIdx--;
756  }
757  if (self->msgRequestInfos.currRequestIdx < 0) {
758  self->msgRequestInfos.currRequestIdx++;
759  // start handling the request
760  self->startNextRequest(self);
761  } else {
762  // nothing to do the current request is different from the deleted one
763  // so just let the current one continue
764  }
765  return COMP_DISP_ERR_OK;
766 }
