120 static uint8_t nextFittingEntry(compMsgDispatcher_t *self, uint8_t u8CmdKey, uint16_t u16CmdKey) {
121  msgParts_t *received;
122  msgHeaderInfos_t *hdrInfos;
123 // dataView_t *dataView;
124  headerPart_t *hdr;
125  int hdrIdx;
126  int found;
127 
128  received = &self->compMsgData->received;
129  hdrInfos = &self->msgHeaderInfos;
130  hdrIdx = hdrInfos->currPartIdx;
131  hdr = &hdrInfos->headerParts[hdrIdx];
132 //ets_printf("§HEAD:!to:0x%04x!from:0x%04x!totalLgth:0x%04x!seqIdx!%d!§", received->toPart, received->fromPart, received->totalLgth, hdrInfos->seqIdx);
133  // and now search in the headers to find the appropriate message
134  hdrInfos->seqIdx = hdrInfos->seqIdxAfterHeader;
135  found = 0;
136  while (hdrIdx < hdrInfos->numHeaderParts) {
137  hdr = &hdrInfos->headerParts[hdrIdx];
138 //ets_printf("hdrIdx: %d to: %d %d\n", hdrIdx, hdr->hdrToPart, received->toPart);
139  if (hdr->hdrToPart == received->toPart) {
140 //ets_printf("to: %d from: %d %d\n", hdr->hdrToPart, hdr->hdrFromPart, received->fromPart);
141  if (hdr->hdrFromPart == received->fromPart) {
142 //ets_printf("lgth: hdr: %d received: %d\n", hdr->hdrTotalLgth, received->totalLgth);
143  if ((hdr->hdrTotalLgth == received->totalLgth) || (hdr->hdrTotalLgth == 0)) {
144 //ets_printf("§cmdKey: 0x%04x 0x%04x§", u16CmdKey, received->u16CmdKey);
145  if (u16CmdKey != 0) {
146  if (u16CmdKey == received->u16CmdKey) {
147  found = 1;
148  break;
149  }
150  } else {
151  found = 1;
152  break;
153  }
154  }
155  }
156  }
157  hdrIdx++;
158  }
159  if (!found) {
160 ets_printf("§nextFitting HANDLE_NOT_FOUND§");
161  return COMP_DISP_ERR_HANDLE_NOT_FOUND;
162  }
163  hdrInfos->currPartIdx = hdrIdx;
164 //ets_printf("§encryption: %c handleType: %c§", hdr->hdrEncryption, hdr->hdrHandleType);
165  received->encryption = hdr->hdrEncryption;
166 //ets_printf("§nextFitting!found!%d!hdrIdx!%d!hdr->cmdKey:0x%04x!received->cmdKey: 0x%04x§", found, hdrIdx, hdr->hdrU16CmdKey, received->u16CmdKey);
167  return COMP_DISP_ERR_OK;

172 static uint8_t getHeaderIndexFromHeaderFields(compMsgDispatcher_t *self, msgHeaderInfos_t *hdrInfos) {
173  int result;
174  dataView_t *dataView;
175  headerPart_t *hdr;
176  int hdrIdx;
177  int found;
178  uint8_t myHeaderLgth;
179  uint16_t seqVal;
180  msgParts_t *received;
181  uint8_t *cp;
182  uint8_t lgth;
183 
184  received = &self->compMsgData->received;
185 //ets_printf("§getHeaderIndexFromHeaderFields newCompMsgDataView§");
186  received->compMsgDataView = newCompMsgDataView(received->buf, received->lgth);
187  checkAllocOK(received->compMsgDataView);
188  dataView = received->compMsgDataView->dataView;
189 //ets_printf("§reclgth: %d§", received->lgth);
190  received->fieldOffset = 0;
191  myHeaderLgth = 0;
192  hdrInfos->seqIdx = 0;
193  seqVal = hdrInfos->headerSequence[hdrInfos->seqIdx];
194  while (seqVal != 0) {
195  switch(seqVal) {
196  case COMP_DISP_U16_DST:
197  result = dataView->getUint16(dataView, received->fieldOffset, &received->toPart);
198  checkErrOK(result);
199 //ets_printf("§to: 0x%04x§", received->toPart);
200  received->fieldOffset += sizeof(uint16_t);
201  break;
202  case COMP_DISP_U16_SRC:
203  result = dataView->getUint16(dataView, received->fieldOffset, &received->fromPart);
204  checkErrOK(result);
205 //ets_printf("§from: 0x%04x§", received->fromPart);
206  received->fieldOffset += sizeof(uint16_t);
207  break;
208  case COMP_DISP_U16_TOTAL_LGTH:
209  result = dataView->getUint16(dataView, received->fieldOffset, &received->totalLgth);
210  checkErrOK(result);
211 //ets_printf("§total: 0x%04x§", received->totalLgth);
212  received->fieldOffset += sizeof(uint16_t);
213  break;
214  case COMP_DISP_U16_SRC_ID:
215  result = dataView->getUint16(dataView, received->fieldOffset, &received->srcId);
216  checkErrOK(result);
217 //ets_printf("§srcId: 0x%04x§", received->srcId);
218  received->fieldOffset += sizeof(uint16_t);
219  break;
220  case COMP_DISP_U8_VECTOR_GUID:
221  cp = received->GUID;
222  result = dataView->getUint8Vector(dataView, received->fieldOffset, &cp, sizeof(received->GUID));
223  checkErrOK(result);
224 //ets_printf("§GUID: %s§", received->GUID);
225  received->fieldOffset += sizeof(received->GUID);
226  break;
227  case COMP_DISP_U8_VECTOR_HDR_FILLER:
228  cp = received->hdrFiller;
229  lgth = hdrInfos->headerLgth - received->fieldOffset;
230  result = dataView->getUint8Vector(dataView, received->fieldOffset, &cp, lgth);
231  checkErrOK(result);
232  received->fieldOffset += lgth;
233  break;
234  }
235  hdrInfos->seqIdx++;
236  seqVal = hdrInfos->headerSequence[hdrInfos->seqIdx];
237  }
238 //ets_printf("§seqIdx: %d§", hdrInfos->seqIdx);
239  hdrInfos->seqIdxAfterHeader = hdrInfos->seqIdx;
240  hdrInfos->currPartIdx = 0;
241  result = nextFittingEntry(self, 0, 0);
242 //ets_printf("§getHeaderIndexFromHeaderFields!result!%d!currPartIdx!%d!§", result, hdrInfos->currPartIdx);
243  if (received->compMsgDataView != NULL) {
244 //ets_printf("§os getHeaderIndexFromHeaderFields: free dataView: %p compMsgDataView: %p§", received->compMsgDataView->dataView, received->compMsgDataView);
245  os_free(received->compMsgDataView->dataView);
246  os_free(received->compMsgDataView);
247  received->compMsgDataView = NULL;
248  }
249  return result;
250 }

271 static uint8_t handleReceivedMsg(compMsgDispatcher_t *self, msgHeaderInfos_t *hdrInfos) {
272  int result;
273  headerPart_t *hdr;
274  int hdrIdx;
275  uint8_t answerType;
276  compMsgField_t fieldInfo;
277  uint16_t u16;
278  uint8_t u8;
279  uint8_t startOffset;
280  uint16_t sequenceEntry;
281  uint8_t *handle;
282  msgParts_t *received;
283 
284 //ets_printf("§handleReceivedMsg§");
285  received = &self->compMsgData->received;
286 //ets_printf("§handleReceivedMsg newCompMsgDataView§");
287  received->compMsgDataView = newCompMsgDataView(received->buf, received->totalLgth);
288  checkAllocOK(received->compMsgDataView);
289  hdrIdx = hdrInfos->currPartIdx;
290  hdr = &hdrInfos->headerParts[hdrIdx];
291 
292  // set received->lgth to end of the header
293  received->lgth = hdrInfos->headerLgth;
294  // we loop over the fieldSequence entries and handle them as needed
295  // attention not all entries of the message are handled here, only some special entries!
296 
297 //self->compMsgMsgDesc->dumpHeaderPart(self, hdr);
298  while (hdr->fieldSequence[hdrInfos->seqIdx] != 0) {
299  sequenceEntry = hdr->fieldSequence[hdrInfos->seqIdx];
300 //ets_printf("§sequenceEntry: 0x%04x seqIdx:%d§", sequenceEntry, hdrInfos->seqIdx);
301  received->fieldOffset = hdrInfos->headerLgth;
302  switch (sequenceEntry) {
303  case COMP_DISP_U16_CMD_KEY:
304  result = received->compMsgDataView->dataView->getUint16(received->compMsgDataView->dataView, received->fieldOffset, &u16);
305  received->u16CmdKey = u16;
306  received->fieldOffset += 2;
307  received->partsFlags |= COMP_DISP_U16_CMD_KEY;
308 //ets_printf("§1 u16CmdKey: 0x%04x!hdr: 0x%04x§", received->u16CmdKey, hdr->hdrU16CmdKey);
309  while (received->u16CmdKey != hdr->hdrU16CmdKey) {
310  hdrInfos->currPartIdx++;
311  result = self->nextFittingEntry(self, 0, received->u16CmdKey);
312  checkErrOK(result);
313  hdr = &hdrInfos->headerParts[hdrInfos->currPartIdx];
314 //ets_printf("§2 u16CmdKey: 0x%04x!hdr: 0x%04x§", received->u16CmdKey, hdr->hdrU16CmdKey);
315  }
316 //ets_printf("§found: currPartIdx: %d§", hdrInfos->currPartIdx);
317  break;
318  case COMP_DISP_U0_CMD_LGTH:
319  received->fieldOffset += 2;
320 //ets_printf("§u0CmdLgth!0!§");
321  break;
322  case COMP_DISP_U16_CMD_LGTH:
323  result = received->compMsgDataView->dataView->getUint16(received->compMsgDataView->dataView, received->fieldOffset, &u16);
324  received->u16CmdLgth = u16;
325  received->fieldOffset += 2;
326 //ets_printf("§u16CmdLgth!0x%04x!§", received->u16CmdLgth);
327  break;
328  case COMP_DISP_U0_CRC:
329 //ets_printf("§u0Crc!0!§");
330  result = COMP_MSG_ERR_OK;
331  break;
332  case COMP_DISP_U8_CRC:
333  fieldInfo.fieldLgth = 1;
334  fieldInfo.fieldOffset = received->totalLgth - 1;
335  startOffset = hdrInfos->headerLgth;
336  result = received->compMsgDataView->getCrc(received->compMsgDataView, &fieldInfo, startOffset, fieldInfo.fieldOffset);
337 //ets_printf("§u8Crc!res!%d!§", result);
338  break;
339  case COMP_DISP_U16_CRC:
340  fieldInfo.fieldLgth = 2;
341  fieldInfo.fieldOffset = received->totalLgth - 2;
342  startOffset = hdrInfos->headerLgth;
343  result = received->compMsgDataView->getCrc(received->compMsgDataView, &fieldInfo, startOffset, fieldInfo.fieldOffset);
344 //ets_printf("§u16Crc!res!%d!§", result);
345  break;
346  case COMP_DISP_U0_TOTAL_CRC:
347 //ets_printf("§u0TotalCrc!0!§");
348  result = COMP_MSG_ERR_OK;
349  break;
350  case COMP_DISP_U8_TOTAL_CRC:
351 //ets_printf("§u8TotalCrc!0!§");
352  fieldInfo.fieldLgth = 1;
353  fieldInfo.fieldOffset = received->totalLgth - 1;
354  startOffset = hdrInfos->headerLgth;
355  result = received->compMsgDataView->getTotalCrc(received->compMsgDataView, &fieldInfo);
356 //ets_printf("§u8totalCrc!res!%d!§", result);
357  break;
358  case COMP_DISP_U16_TOTAL_CRC:
359  fieldInfo.fieldLgth = 2;
360  fieldInfo.fieldOffset = received->totalLgth - 2;
361  startOffset = hdrInfos->headerLgth;
362  result = received->compMsgDataView->getTotalCrc(received->compMsgDataView, &fieldInfo);
363 //ets_printf("§u16TotalCrc!res!%d!§", result);
364  break;
365  }
366  checkErrOK(result);
367  hdrInfos->seqIdx++;
368  }
369  // free all space of received message
370  self->compMsgData->deleteMsg(self);
371 //ets_printf("§received msg deleted§");
372 //ets_printf("§call prepareAnswerMsg§");
373  result = prepareAnswerMsg(self, hdrInfos, &handle);
374  checkErrOK(result);
375  result = -self->resetMsgInfo(self, received);
376  checkErrOK(result);
377 //ets_printf("§handleReceivedMsg done§");
378  return COMP_MSG_ERR_OK;
379 }

403 //ets_printf("§handleReceivedPart: %c 0x%02x§", buffer[0], buffer[0]);
404 if (buffer == NULL) {
405 //ets_printf("§++++handleReceivedPart: buffer == NULL lgth: %d§", lgth);

407 //ets_printf("§++++handleReceivedPart: 0x%02x§", buffer[0]);

440 
441  // decrypt encrypted message part (after header)
442 cryptKey = "a1b2c3d4e5f6g7h8";

448  checkErrOK(result);

450  }
451  result = self->handleReceivedMsg(self, hdrInfos);
452 //ets_printf("§handleReceivedMsg end buffer idx: %d result: %d§", idx, result);
453  return result;
454  }
455  idx++;
456  }
457  return COMP_DISP_ERR_OK;
458 }
