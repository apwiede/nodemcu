121 static uint8_t nextFittingEntry(compMsgDispatcher_t *self, uint8_t u8CmdKey, uint16_t u16CmdKey) {
122  msgParts_t *received;
123  msgHeaderInfos_t *hdrInfos;
124  headerPart_t *hdr;
125  int hdrIdx;
126  int found;
127 
128  received = &self->compMsgData->received;
129  hdrInfos = &self->msgHeaderInfos;
130  hdrIdx = hdrInfos->currPartIdx;
131  hdr = &hdrInfos->headerParts[hdrIdx];
132 //ets_printf("§HEAD:!to:0x%04x!from:0x%04x!totalLgth:0x%04x!seqIdx!%d!§\n", received->toPart, received->fromPart, received->totalLgth, hdrInfos->seqIdx);
133  // and now search in the headers to find the appropriate message
134  hdrInfos->seqIdx = hdrInfos->seqIdxAfterHeader;
135  found = 0;
136  while (hdrIdx < hdrInfos->numHeaderParts) {
137  hdr = &hdrInfos->headerParts[hdrIdx];
138 //ets_printf("hdrIdx: %d to: %d %d\n", hdrIdx, hdr->hdrToPart, received->toPart);
139  if (hdr->hdrToPart == received->toPart) {
140 //ets_printf("to: %d from: %d %d\n", hdr->hdrToPart, hdr->hdrFromPart, received->fromPart);
141  if (hdr->hdrFromPart == received->fromPart) {
142 //ets_printf("lgth: hdr: %d received: %d\n", hdr->hdrTotalLgth, received->totalLgth);
143  if ((hdr->hdrTotalLgth == received->totalLgth) || (hdr->hdrTotalLgth == 0)) {
144 //ets_printf("§cmdKey: 0x%04x 0x%04x§", u16CmdKey, received->u16CmdKey);
145  if (u16CmdKey != 0) {
146  if (u16CmdKey == received->u16CmdKey) {
147  found = 1;
148  break;
149  }
150  } else {
151  found = 1;
152  break;
153  }
154  }
155  }
156  }
157  hdrIdx++;
158  }
159  if (!found) {
160 ets_printf("§nextFitting HEADER_NOT_FOUND§\n");
161  return COMP_DISP_ERR_HEADER_NOT_FOUND;
162  }
163  hdrInfos->currPartIdx = hdrIdx;
164 //ets_printf("§encryption: %c handleType: %c§", hdr->hdrEncryption, hdr->hdrHandleType);
165  received->encryption = hdr->hdrEncryption;
166 //ets_printf("§nextFitting!found!%d!hdrIdx!%d!hdr->cmdKey:0x%04x!received->cmdKey: 0x%04x§", found, hdrIdx, hdr->hdrU16CmdKey, received->u16CmdKey);
167  return COMP_DISP_ERR_OK;
168 }

172 static uint8_t getHeaderIndexFromHeaderFields(compMsgDispatcher_t *self, msgHeaderInfos_t *hdrInfos) {
173  int result;
174  dataView_t *dataView;
175  headerPart_t *hdr;
176  int hdrIdx;
177  int found;
178  uint8_t myHeaderLgth;
179  uint16_t seqVal;
180  msgParts_t *received;
181  uint8_t *cp;
182  uint8_t lgth;
183 
184  received = &self->compMsgData->received;
185 //ets_printf("§getHeaderIndexFromHeaderFields newCompMsgDataView§");
186  received->compMsgDataView = newCompMsgDataView(received->buf, received->lgth);
187  checkAllocOK(received->compMsgDataView);
188  dataView = received->compMsgDataView->dataView;
189 //ets_printf("§reclgth: %d§", received->lgth);
190  received->fieldOffset = 0;
191  myHeaderLgth = 0;
192  hdrInfos->seqIdx = 0;
193  seqVal = hdrInfos->headerSequence[hdrInfos->seqIdx];
194  while (seqVal != 0) {
195  switch(seqVal) {
196  case COMP_DISP_U16_DST:
197  result = dataView->getUint16(dataView, received->fieldOffset, &received->toPart);
198  checkErrOK(result);
199 //ets_printf("§to: 0x%04x§\n", received->toPart);
200  received->fieldOffset += sizeof(uint16_t);
201  break;
202  case COMP_DISP_U16_SRC:
203  result = dataView->getUint16(dataView, received->fieldOffset, &received->fromPart);
204  checkErrOK(result);
205 //ets_printf("§from: 0x%04x§\n", received->fromPart);
206  received->fieldOffset += sizeof(uint16_t);
207  break;
208  case COMP_DISP_U16_TOTAL_LGTH:
209  result = dataView->getUint16(dataView, received->fieldOffset, &received->totalLgth);
210  checkErrOK(result);
211 //ets_printf("§total: 0x%04x§\n", received->totalLgth);
212  received->fieldOffset += sizeof(uint16_t);
213  break;
214  case COMP_DISP_U16_SRC_ID:
215  result = dataView->getUint16(dataView, received->fieldOffset, &received->srcId);
216  checkErrOK(result);
217 //ets_printf("§srcId: 0x%04x§", received->srcId);
218  received->fieldOffset += sizeof(uint16_t);
219  break;
220  case COMP_DISP_U8_VECTOR_GUID:
221  cp = received->GUID;
222  result = dataView->getUint8Vector(dataView, received->fieldOffset, &cp, sizeof(received->GUID));
223  checkErrOK(result);
224 //ets_printf("§GUID: %s§", received->GUID);
225  received->fieldOffset += sizeof(received->GUID);
226  break;
227  case COMP_DISP_U8_VECTOR_HDR_FILLER:
228  cp = received->hdrFiller;
229  lgth = hdrInfos->headerLgth - received->fieldOffset;
230  result = dataView->getUint8Vector(dataView, received->fieldOffset, &cp, lgth);
231  checkErrOK(result);
232  received->fieldOffset += lgth;
233  break;
234  }
235  hdrInfos->seqIdx++;
236  seqVal = hdrInfos->headerSequence[hdrInfos->seqIdx];
237  }
238 //ets_printf("§seqIdx: %d§", hdrInfos->seqIdx);
239  hdrInfos->seqIdxAfterHeader = hdrInfos->seqIdx;
240  hdrInfos->currPartIdx = 0;
241  result = nextFittingEntry(self, 0, 0);
242 //ets_printf("§getHeaderIndexFromHeaderFields!result!%d!currPartIdx!%d!§", result, hdrInfos->currPartIdx);
243  if (received->compMsgDataView != NULL) {
244 //ets_printf("§os getHeaderIndexFromHeaderFields: free dataView: %p compMsgDataView: %p§", received->compMsgDataView->dataView, received->compMsgDataView);
245  os_free(received->compMsgDataView->dataView);
246  os_free(received->compMsgDataView);
247  received->compMsgDataView = NULL;
248  }
249  return result;
250 }

279 static uint8_t handleReceivedHeader(compMsgDispatcher_t *self) {
280  int result;
281  headerPart_t *hdr;
282  int hdrIdx;
283  msgParts_t *received;
284  msgHeaderInfos_t *hdrInfos;
285  compMsgField_t fieldInfo;
286  uint16_t u16;
287  uint8_t u8;
288  uint16_t sequenceEntry;
289  bool u8TotalCrc;
290  bool u16TotalCrc;
291  size_t startOffset;
292  size_t idx;
293 
294 //ets_printf("§handleReceivedHeader§\n");
295  hdrInfos = &self->msgHeaderInfos;
296  received = &self->compMsgData->received;
297  u8TotalCrc = false;
298  u16TotalCrc = false;
299 //ets_printf("§handleReceivedHeader newCompMsgDataView§");
300  received->compMsgDataView = newCompMsgDataView(received->buf, received->totalLgth);
301  checkAllocOK(received->compMsgDataView);
302 //ets_printf("§");
303 //received->compMsgDataView->dataView->dumpBinary(received->buf+30, 10, "Received->buf");
304 //ets_printf("§");
305  hdrIdx = hdrInfos->currPartIdx;
306 //ets_printf("§handleReceivedHeader: currPartIdx: %d totalLgth: %d§\n", hdrInfos->currPartIdx, received->totalLgth);
307  hdr = &hdrInfos->headerParts[hdrIdx];
308 
309  // set received->lgth to end of the header
310  received->lgth = hdrInfos->headerLgth;
311  // we loop over the fieldSequence entries and handle them as needed
312  // attention not all entries of the message are handled here, only some special entries!
313 
314 //self->compMsgMsgDesc->dumpHeaderPart(self, hdr);
315  // check if we have a U8_TOTAL_CRC or a U16_TOTAL_CRC or no TOTAL_CRC
316  idx = 0;
317  while (hdr->fieldSequence[idx] != 0) {
318  if (hdr->fieldSequence[idx] == COMP_DISP_U8_TOTAL_CRC) {
319  u8TotalCrc = true;
320  }
321  if (hdr->fieldSequence[idx] == COMP_DISP_U16_TOTAL_CRC) {
322  u16TotalCrc = true;
323  }
324  idx++;
325  }
326  while (hdr->fieldSequence[hdrInfos->seqIdx] != 0) {
327  sequenceEntry = hdr->fieldSequence[hdrInfos->seqIdx];
328 //ets_printf("§sequenceEntry: 0x%04x seqIdx:%d§", sequenceEntry, hdrInfos->seqIdx);
329  received->fieldOffset = hdrInfos->headerLgth;
330  switch (sequenceEntry) {
331  case COMP_DISP_U16_CMD_KEY:
332  result = received->compMsgDataView->dataView->getUint16(received->compMsgDataView->dataView, received->fieldOffset, &u16);
333  received->u16CmdKey = u16;
334 //ets_printf("§1 u16CmdKey: 0x%04x!hdr: 0x%04x!offset: %d§", received->u16CmdKey, hdr->hdrU16CmdKey, received->fieldOffset);
335  received->fieldOffset += 2;
336  received->partsFlags |= COMP_DISP_U16_CMD_KEY;
337  while (received->u16CmdKey != hdr->hdrU16CmdKey) {
338  hdrInfos->currPartIdx++;
339  result = self->nextFittingEntry(self, 0, received->u16CmdKey);
340  checkErrOK(result);
341  hdr = &hdrInfos->headerParts[hdrInfos->currPartIdx];
342 //ets_printf("§2 u16CmdKey: 0x%04x!hdr: 0x%04x§", received->u16CmdKey, hdr->hdrU16CmdKey);
343  }
344 //ets_printf("§found: currPartIdx: %d§", hdrInfos->currPartIdx);
345  break;
346  case COMP_DISP_U0_CMD_LGTH:
347  received->fieldOffset += 2;
348 //ets_printf("§u0CmdLgth!0!§");
349  break;
350  case COMP_DISP_U16_CMD_LGTH:
351  result = received->compMsgDataView->dataView->getUint16(received->compMsgDataView->dataView, received->fieldOffset, &u16);
352  received->u16CmdLgth = u16;
353  received->fieldOffset += 2;
354 //ets_printf("§u16CmdLgth!0x%04x!§", received->u16CmdLgth);
355  break;
356  case COMP_DISP_U0_CRC:
357 //ets_printf("§u0Crc!0!§");
358  result = COMP_MSG_ERR_OK;
359  break;
360  case COMP_DISP_U8_CRC:
361  fieldInfo.fieldLgth = 1;
362  fieldInfo.fieldOffset = received->totalLgth - 1;
363  if (hdr->hdrFlags & COMP_DISP_TOTAL_CRC) {
364  if (u8TotalCrc) {
365  fieldInfo.fieldOffset -= 1;
366  } else {
367  fieldInfo.fieldOffset -= 2;
368  }
369  }
370  startOffset = hdrInfos->headerLgth;
371  result = received->compMsgDataView->getCrc(received->compMsgDataView, &fieldInfo, startOffset, fieldInfo.fieldOffset);
372 //ets_printf("§u8Crc!res!%d!§", result);
373  break;
374  case COMP_DISP_U16_CRC:
375  fieldInfo.fieldLgth = 2;
376  fieldInfo.fieldOffset = received->totalLgth - 2;
377  if (hdr->hdrFlags & COMP_DISP_TOTAL_CRC) {
378  if (u8TotalCrc) {
379  fieldInfo.fieldOffset -= 1;
380  } else {
381  fieldInfo.fieldOffset -= 2;
382  }
383  }
384  startOffset = hdrInfos->headerLgth;
385  result = received->compMsgDataView->getCrc(received->compMsgDataView, &fieldInfo, startOffset, fieldInfo.fieldOffset);
386 //ets_printf("§u16Crc!res!%d!§", result);
387  break;
388  case COMP_DISP_U0_TOTAL_CRC:
389 //ets_printf("§u0TotalCrc!0!§");
390  result = COMP_MSG_ERR_OK;
391  break;
392  case COMP_DISP_U8_TOTAL_CRC:
393  fieldInfo.fieldLgth = 1;
394  fieldInfo.fieldOffset = received->totalLgth - 1;
395 //ets_printf("§u8TotalCrc!fieldOffset: %d§", fieldInfo.fieldOffset);
396  result = received->compMsgDataView->getTotalCrc(received->compMsgDataView, &fieldInfo);
397 //ets_printf("§u8totalCrc!res!%d!§", result);
398  break;
399  case COMP_DISP_U16_TOTAL_CRC:
400  fieldInfo.fieldLgth = 2;
401  fieldInfo.fieldOffset = received->totalLgth - 2;
402  result = received->compMsgDataView->getTotalCrc(received->compMsgDataView, &fieldInfo);
403 //ets_printf("§u16TotalCrc!res!%d!§", result);
404  break;
405  }
406  checkErrOK(result);
407  hdrInfos->seqIdx++;
408  }
409  // free all space of received message
410 //ets_printf("call deleteMsg\n");
411  self->compMsgData->deleteMsg(self);
412 //ets_printf("§received msg deleted§");
413  return COMP_MSG_ERR_OK;
414 }

489  fieldsToSave->fieldValueStr = stringValue;
490  fieldsToSave->fieldValue = numericValue;
491  }
492  idx++;
493  }
494  fieldToSaveIdx++;
495  }
496  idx = 0;
497  hadActionCb = false;
498  while (idx < self->compMsgData->numMsgValParts) {
499  msgValPart = &self->compMsgData->msgValParts[idx];
500  if (msgValPart->fieldValueActionCb != NULL) {
501 //ets_printf("§have actionCb: %s\n", msgValPart->fieldValueActionCb);
502  hadActionCb = true;
503  result = self->getActionCallback(self, msgValPart->fieldValueActionCb+1, &actionCallback);
504  checkErrOK(result);
505  result = actionCallback(self);
506  checkErrOK(result);
507  }
508  idx++;
509  }
510  if (!hadActionCb) {
511  result = self->prepareAnswerMsg(self, COMP_MSG_ACK_MSG, &handle);
512  checkErrOK(result);
513  result = self->resetMsgInfo(self, received);
514  checkErrOK(result);
515  }
516 //ets_printf("§storeReceivedMsg done§\n");
517  return COMP_MSG_ERR_OK;
518 }

562 static uint8_t handleReceivedPart(compMsgDispatcher_t *self, const uint8_t * buffer, uint8_t lgth) {
563  int idx;
564  msgHeaderInfos_t *hdrInfos;
565  headerPart_t *hdr;

574  bool u16TotalCrc;

576 
577 //ets_printf("§handleReceivedPart: %c 0x%02x§", buffer[0], buffer[0]);
578 if (buffer == NULL) {
579 //ets_printf("§++++handleReceivedPart: buffer == NULL lgth: %d§", lgth);
580 } else {
581 //ets_printf("§++++handleReceivedPart: 0x%02x lgth: %d§", buffer[0], lgth);
582 }
583  hdrInfos = &self->msgHeaderInfos;
584  compMsgData = self->compMsgData;
585  received = &compMsgData->received;
586 //FIXME need to free at end of message handling !!!
587 //ets_printf("§handleReceivedPart: !received->buf: %p!§", received->buf);
588 //ets_printf("§receivedLgth: %d lgth: %d fieldOffset: %d headerLgth: %d!§", received->lgth, lgth, received->fieldOffset, hdrInfos->headerLgth);
589  idx = 0;
590  while (idx < lgth) {
591  received->buf[received->lgth++] = buffer[idx];
592  received->realLgth++;
593  if (received->lgth == hdrInfos->headerLgth) {
594 //ets_printf("§received lgth: %d lgth: %d idx: %d§", received->lgth, lgth, idx);
595 //ets_printf("§receveived->lgth: %d§", received->lgth);

597 //ets_printf("§getHeaderIndexFromHeaderFields result: %d currPartIdx: %d§\n", result, hdrInfos->currPartIdx);
598  }
599  // loop until we have full message then decrypt if necessary and then handle the message
600 //ets_printf("§totalLgth: %d§", received->totalLgth);
601  if (received->lgth == received->totalLgth) {
602  hdrIdx = hdrInfos->currPartIdx;
603  hdr = &hdrInfos->headerParts[hdrIdx];
604 //ets_printf("§hdrIdx: %d§", hdrIdx);
605 //ets_printf("§receveived->totalLgth: %d§", received->totalLgth);
606  // check if we have a U8_TOTAL_CRC or a U16_TOTAL_CRC or no TOTAL_CRC
607  seqIdx = 0;
608  u8TotalCrc = false;
609  u16TotalCrc = false;
610  while (hdr->fieldSequence[seqIdx] != 0) {
611  if (hdr->fieldSequence[seqIdx] == COMP_DISP_U8_TOTAL_CRC) {
612  u8TotalCrc = true;
613  }
614  if (hdr->fieldSequence[seqIdx] == COMP_DISP_U16_TOTAL_CRC) {
615  u16TotalCrc = true;
616  }
617  seqIdx++;
618  }
619 //ets_printf("§hdr->hdrHandleType: %c§", hdr->hdrHandleType);
620  switch (hdr->hdrHandleType) {
621  case 'G':
622  case 'R':
623  if (hdr->hdrEncryption == 'E') {
624  uint8_t *cryptedPtr;
625  uint8_t *cryptKey;
626  uint8_t *decrypted;;
627  uint8_t mlen;
628  uint8_t klen;
629  uint8_t ivlen;
630  int decryptedLgth;
631 
632  // decrypt encrypted message part (after header)
633 cryptKey = "a1b2c3d4e5f6g7h8";
634  mlen = received->totalLgth - hdrInfos->headerLgth;
635  if (hdr->hdrFlags & COMP_DISP_TOTAL_CRC) {
636  if (u8TotalCrc) {
637  mlen -= 1;
638  } else {
639  mlen -= 2;
640  }
641  }
642  ivlen = 16;
643  klen = 16;
644  cryptedPtr = received->buf + hdrInfos->headerLgth;
645  result = self->decryptMsg(cryptedPtr, mlen, cryptKey, klen, cryptKey, ivlen, &decrypted, &decryptedLgth);
646  checkErrOK(result);
647 //ets_printf("mlen: %d decryptedLgth: %d\n", mlen, decryptedLgth);
648  c_memcpy(cryptedPtr, decrypted, decryptedLgth);
649  }
650  result = self->storeReceivedMsg(self);
651 //ets_printf("§handleReceivedMsg end buffer idx: %d result: %d§", idx, result);
652  return result;
653  case 'U':
654  case 'W':
655  self->compMsgData->currHdr = hdr;
656  result = self->forwardMsg(self);
657 ets_printf("§forwardMsg result: %d§", result);
658  return result;
659  default:
660 ets_printf("handleReceivedPart: funny handleType: %c 0x%02x\n", hdr->hdrHandleType, hdr->hdrHandleType);
661  return COMP_DISP_ERR_FUNNY_HANDLE_TYPE;
662  }
663  }
664  idx++;
665  }
666  return COMP_DISP_ERR_OK;
667 }
