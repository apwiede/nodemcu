91 static uint8_t readLine(compMsgMsgDesc_t *self, uint8_t **buffer, uint8_t *lgth) {
92  size_t n = BUFSIZ;
93  char buf[BUFSIZ];
94  int i;
95  uint8_t *cp;
96  uint8_t end_char = '\n';
97 
98  if (fileFd == (FS_OPEN_OK - 1)) {
99  return COMP_MSG_ERR_FILE_NOT_OPENED;
100  }
101  n = fs_read(fileFd, buf, n);
102  cp = *buffer;
103  *lgth = 0;
104  for (i = 0; i < n; ++i) {
105  cp[i] = buf[i];
106  if (buf[i] == end_char) {
107  ++i;
108  break;
109  }
110  }
111  cp[i] = 0;
112  *lgth = i;
113  fs_seek (fileFd, -(n - i), SEEK_CUR);
114  return COMP_MSG_ERR_OK;
115 }
