155 static uint8_t *toBase64 ( const uint8_t *msg, size_t *len){
156  size_t i;
157  size_t n;
158  size_t lgth;
159  uint8_t * q;
160  uint8_t *out;
161 
162  n = *len;
163  if (!n) // handle empty string case
164  return NULL;
165 
166  lgth = (n + 2) / 3 * 4;
167  out = (uint8_t *)os_malloc(lgth + 1);
168  out[lgth] = '\0';
169  if (out == NULL) {
170  return NULL;
171  }
172  uint8 bytes64[sizeof(b64)];
173  c_memcpy(bytes64, b64, sizeof(b64)); //Avoid lots of flash unaligned fetches
174 
175  for (i = 0, q = out; i < n; i += 3) {
176  int a = msg[i];
177  int b = (i + 1 < n) ? msg[i + 1] : 0;
178  int c = (i + 2 < n) ? msg[i + 2] : 0;
179  *q++ = bytes64[a >> 2];
180  *q++ = bytes64[((a & 3) << 4) | (b >> 4)];
181  *q++ = (i + 1 < n) ? bytes64[((b & 15) << 2) | (c >> 6)] : BASE64_PADDING;
182  *q++ = (i + 2 < n) ? bytes64[(c & 63)] : BASE64_PADDING;
183  }
184  *q = '\0';
185  *len = q - out;
186  return out;
187 }

438 static void serverConnected(void *arg) {
439  struct espconn *pesp_conn;
440  netsocketUserData_t *nud;
441  int result;
442  int i;
443 
444 //ets_printf("§serverConnected: arg: %p§", arg);
445  pesp_conn = arg;
446  nud = (netsocketUserData_t *)pesp_conn->reverse;
447  if(nud == NULL) {
448  return;
449  }
450  for(i = 0; i < MAX_SOCKET; i++) {
451  if (socket[i] == NULL) { // found empty slot
452  break;
453  }
454  }
455  if(i>=MAX_SOCKET) {// can't create more socket
456 #ifdef CLIENT_SSL_ENABLE
457  if (nud->secure) {
458  if(pesp_conn->proto.tcp->remote_port || pesp_conn->proto.tcp->local_port) {
459  espconn_secure_disconnect(pesp_conn);
460  }
461  } else
462 #endif
463  {
464  if(pesp_conn->proto.tcp->remote_port || pesp_conn->proto.tcp->local_port) {
465  espconn_disconnect(pesp_conn);
466  }
467  }
468 // checkErrOK(gL, NETSOCKET_ERR_MAX_SOCKET_REACHED, "netsocket_server_connected");
469  pesp_conn->reverse = NULL; // not accept this conn
470  return;
471  }
472 //ets_printf("registstart\n");
473  result = espconn_regist_recvcb(pesp_conn, socketReceived);
474  if (result != COMP_DISP_ERR_OK) {
475 ets_printf("§regist socketReceived err: %d§", result);

479 ets_printf("§regist socketSent err: %d§", result);

488  }
