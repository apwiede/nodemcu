224 static uint8_t sendMsg(compMsgDispatcher_t *self, uint8_t *msgData, size_t msgLgth) {
225  uint8_t result;
226 
227 //ets_printf("§sendMsg: %c§", self->compMsgData->currHdr->hdrHandleType);
228  switch (self->compMsgData->currHdr->hdrHandleType) {
229  case 'A':
230 //ets_printf("wud: %p\n", self->compMsgData->wud);
231  if (self->compMsgData->wud == NULL) {
232  return COMP_DISP_ERR_NO_WEBSOCKET_OPENED;
233  }
234 ets_printf("remote_ip: %d %d %d %d port: %d\n", self->compMsgData->wud->remote_ip[0], self->compMsgData->wud->remote_ip[1], self->compMsgData->wud->remote_ip[2], self->compMsgData->wud->remote_ip[3], self->compMsgData->wud->remote_port);
235  result = self->websocketSendData(self->compMsgData->wud, msgData, msgLgth, OPCODE_BINARY);
236  checkErrOK(result);
237  break;
238  case 'G':
239  break;
240  case 'S':
241  result = typeRSendAnswer(self, msgData, msgLgth);
242  checkErrOK(result);
243  result = self->compMsgData->deleteMsg(self);
244  checkErrOK(result);
245  break;
246  case 'R':
247  break;
248  case 'U':
249 ets_printf("§type U msg send to cloud§");
250  self->cloudMsgData = msgData;
251  self->cloudMsgDataLgth = msgLgth;
252  self->sendCloudMsg = sendCloudMsg;
253  result = self->prepareCloudMsg(self);
254  checkErrOK(result);
255  result = self->netsocketStartCloudSocket(self);
256 //ets_printf("§netsocketStartCloudSocket: result: %d§", result);
257  checkErrOK(result);
258  break;
259  case 'W':
260  break;
261  case 'N':
262  // just ignore
263  return COMP_DISP_ERR_OK;
264  break;
265  default:
266  return COMP_DISP_ERR_BAD_HANDLE_TYPE;
267  }
268 //ets_printf("§sendMsg: done§");
269  return COMP_DISP_ERR_OK;
270 }
