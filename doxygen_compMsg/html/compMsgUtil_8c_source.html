61 /**
62  * \brief encode message with base64
63  * \param msg The message
64  * \param len The length of the message
65  * \param encoded The out param encoded message
66  * \return Error code or ErrorOK
67  *
68  */
69 static uint8_t toBase64(const uint8_t *msg, size_t *len, uint8_t **encoded) {
70  size_t i;
71  size_t n;
72  uint8_t *q;
73  uint8_t *out;
74  uint8_t bytes64[sizeof(b64)];
75 
76  n = *len;
77  if (!n) { // handle empty string case
78  return COMP_MSG_ERR_OUT_OF_MEMORY;
79  }
80  out = (uint8_t *)os_zalloc(((n + 2) / 3 * 4) + 1);
81  checkAllocOK(out);
82  c_memcpy(bytes64, b64, sizeof(b64)); //Avoid lots of flash unaligned fetches
83 
84  for (i = 0, q = out; i < n; i += 3) {
85  int a = msg[i];
86  int b = (i + 1 < n) ? msg[i + 1] : 0;
87  int c = (i + 2 < n) ? msg[i + 2] : 0;
88  *q++ = bytes64[a >> 2];
89  *q++ = bytes64[((a & 3) << 4) | (b >> 4)];
90  *q++ = (i + 1 < n) ? bytes64[((b & 15) << 2) | (c >> 6)] : BASE64_PADDING;
91  *q++ = (i + 2 < n) ? bytes64[(c & 63)] : BASE64_PADDING;
92  }
93  *q = '\0';
94  *len = q - out;
95 //ets_printf("§b64Len: %d *len: %d§", ((n + 2) / 3 * 4) + 1, *len);
96  // ATTENTION the caller has to free *encoded!!
97  *encoded = out;
98  return COMP_MSG_ERR_OK;
99 }

103 static uint8_t fromBase64(const uint8_t *encodedMsg, size_t *len, uint8_t **decodedMsg) {
104  int i;
105  int n;
106  int blocks;
107  int pad;
108  const uint8 *p;
109  uint8_t unbytes64[UCHAR_MAX+1];
110  uint8_t *msg;
111  uint8_t *q;
112 
113  n = *len;
114  blocks = (n>>2);
115  pad = 0;
116  if (!n) { // handle empty string case
117  return COMP_MSG_ERR_OUT_OF_MEMORY;
118  }
119  if (n & 3) {
120  return COMP_MSG_ERR_INVALID_BASE64_STRING;
121  }
122  c_memset(unbytes64, BASE64_INVALID, sizeof(unbytes64));
123  for (i = 0; i < sizeof(b64)-1; i++) {
124  unbytes64[b64[i]] = i; // sequential so no exceptions
125  }
126  if (encodedMsg[n-1] == BASE64_PADDING) {
127  pad = (encodedMsg[n-2] != BASE64_PADDING) ? 1 : 2;
128  blocks--; //exclude padding block
129  }
130 
131  for (i = 0; i < n - pad; i++) {
132  if (!ISBASE64(encodedMsg[i])) {
133  return COMP_MSG_ERR_INVALID_BASE64_STRING;
134  }
135  }
136  unbytes64[BASE64_PADDING] = 0;
137  q = (uint8_t *) os_zalloc(1+ (3 * n / 4));
138  checkAllocOK(q);
139  msg = q;
140  for (i = 0, p = encodedMsg; i<blocks; i++) {
141  uint8 a = unbytes64[*p++];
142  uint8 b = unbytes64[*p++];
143  uint8 c = unbytes64[*p++];
144  uint8 d = unbytes64[*p++];
145  *q++ = (a << 2) | (b >> 4);
146  *q++ = (b << 4) | (c >> 2);
147  *q++ = (c << 6) | d;
148  }
149  if (pad) { //now process padding block bytes
150  uint8 a = unbytes64[*p++];
151  uint8 b = unbytes64[*p++];
152  *q++ = (a << 2) | (b >> 4);
153  if (pad == 1) *q++ = (b << 4) | (unbytes64[*p] >> 2);
154  }
155  *len = q - msg;
156  *decodedMsg = msg;
157  return COMP_MSG_ERR_OK;;
158 }

162 static uint8_t encryptMsg(const uint8_t *msg, size_t mlen, const uint8_t *key, size_t klen, const uint8_t *iv, size_t ivlen, uint8_t **buf, int *lgth) {
163  const crypto_mech_t *mech;
164  size_t bs;
165  size_t clen;
166  uint8_t *crypted;
167 
168  *buf = NULL;
169  *lgth = 0;
170  mech = crypto_encryption_mech ("AES-CBC");
171  if (mech == NULL) {
172  return COMP_MSG_ERR_CRYPTO_BAD_MECHANISM;
173  }
174  bs = mech->block_size;
175  clen = ((mlen + bs - 1) / bs) * bs;
176  *lgth = clen;
177 //ets_printf("dlen: %d lgth: %d clen: %d data: %p\n", dlen, *lgth, clen, data);
178  crypted = (uint8_t *)os_zalloc (clen);
179  if (!crypted) {
180  return COMP_MSG_ERR_CRYPTO_INIT_FAILED;
181  }
182  *buf = crypted;
183  crypto_op_t op =
184  {
185  key, klen,
186  iv, ivlen,
187  msg, mlen,
188  crypted, clen,
189  OP_ENCRYPT
190  };
191  if (!mech->run (&op)) {
192  os_free (crypted);
193  return COMP_MSG_ERR_CRYPTO_INIT_FAILED;
194  }
195  return COMP_MSG_ERR_OK;
196 }
