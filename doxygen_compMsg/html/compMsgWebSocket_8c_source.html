140 /**
141  * \brief send message data to a web scoket
142  * \param wud The websocket user data
143  * \param payload The message data
144  * \param size The number of characters of the message
145  * \param opcode The type for sending the data (text/binary/...)
146  * \return Error code or ErrorOK
147  *
148  */
149 static uint8_t websocketSendData(websocketUserData_t *wud, const char *payload, int size, int opcode)
150 {
151  uint8_t hdrBytes[4]; // we have either 2 or 4 bytes depending on length of message
152  int hdrLgth;
153  int fsize;
154  uint8_t*buff;
155  uint8_t result;
156  int i;
157 
158 //ets_printf("websocketSendData: size: %d\n", size);
159  hdrLgth = 2;
160  hdrBytes[0] = 0x80; //set first bit
161  hdrBytes[0] |= opcode; //frame->opcode; //set op code
162  if (size < 126) {
163  hdrBytes[1] = size;
164  } else {
165  if (size < SSL_BUFFER_SIZE - sizeof(hdrBytes)) {
166  hdrLgth += 2;
167  hdrBytes[1] = 126;
168  hdrBytes[2] = (size >> 8) & 0xFF;
169  hdrBytes[3] = size & 0xFF;
170  } else {
171  return WEBSOCKET_ERR_TOO_MUCH_DATA;
172  }
173  }
174  fsize = size + hdrLgth;
175  buff = os_malloc(fsize);
176  if (buff == NULL) {
177  return WEBSOCKET_ERR_OUT_OF_MEMORY;
178  }
179  for (i = 0; i < hdrLgth; i++) {
180  buff[i] = hdrBytes[i];
181  }
182 
183  os_memcpy(&buff[hdrLgth], (uint8_t *)payload, size);
184  result = espconn_sent(wud->pesp_conn, (unsigned char *)buff, fsize);
185 //ets_printf("espconn_sent: result: %d\n", result);
186  os_free(buff);
187 //ets_printf("espconn_sent: done\n");
188  return WEBSOCKET_ERR_OK;
189 }

193 static int ICACHE_FLASH_ATTR websocket_parse(char * data, size_t dataLenb, char **resData, int *len, websocketUserData_t *wud) {
194  uint8_t byte = data[0];
195  uint8_t FIN = byte & 0x80;
196  uint8_t opcode = byte & 0x0F;
197 
198  if ((opcode > 0x03 && opcode < 0x08) || opcode > 0x0B) {
199  ets_sprintf(err_opcode, "%d", opcode);
200  checkErrOK(WEBSOCKET_ERR_INVALID_FRAME_OPCODE);
201  }
202 
203  // opcodes: {1 text 2 binary 8 close 9 ping 10 pong}
204  switch (opcode) {
205  case OPCODE_TEXT:
206 //ets_printf("parse text\n");
207  break;
208  case OPCODE_BINARY:
209 //ets_printf("parse binary\n");
210  break;
211  case OPCODE_CLOSE:
212  break;
213  case OPCODE_PING:
214  websocketSendData(wud, " ", 1, OPCODE_PONG);
215  return WEBSOCKET_ERR_OK;
216  break;
217  case OPCODE_PONG:
218  return WEBSOCKET_ERR_OK;
219  break;
220  }
221  byte = data[1];
222 
223  char * recv_data;
224  int masked = byte & 0x80;
225  int size = byte & 0x7F;
226  int offset = 2;
227 
228  if (size == 126) {
229  size = 0;
230  size = data[3]; //LSB
231  size |= (uint64_t) data[2] << 8; //MSB
232  offset = 4;
233  } else if (size == 127) {
234  size = 0;
235  size |= (uint64_t) data[2] << 56;
236  size |= (uint64_t) data[3] << 48;
237  size |= (uint64_t) data[4] << 40;
238  size |= (uint64_t) data[5] << 32;
239  size |= (uint64_t) data[6] << 24;
240  size |= (uint64_t) data[7] << 16;
241  size |= (uint64_t) data[8] << 8;
242  size |= (uint64_t) data[9];
243  offset = 10;
244  }
245 
246  if (masked) {
247  //read mask key
248  char mask[4];
249  uint64_t i;
250 
251  mask[0] = data[offset];
252  mask[1] = data[offset + 1];
253  mask[2] = data[offset + 2];
254  mask[3] = data[offset + 3];
255  offset += 4;
256  for (i = 0; i < size; i++) {
257  data[i + offset] ^= mask[i % 4];
258  }
259  }
260  recv_data = &data[offset];
261 for (int i = 0; i < size; i++) {
262 // ets_printf("i: %d 0x%02x\n", i, recv_data[i]&0xFF);
263 }
264  switch (opcode) {
265  case OPCODE_TEXT:
266 //ets_printf("cb text\n");
267  wud->websocketTextReceived(wud->compMsgDispatcher, wud, recv_data, size);
268  break;
269  case OPCODE_BINARY:
270 //ets_printf("cb binary\n");
271 //ets_printf("parse binary: size: %d\n", size);
272  wud->websocketBinaryReceived(wud->compMsgDispatcher, wud, recv_data, size);
273  break;
274  }
275  *resData = recv_data;
276  *len = size;
277  recv_data[size] = 0;
278 
279  if (size + offset < dataLenb) {
280  websocket_parse(&data[size + offset], dataLenb - (size + offset), resData, len, wud);
281  }
282  return WEBSOCKET_ERR_OK;
283 }

287 static uint8_t websocket_recv(char *string, websocketUserData_t *wud, char **data, int *lgth) {
288  char * key = NULL;
289  int idx;
290  int found;
291  uint8_t result;
292 
293  idx = 0;
294 //ets_printf("websocket_recv: %s!remote_port: %d\n", wud->curr_url, wud->remote_port);
295 //ets_printf("websocket_recv: %s!%s!remote_port: %d\n",string, wud->curr_url, wud->remote_port);
296  if ((wud->curr_url != NULL) && (strstr(string, wud->curr_url) != NULL)) {
297  if (strstr(string, header_key) != NULL) {
298  char *begin = strstr(string, header_key) + os_strlen(header_key);
299  char *end = strstr(begin, "\r");
300  key = os_malloc((end - begin) + 1);
301  checkAllocOK(key);
302  os_memcpy(key, begin, end - begin);
303  key[end - begin] = 0;
304  }
305 //ets_printf("websocket_recv2: key: %s\n", key);
306  const char *trailer;
307  trailer = "\r\n\r\n";
308  int trailerLen;
309  trailerLen = os_strlen(trailer);
310  size_t digestLen;
311  digestLen = 20; //sha1 is always 20 byte long
312  uint8_t digest[digestLen];
313  int payloadLen;
314  char *payload;
315  uint8_t *base64Digest;
316  SHA1_CTX ctx;
317  // Use the SHA* functions in the rom
318  SHA1Init(&ctx);
319  SHA1Update(&ctx, key, os_strlen(key));
320  SHA1Update(&ctx, ws_uuid, os_strlen(ws_uuid));
321  SHA1Final(digest, &ctx);
322 
323  result = wud->compMsgDispatcher->toBase64(digest, &digestLen, &base64Digest);
324  checkErrOK(result);
325  checkAllocOK(base64Digest);
326  payloadLen = os_strlen(HEADER_WEBSOCKET_START) + os_strlen(HEADER_WEBSOCKET_URL) +os_strlen(HEADER_WEBSOCKET_END) + digestLen + trailerLen;
327  payload = os_malloc(payloadLen + 1); // +1 for trailing '\0' character
328  checkAllocOK(payload);
329  os_sprintf(payload, "%s%s%s%s%s\0", HEADER_WEBSOCKET_START, HEADER_WEBSOCKET_URL, HEADER_WEBSOCKET_END, base64Digest, trailer);
330 //ets_printf("§payloadLen: %d payload: %d§\n", payloadLen, c_strlen(payload));
331  os_free(base64Digest);
332  struct espconn *pesp_conn = NULL;
333  pesp_conn = wud->pesp_conn;
334 
335  // FIXME!! reboot called if setting here for socket!!
336  if (wud->isWebsocket != 1) {
337  wud->isWebsocket = 1;
338  }
339 
340 //ets_printf("payload: %d!%s!\n", payloadLen, payload);
341  result = espconn_sent(wud->pesp_conn, (unsigned char *)payload, payloadLen);
342  os_free(key);
343 //ets_printf("espconn_sent: result: %d\n", result);
344  checkErrOK(result);
345  } else if (wud->isWebsocket == 1) {
346 //ets_printf("websocket_parse: %d!curr_url: %s!\n", os_strlen(string), wud->curr_url);
347  websocket_parse(string, os_strlen(string), data, lgth, wud);
348  }
349  return WEBSOCKET_ERR_OK;
