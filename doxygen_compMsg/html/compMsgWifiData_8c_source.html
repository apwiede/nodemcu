175 static void bssScanDoneCb(void *arg, STATUS status) {
176  struct bss_info *bss_link;
177  uint8_t numEntries;
178  bssScanInfo_t *scanInfo;
179 
180 //ets_printf("bssScanDoneCb bssScanRunning: arg: %p %d status: %d!\n", arg, bssScanRunning, status);
181  if (arg == NULL) {
182  return;
183  }
184  if (status != COMP_MSG_ERR_OK) {
185  return;
186  }
187  if (bssScanRunning == false) {
188  return;
189  }
190  if (bssScanRunning == true) {
191  bssScanRunning = false;
192  }
193 //ets_printf("bssScanDoneCb bssScanRunning2: %d status: %d!\n", bssScanRunning, status);
194  numEntries = 0;
195  bss_link = (struct bss_info *)arg;
196  while (bss_link != NULL) {
197  numEntries++;
198  bss_link = bss_link->next.stqe_next;
199  }
200  if (bssScanInfos.infos != NULL) {
201  os_free(bssScanInfos.infos);
202  bssScanInfos.infos = NULL;
203  bssScanInfos.numScanInfos = 0;
204  bssScanInfos.maxScanInfos = 0;
205  }
206  bssScanInfos.maxScanInfos = numEntries;
207  bssScanInfos.infos = os_zalloc(bssScanInfos.maxScanInfos * sizeof(bssScanInfo_t));
208  if (bssScanInfos.infos == NULL) {
209  return;
210  }
211  bss_link = (struct bss_info *)arg;
212  compMsgWifiData.bssScanSizes.bssidSize = 0;
213  compMsgWifiData.bssScanSizes.bssidStrSize = 0;
214  compMsgWifiData.bssScanSizes.ssidSize = 0;
215  compMsgWifiData.bssScanSizes.channelSize = 0;
216  compMsgWifiData.bssScanSizes.rssiSize = 0;
217  compMsgWifiData.bssScanSizes.authmodeSize = 0;
218  compMsgWifiData.bssScanSizes.is_hiddenSize = 0;
219  compMsgWifiData.bssScanSizes.freq_offsetSize = 0;
220  compMsgWifiData.bssScanSizes.freqcal_valSize = 0;
221  while (bss_link != NULL) {
222  scanInfo = &bssScanInfos.infos[bssScanInfos.numScanInfos];
223  c_memset(scanInfo->ssid, 0, sizeof(scanInfo->ssid));
224  if (bss_link->ssid_len <= sizeof(scanInfo->ssid)) {
225  c_memcpy(scanInfo->ssid, bss_link->ssid, bss_link->ssid_len);
226  compMsgWifiData.bssScanSizes.ssidSize += bss_link->ssid_len + 1;
227  } else {
228  c_memcpy(scanInfo->ssid, bss_link->ssid, sizeof(scanInfo->ssid));
229  compMsgWifiData.bssScanSizes.ssidSize += sizeof(scanInfo->ssid) + 1;
230  }
231  scanInfo->ssid_len = bss_link->ssid_len;
232  c_memset(scanInfo->bssidStr, 0, sizeof(scanInfo->bssidStr));
233  c_memcpy(scanInfo->bssid, bss_link->bssid, sizeof(scanInfo->bssid));
234  compMsgWifiData.bssScanSizes.bssidSize += sizeof(scanInfo->bssid) + 1;
235  c_sprintf(scanInfo->bssidStr,MACSTR, MAC2STR(bss_link->bssid));
236  scanInfo->channel = bss_link->channel;
237  compMsgWifiData.bssScanSizes.channelSize += 1;
238  scanInfo->rssi = bss_link->rssi;
239  compMsgWifiData.bssScanSizes.rssiSize += 1;
240  scanInfo->authmode = bss_link->authmode;
241  compMsgWifiData.bssScanSizes.authmodeSize += 1;
242  scanInfo->is_hidden = bss_link->is_hidden;
243  compMsgWifiData.bssScanSizes.is_hiddenSize += 1;
244  scanInfo->freq_offset = bss_link->freq_offset;
245  compMsgWifiData.bssScanSizes.freq_offsetSize += 2;
246  scanInfo->freqcal_val = bss_link->freqcal_val;
247  compMsgWifiData.bssScanSizes.freqcal_valSize += 2;
248  bss_link = bss_link->next.stqe_next;
249  bssScanInfos.numScanInfos++;
250  }
251  bssScanInfos.scanInfoComplete = true;
252  bssScanInfos.compMsgDispatcher->compMsgBuildMsg->buildMsg(bssScanInfos.compMsgDispatcher);
253 }
